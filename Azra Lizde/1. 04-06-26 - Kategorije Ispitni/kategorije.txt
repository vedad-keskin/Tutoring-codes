string GenerisiSifru(string imePreime, int broj) {
	string sifra;

	time_t sekundeOdUnix = time(nullptr);

	tm trenutnoVrijeme;

	localtime_s(&trenutnoVrijeme, &sekundeOdUnix);


	int godina = trenutnoVrijeme.tm_year + 1900;


	char prvoSlovoImena = toupper(imePreime[0]);


	int indeks = imePreime.find(" ") + 1;



	char prvoSlovoPrezimena = toupper(imePreime[indeks]);


	sifra += prvoSlovoImena;
	sifra += prvoSlovoPrezimena;
	sifra += ":";
	sifra += to_string(godina);
	sifra += "-";
	if (broj > 0 && broj < 10)
	{
		sifra += "00";
	}if (broj > 9 && broj < 100)
	{
		sifra += "0";
	}
	sifra += to_string(broj);
	return sifra;
}




bool ValidirajSifru(string tekst) {
	//return regex_match(tekst, regex("[A-Z]{2}\\:[0-9]{4}\\-00[1-9]{1}"))

	return regex_match(tekst, regex("[A-Z]{2}[:][0-9]{4}[-][0-9]{3}"));

}




template<class T1, class T2, int max>
class KolekcijaParova {
	T1* _prvi[max] = { nullptr };
	T2* _drugi[max] = { nullptr };
	int _trenutno = 0;
public:

	KolekcijaParova() = default;


	~KolekcijaParova() {
		for (int i = 0; i < _trenutno; i++) {
			delete _prvi[i]; _prvi[i] = nullptr;
			delete _drugi[i]; _drugi[i] = nullptr;
		}
	}

	int GetTrenutno() const { return _trenutno; }
	T1& GetPrvi(int indeks) { return *_prvi[indeks]; }
	T2& GetDrugi(int indeks) { return *_drugi[indeks]; }
	T1& GetPrvi(int indeks) const { return *_prvi[indeks]; }
	T2& GetDrugi(int indeks) const { return *_drugi[indeks]; }
	T1& operator[](int indeks) { return *_prvi[indeks]; }


	friend ostream& operator<<(ostream& COUT,const KolekcijaParova& obj) {
		for (int i = 0; i < obj._trenutno; i++)
			COUT << obj.GetPrvi(i) << " " << obj.GetDrugi(i) << "\n";
		return COUT;
	}




	KolekcijaParova(const KolekcijaParova& obj) {

		_trenutno = obj.GetTrenutno();

		for (int i = 0; i < GetTrenutno(); i++)
		{
			_prvi[i] = new T1(obj.GetPrvi(i));
			_drugi[i] = new T2(obj.GetDrugi(i));
		}

	}


	KolekcijaParova& operator=(const KolekcijaParova& obj) {
		if (this == &obj)return *this;
		for (int i = 0; i < _trenutno; i++) {
			delete _prvi[i]; 
			delete _drugi[i];
		}




		_trenutno = obj.GetTrenutno();
		for (int i = 0; i < GetTrenutno(); i++)
		{
			_prvi[i] = new T1(obj.GetPrvi(i));
			_drugi[i] = new T2(obj.GetDrugi(i));
		}




		return *this;
	}

	void Dodaj(T1 el1, T2 el2) {
		if (GetTrenutno() >= max)
		{
			return;
		}
		_prvi[GetTrenutno()] = new T1(el1);
		_drugi[GetTrenutno()] = new T2(el2);
		_trenutno++;
	}




	KolekcijaParova& DodajNaPoziciju(int lokacija, T1 el1, T2 el2) {
		if (lokacija<0 || lokacija>GetTrenutno())
		{
			return *this;
		}
		KolekcijaParova temp;
		for (int i = 0; i < GetTrenutno(); i++)
		{
			if (lokacija == i)
			{
				temp.Dodaj(el1, el2);
			}
			temp.Dodaj(GetPrvi(i), GetDrugi(i));
		}
		*this = temp;
		return *this;
	}




	KolekcijaParova* UkloniRaspon(int start, int brojElemenataZaUkloniti) {
		int end = start + brojElemenataZaUkloniti - 1;
		if (start<0 || end >= GetTrenutno() || start>end)
		{
			throw exception("Nemoguce uklanjanje elemenata iz kolekcije");
		}
		KolekcijaParova* uklonjeni = new KolekcijaParova();
		KolekcijaParova zadrzani;
		for (int i = 0; i < GetTrenutno(); i++)
		{
			if (i >= start && i <= end)
			{
				uklonjeni->Dodaj(GetPrvi(i), GetDrugi(i));
			}
			else
				zadrzani.Dodaj(GetPrvi(i), GetDrugi(i));
		}
		*this = zadrzani;
		return uklonjeni;
	}
};




string Formatiraj(int broj) {
	ostringstream forma;
	forma << setw(2) << setfill('0') << broj;
	return forma.str();
}




class DatumVrijeme {
	int* _godina, * _mjesec, * _dan, * _sati, * _minute, * _sekunde;
public:
	DatumVrijeme(int dan = 1, int mjesec = 1, int godina = 2000, int
		sati = 0, int minute = 0, int sekunde = 0) {
		_godina = new int(godina);
		_mjesec = new int(mjesec);
		_dan = new int(dan);
		_sati = new int(sati);
		_minute = new int(minute);
		_sekunde = new int(sekunde);
	}

	~DatumVrijeme() {
		delete _godina; delete _mjesec; delete _dan;
		_godina = nullptr; _mjesec = nullptr; _dan = nullptr;
		delete _sati; delete _minute; delete _sekunde;
		_sati = nullptr; _minute = nullptr; _sekunde = nullptr;
	}




	DatumVrijeme(const DatumVrijeme& obj) {
		_godina = new int(*obj._godina);
		_mjesec = new int(*obj._mjesec);
		_dan = new int(*obj._dan);
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);
	}
	DatumVrijeme& operator=(const DatumVrijeme& obj) {
		if (this == &obj)return *this;
		delete _godina; delete _mjesec; delete _dan;
		delete _sati; delete _minute; delete _sekunde;




		_godina = new int(*obj._godina);
		_mjesec = new int(*obj._mjesec);
		_dan = new int(*obj._dan);
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);
		return *this;




	}




	friend ostream& operator<<(ostream& COUT, const DatumVrijeme& obj) {
		COUT << Formatiraj(*obj._dan) << "." << Formatiraj(*obj._mjesec) << "." << Formatiraj(*obj._godina) << " " << Formatiraj(*obj._sati) << ":" << Formatiraj(*obj._minute) << ":" << Formatiraj(*obj._sekunde);
		return COUT;
	}




	bool operator==(const DatumVrijeme& obj)const {
		return *_godina == *obj._godina
			&& *_mjesec == *obj._mjesec
			&& *_dan == *obj._dan
			&& *_sati == *obj._sati
			&& *_minute == *obj._minute
			&& *_sekunde == *obj._sekunde;
	}
	bool operator!=(const DatumVrijeme& obj)const { return !(*this == obj); }
};
class Proizvod {
	char* _naziv;
	Kategorija _kategorija;
	int _cijena;
public:
	Proizvod(const char* naziv = "", Kategorija kategorija =
		ELEKTRONIKA, int cijena = 0)
		: _kategorija(kategorija), _cijena(cijena) {
		_naziv = AlocirajTekst(naziv);
	}

	~Proizvod() { 
		delete[] _naziv;
		_naziv = nullptr;
	}


	const char* GetNaziv() const { return _naziv; }
	Kategorija GetKategorija() const { return _kategorija; }
	int GetCijena() const { return _cijena; }


	friend ostream& operator<<(ostream& COUT, const Proizvod& p) {
		COUT << p._naziv << " " << KategorijaNazivi[(int)p._kategorija] << " " << p._cijena;
		return COUT;
	}




	Proizvod(const Proizvod& obj) {
		_naziv = AlocirajTekst(obj.GetNaziv());
		_kategorija = obj.GetKategorija();
		_cijena = obj.GetCijena();
	}


	Proizvod& operator=(const Proizvod& obj) {
		if (this == &obj)return *this;
		delete[] _naziv;
		_naziv = AlocirajTekst(obj.GetNaziv());
		_kategorija = obj.GetKategorija();
		_cijena = obj.GetCijena();
		return *this;
	}




	bool operator==(const Proizvod& obj)const {
		return strcmp(_naziv, obj.GetNaziv()) == 0 &&
			_cijena == obj.GetCijena() &&
			_kategorija == obj.GetKategorija();
	}
	bool operator!=(const Proizvod& obj)const { return !(*this == obj); }
};
class Transakcija {
protected:
	DatumVrijeme _vrijemeRealizacije;
	int _iznos;
public:
	Transakcija(DatumVrijeme vrijemeRealizacije, int iznos = 0)
		: _vrijemeRealizacije(vrijemeRealizacije), _iznos(iznos) {
	}

	virtual ~Transakcija() {}

	virtual string Info() const = 0;

	const DatumVrijeme& GetVrijemeRealizacije() const {
		return
			_vrijemeRealizacije;
	}

	bool operator == (const Transakcija& obj) {
		return _vrijemeRealizacije == obj._vrijemeRealizacije && _iznos == obj._iznos;
	}

	int GetIznos() const { return _iznos; }

	Transakcija(const Transakcija& obj) {
		_vrijemeRealizacije = obj.GetVrijemeRealizacije();
		_iznos = obj.GetIznos();
	}
	Transakcija& operator=(const Transakcija& obj) {
		if (this == &obj)return *this;
		_vrijemeRealizacije = obj.GetVrijemeRealizacije();
		_iznos = obj.GetIznos();
		return *this;
	}
};

class Kupac {
	static int _id;
	char* _sifra;
	char* _imePrezime;
	vector<Transakcija*> _transakcije;
public:
	Kupac(const char* imePrezime) {
		_imePrezime = AlocirajTekst(imePrezime);
		_sifra = AlocirajTekst(GenerisiSifru(imePrezime, _id).c_str());
		_id++;
	}
	~Kupac() {
		delete[] _sifra;
		_sifra = nullptr;
		delete[] _imePrezime;
		_imePrezime = nullptr;
		for (auto* transakcija : _transakcije) {
			delete transakcija;
			transakcija = nullptr;
		}
		_transakcije.clear();
	}
	const char* GetSifra() const { return _sifra; }
	const char* GetImePrezime() const { return _imePrezime; }
	vector<Transakcija*>& GetTransakcije() { return _transakcije; }

	friend ostream& operator<<(ostream& COUT, const Kupac& kupac) {
		COUT << crt << kupac._imePrezime << " [" << kupac._sifra <<
			"]\n";
		for (int i = 0; i < kupac._transakcije.size(); i++)
			COUT << " - " << kupac._transakcije[i]->Info() << "\n";
		COUT << crt;
		return COUT;
	}




	Kupac(const Kupac& obj) {
		_imePrezime = AlocirajTekst(obj.GetImePrezime());
		_sifra = AlocirajTekst(obj.GetSifra());

		for (auto& transakcija : obj._transakcije) // foreach 
		{
			Kupovina* kupovina = dynamic_cast<Kupovina*>(transakcija);

			if (kupovina)
			{
				_transakcije.push_back(new Kupovina(*kupovina));
			}

			Povrat* povrat = dynamic_cast<Povrat*>(transakcija);

			if (povrat)
			{
				_transakcije.push_back(new Povrat(*povrat));
			}
		}
	}
	Kupac& operator=(const Kupac& obj) {
		if (this == &obj)return *this;
		delete[] _sifra;
		delete[] _imePrezime;
		for (auto* transakcija : _transakcije) delete transakcija;
		_transakcije.clear();




		_imePrezime = AlocirajTekst(obj.GetImePrezime());
		_sifra = AlocirajTekst(obj.GetSifra());
		for (auto& transakcija : obj._transakcije)
		{
			Kupovina* kupovina = dynamic_cast<Kupovina*>(transakcija);
			if (kupovina)
			{
				_transakcije.push_back(new Kupovina(*kupovina));
			}




			Povrat* povrat = dynamic_cast<Povrat*>(transakcija);
			if (povrat)
			{
				_transakcije.push_back(new Povrat(*povrat));
			}
		}
		return *this;
	}




	bool operator==(const Kupac& obj)const {
		return strcmp(_imePrezime, obj.GetImePrezime()) == 0 && strcmp(_sifra, obj.GetSifra()) == 0;
	}
	bool operator!=(const Kupac& obj)const { return !(*this == obj); }

class Prodavnica {
	char* _naziv;
	vector<Kupac> _kupci;
public:
	Prodavnica(const char* naziv) { _naziv = AlocirajTekst(naziv); }
	
	~Prodavnica() { 
		delete[] _naziv;
		_naziv = nullptr;
	}

	Prodavnica(const Prodavnica& obj) {
		_naziv = AlocirajTekst(obj._naziv); 
		_kupci = obj._kupci;
	}

	Prodavnica& operator = (const Prodavnica& obj) {
		if (this != &obj) {
			delete[] _naziv;


			_naziv = AlocirajTekst(obj._naziv);
			_kupci = obj._kupci;
		}
		return *this;
	}

	bool operator == (const Prodavnica& obj) {
		return strcmp(_naziv, obj._naziv) == 0;
	}

	const char* GetNaziv() const { return _naziv; }
	vector<Kupac>& GetKupci() { return _kupci; }

	bool RegistrujTransakcijuKupcu(const char* sifraKupca, Kupovina kupovina) {


		for (int i = 0; i < _kupci.size(); i++)
		{

			if (strcmp(sifraKupca, _kupci[i].GetSifra()) == 0) {

				return _kupci[i].DodajTransakciju(kupovina); // false

				//if (_kupci[i].DodajTransakciju(kupovina)) {
				//	return true;
				//}
				//else {
				//	return false;
				//}


			}

		}
		return false;

	}


	void DodajKupca(Kupac kupacZaDodati) {
		for (auto& kupac : _kupci)
		{
			if (kupac == kupacZaDodati)
			{
				throw exception("Kupac je vec dodan");
			}
		}
		_kupci.push_back(kupacZaDodati);
	}

	friend ostream& operator << (ostream& COUT, const Prodavnica& obj) {
		COUT << "Prodavnica: " << obj._naziv << endl;

		for (int i = 0; i < obj._kupci.size(); i++)
		{
			COUT << obj._kupci[i] << endl;
		}

		return COUT;
	}


	////vraca listu svih kupaca iz prodavnice koji su imali najmanje jednu transakciju u proslijedjenoj kategoriji,
////te koliko su ukupno potrosili na proizvode iz te kategorije (ukupna potrosnja za kategoriju se
//// izracunava: kupovina - povrat)
	//KolekcijaParova<Kupac, int, 50> potrosnja = tehnika.PotrosnjaPoKategoriji(ELEKTRONIKA);
	//for (int i = 0; i < potrosnja.GetTrenutno(); i++)
	//	cout << potrosnja.GetPrvi(i).GetImePrezime() << " [" << potrosnja.GetPrvi(i).GetSifra() << "] " << " potrosio/la " << potrosnja.GetDrugi(i) << " KM na proizvode iz kategorije ELEKTRONIKA" << crt;

	KolekcijaParova<Kupac, int, 50> PotrosnjaPoKategoriji(Kategorija kategorija) {


		KolekcijaParova<Kupac, int, 50> potrosnja;

		for (int i = 0; i < _kupci.size(); i++)
		{

			int ukupnaPotrosnja = 0;
			bool imaTransakciju = false;

			for (int j = 0; j < _kupci[i].getKupovine().size(); j++)
			{

				for (int k = 0; k < _kupci[i].getKupovine()[j].GetProizvodi().size(); k++)
				{

					if (_kupci[i].getKupovine()[j].GetProizvodi()[k].GetKategorija() == kategorija) {

						ukupnaPotrosnja += _kupci[i].getKupovine()[j].GetProizvodi()[k].GetCijena();

						imaTransakciju = true;

					}

				}


			}


			for (int j = 0; j < _kupci[i].getPovrati().size(); j++)
			{

				for (int k = 0; k < _kupci[i].getPovrati()[j].GetProizvodi().size(); k++)
				{

					if (_kupci[i].getPovrati()[j].GetProizvodi()[k].GetKategorija() == kategorija) {

						ukupnaPotrosnja -= _kupci[i].getPovrati()[j].GetProizvodi()[k].GetCijena();


					}

				}
			}



			if (imaTransakciju) {

				potrosnja.Dodaj(_kupci[i], ukupnaPotrosnja);

			}


		}

		

		return potrosnja;

	}


};


/*
Funkcija UcitajPodatke ima zadatak ucitati podatke o prodavnicama
   i njihovim kupcima iz fajla cije ime se proslijedjuje kao parametar
(fajl mozete pronaci zajedno sa ispitnim zadatkom). Svaka linija u
   fajlu treba biti u formatu "ime i prezime kupca|naziv prodavnice".
   Funkcija za
svaki red u fajlu:
- unutar vector-a, po nazivu, pronadje ranije dodatu ili
   kreira novu prodavnicu,
- kreira novog kupca ukoliko vec nije registrovan u naznacenoj
   prodavnici,
- dodaje kupca u naznacenu prodavnicu (onemoguciti
   dupliciranje kupaca u istoj prodavnici).
Na kraju, sve prodavnice sa svojim kupcima se trebaju nalaziti u
   proslijedjenom vektoru prodavnice.
Funkcija vraca true ako je ucitavanje podataka bilo uspjesno (u
   vector ucitan najmanje jedan podatak
o prodavnici ili kupcu), a false ako se desilo nesto neocekivano
   ili niti jedan podatak nije ucitan.
Primjer sadrzaja fajla:
Emina Junuz|Tehnika";
Jasmin Azemovic|Tehnika";
Zanin Vejzovic|Knjizara";
*/

bool UcitajPodatke(string fajlPath, vector<Prodavnica>& prodavnice) {

	ifstream fajl(fajlPath);

	if (!fajl.is_open())
	{
		return false;
	}

	string eachRow;

	while (getline(fajl, eachRow)) {

		int pozCrtice = eachRow.find('|');

		string imePrezime = eachRow.substr(0, pozCrtice);
		string nazivProdavnice = eachRow.substr(pozCrtice + 1);

		Prodavnica novaProdavnica(nazivProdavnice.c_str());

		bool prodavnicaExists = false;

		for (int i = 0; i < prodavnice.size(); i++)
		{

			if (novaProdavnica == prodavnice[i]) {

				prodavnicaExists = true;

			}

		}
		if (!prodavnicaExists) {

			prodavnice.push_back(novaProdavnica);

		}

		Kupac noviKupac(imePrezime.c_str());

		bool kupacExists = false;

		for (int i = 0; i < prodavnice.size(); i++)
		{

			//if (strcmp(prodavnice[i].GetNaziv() , nazivProdavnice.c_str()) == 0) {
			if(prodavnice[i] == novaProdavnica) {

				for (int j = 0; j < prodavnice[i].GetKupci().size(); j++)
				{

					if (noviKupac == prodavnice[i].GetKupci()[j]) {


						kupacExists = true;

					}

				}
				if (!kupacExists) {

					prodavnice[i].DodajKupca(noviKupac);

				}

			}

		}




	}


	return true;


}
