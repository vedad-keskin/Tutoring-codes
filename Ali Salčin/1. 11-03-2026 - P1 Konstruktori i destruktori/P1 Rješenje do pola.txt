class Predmet {
public:			   // dft. constr.	//

	//char _naziv[50];
	int _ocjena;   //     5			//
	int* _ECTS;	   //     0			//


	// dft. constr.

		_ocjena = 5;

		_ECTS = new int(0);

		//char tekst1[] = "Nema predmeta";

		const char* tekst3 = "Nema predmeta"; // string literal --> read-only memoriju

		int vel = strlen(tekst3) + 1; // 13 + 1 = 14 -> zbog null terminatora

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, tekst3 ); // 1. gdje pohranjujemo neki tekst , 2. koliko slova kopiramo , 3. sta kopiramo 
	}

	// 99.9% --> user-def constr.

	//Predmet predmet1("Programiranje II", 9, 7);

	Predmet(const char* naziv, int ocjena, int ects ) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);

	}


	//Predmet predmetKopija2(predmet1);	// <-- copy constr.

	// bez & --> saljemo kopiju 
	// sa & --> saljemo original 
	// sa const --> original koji smo poslali se ne moze mijenjati 


	Predmet(const Predmet& predmet) { // napravio kopiju predmeta1

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);


	}

	//predmetOperatorDodjele = predmet1; // operator dodjele

	Predmet& operator = (const Predmet& predmet) {


		if (this != &predmet) {

			delete _ECTS;
			delete[] _naziv;


			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);


		}

		return *this;

	}



	// TILDA alt gr + 1 ~

	~Predmet() {

		cout << "DESTURKTOR SE POZVAO" << endl;

		delete _ECTS;

		_ECTS = nullptr;

		delete[] _naziv;

		_naziv = nullptr;

	}




	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;


	}





};
