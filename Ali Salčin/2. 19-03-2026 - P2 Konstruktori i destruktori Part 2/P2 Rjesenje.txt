// 2. DEREFERENCIRANJE 

// AKO NAM TREBA JEDNA VRIJEDNOST --> ( ) --> _ECTS = 000002DA71C57BD0 , *_ECTS = 0 

// AKO NAM TREBA VISE VRIJEDNOSTI --> [ ] --> *_naziv = N (prvo slovo niza) , _naziv = Nema predmeta

// 3. DESTRUKTOR / DEALOKCAIJA

// AKO NAM TREBA VISE VRIJEDNOSTI --> [ ] -->  		delete[] _naziv;

class Predmet {
public:			   // dft. constr.	//
	char* _naziv;  // Nema predmeta	//
	//char _naziv[50];
	int _ocjena;   //     5			//
	int* _ECTS;	   //     0			//


	// dft. constr.

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char tekst1[] = "Nema predmeta";
		//string tekst2 = "Nema predmeta";
		const char* tekst3 = "Nema predmeta"; // string literal --> read-only memoriju

		int vel = strlen(tekst3) + 1; // 13 + 1 = 14 -> zbog null terminatora

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, tekst3); // 1. gdje pohranjujemo neki tekst , 2. koliko slova kopiramo , 3. sta kopiramo 
	}

	// 99.9% --> user-def constr.

	//Predmet predmet1("Programiranje II", 9, 7);

	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);

	}


	//Predmet predmetKopija2(predmet1);	// <-- copy constr.

	// bez & --> saljemo kopiju 
	// sa & --> saljemo original 
	// sa const --> original koji smo poslali se ne moze mijenjati 


	Predmet(const Predmet& predmet) { // napravio kopiju predmeta1

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);


	}

	//predmetOperatorDodjele = predmet1; // operator dodjele

	Predmet& operator = (const Predmet& predmet) {


		if (this != &predmet) {

			delete _ECTS;
			delete[] _naziv;


			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);


		}

		return *this;

	}



	// TILDA alt gr + 1 ~

	~Predmet() {

		//cout << "DESTURKTOR SE POZVAO" << endl;

		delete _ECTS;

		_ECTS = nullptr;

		delete[] _naziv;

		_naziv = nullptr;

	}




	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;

	}



};





class Student {
	char* _imePrezime; // char [ ] 

	Predmet* _polozeni; // ( ) ili [ ] --> 1000% neki niz 
	int _trenutno;		// nam kaze koliko trenutno imamo elemenata u nizu 
	int _max;			// nam kaze koliko maksimalno mozemo imati elemenata u nizu 
public:


	Student() {

		const char* dftNaziv = "---";

		int vel = strlen(dftNaziv) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, dftNaziv);

		_trenutno = 0;

		_max = 3;

		_polozeni = new Predmet[_max];

	}

	//Student student1("Vedad Keskin" , 3);

	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, imeprezime);

		_trenutno = 0;

		_max = max;

		_polozeni = new Predmet[_max];


	}

	Student(const Student& obj) {

		int vel = strlen(obj._imePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, obj._imePrezime);

		_trenutno = obj._trenutno;

		_max = obj._max;

		_polozeni = new Predmet[_max];

		for (int i = 0; i < _trenutno; i++) // 3 -> 0 1 2 
		{
			_polozeni[i] = obj._polozeni[i];
		}


	}

	Student& operator = (const Student& obj) {
		if (this != &obj) {


			delete[] _imePrezime;
			delete[] _polozeni;


			int vel = strlen(obj._imePrezime) + 1;

			_imePrezime = new char[vel];

			strcpy_s(_imePrezime, vel, obj._imePrezime);

			_trenutno = obj._trenutno;

			_max = obj._max;

			_polozeni = new Predmet[_max];

			for (int i = 0; i < _trenutno; i++) // 3 -> 0 1 2 
			{
				_polozeni[i] = obj._polozeni[i];
			}


		}
		return *this;
	}



	~Student() {

		delete[] _imePrezime;
		_imePrezime = nullptr;


		delete[] _polozeni;
		_polozeni = nullptr;


	}


	void Ispis() {

		cout << "Ime i prezime: " << _imePrezime << endl;

		cout << "=== POLOZENI PREDMETI === " << endl << endl;

		for (int i = 0; i < _trenutno; i++) // 0 1 
		{

			_polozeni[i].Ispis(); 

		}

	}


	//student1.AddPolozeniPredmet(predmet1); // WRD

	void AddPolozeniPredmet(Predmet predmet) {


		// Voditi računa da se ne može prekoračiti kapacitet niza

		if (_max == _trenutno) {

			cout << "Student ne moze poloziti vise predmeta. Kapacitet niza je popunjen." << endl << endl;

			return;

		}


		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		        

		for (int i = 0; i < _trenutno; i++) // 2 --> 0 1 
		{
			//   MATII         == pol[0] = PRII .. pol[1] = MATII 


			//           char* == char* ovo ne radi

			//          strcmp ( naziv1   , naziv2  ) == 0  

			if (strcmp(predmet._naziv, _polozeni[i]._naziv) == 0) {

				cout << "Student je vec polozio odabrani predmet. Duplikati nisu dopusteni." << endl << endl;

				return;

			}


		}



		// Dodaje novi polozeni predmet u niz

		_polozeni[_trenutno] = predmet; // pol[0] = PRII , pol[1] = MATII 

		_trenutno++; // 0 --> 1 --> 2

	}




};
