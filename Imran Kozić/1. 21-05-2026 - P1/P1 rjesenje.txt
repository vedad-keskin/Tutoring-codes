// 2. DEREFERENCIRANJE ( * )

// 3. DEALOKACIJA / DESTRUKTOR

// -- AKO NAM TREBA VISE VRIJEDNOSTI [ ] --> delete[] _naziv



class Predmet {
public:
	char* _naziv;
	//char _naziv[50];
	int _ocjena;
	int* _ECTS;

	// Kreirati i implementirati
	// 1. Defaultni konstruktor (default constructor)
	Predmet() // : _naziv(nullptr), _ocjena(0), _ECTS(nullptr) {
	{
		_ocjena = 0;

		//_ECTS = nullptr;
		_ECTS = new int(0);

		//char dftNaziv[] = "Nema predmeta";
		//string dftNaziv = "Nema predmeta";
		const char* dftNaziv = "Nema predmeta"; // STRING LITERAL --> READ-ONLY MEM.

		int vel = strlen(dftNaziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, dftNaziv);


	};

	 //2. Konstruktor koji prima parametre (user-defined constructor)

	Predmet(const char* naziv, int ocjena, int ECTS) {

		_ocjena = ocjena;

		_ECTS = new int(ECTS);

		int vel = strlen(naziv) + 1;
		_naziv = new char[vel];
		strcpy_s(_naziv, vel, naziv);


		//*_ECTS = ECTS; //CHAT GPT
	}

	// 3. Konstruktor kopije (copy constructor)

	// bez & --> narpavila kopija Predmeta
	// sa & --> posalji original 
	// const + & -> salje origianl sa zastitom 

	Predmet(const Predmet& obj) {

		_ocjena = obj._ocjena;

		int vel = strlen(obj._naziv) + 1;
		_naziv = new char[vel];
		strcpy_s(_naziv, vel, obj._naziv);


		_ECTS = new int(*obj._ECTS);

	}


	// 4. Operator dodjele (operator =)
	Predmet& operator = (const Predmet& obj) {


		if (this != &obj) {

			delete[] _naziv;
			delete _ECTS;

			_ocjena = obj._ocjena;

			int vel = strlen(obj._naziv) + 1;
			_naziv = new char[vel];
			strcpy_s(_naziv, vel, obj._naziv);


			_ECTS = new int(*obj._ECTS);
		}

		return *this;
	}


	//// 5. Dealokacija (destructor)
	~Predmet() {

		//cout << "DESTRUTKOR SE POZVAO" << endl;

		delete[] _naziv;
		_naziv = nullptr;

		delete _ECTS;
		_ECTS = nullptr;
	}
	//// 6. Ispis ili operator ispisa (operator <<) 
	friend ostream& operator<<(ostream& OUT, const Predmet& obj) {
		
		OUT << "Naziv predmeta: " << obj._naziv << endl;
		OUT << "Ocjena predmeta: " << obj._ocjena << endl;
		OUT << "ECTS predmeta: " << *obj._ECTS << endl << endl;

		return OUT;
	}

};

	void AddPolozeniPredmet(Predmet predmet) { // MATII

		// Voditi računa da se ne može prekoračiti kapacitet niza

			cout << "Prekoracen broj elemenata" << endl;
			return;

		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		        

		for (int i = 0; i < _trenutno; i++) // 0 1

			}

		}


		// Dodaje novi polozeni predmet u niz
		_polozeni[_trenutno] = predmet;
		_trenutno++; // 0 ---> 1 ---> 2


	}
