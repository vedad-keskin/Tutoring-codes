char* GetNizKaraktera(const char* sadrzaj) {

	int vel = strlen(sadrzaj) + 1;
	char* novi = new char[vel];
	strcpy_s(novi, vel, sadrzaj);
	return novi;

}

class Predmet {
public:
	char* _naziv;
	int* _ECTS;
	Profesor* _predmetniProfesor; // nema trenjutno ili max ---> ( ) 


	// 2. Konstruktor koji prima parametre (user-defined constructor)
	Predmet(const char* naziv = "---", int ECTS = 0, Profesor predmetniProfesor = Profesor() ) {
		_naziv = GetNizKaraktera(naziv);

		_ECTS = new int(ECTS);

		_predmetniProfesor = new Profesor(predmetniProfesor); // GPT
	}
	// 3. Konstruktor kopije (copy constructor)
	Predmet(const Predmet& obj) {
		_naziv = GetNizKaraktera(obj._naziv);

		_ECTS = new int(*obj._ECTS);

		_predmetniProfesor = new Profesor(*obj._predmetniProfesor);//GPT
	}
	// 4. Operator dodjele (operator =)
	Predmet& operator=(const Predmet& obj) {
		if (this != &obj) {
			delete _predmetniProfesor;
			delete[] _naziv;
			delete _ECTS;

			_naziv = GetNizKaraktera(obj._naziv);
			_ECTS = new int(*obj._ECTS);
			_predmetniProfesor = new Profesor(*obj._predmetniProfesor);
		}
		return *this;
	}
	// 5. Dealokacija (destructor)
	~Predmet() {
		delete _predmetniProfesor; _predmetniProfesor = nullptr; //GPT ali znam sto ide
		delete[] _naziv; _naziv = nullptr;
		delete _ECTS; _ECTS = nullptr;
	}
	// 6. Operator ispisa (operator <<) 
	friend ostream& operator<<(ostream& COUT, const Predmet& obj) {
		COUT << "Naziv predmeta: " << obj._naziv << endl;
		COUT << "ECTS: " << *obj._ECTS << endl;
		COUT << "Predmetni profesor: " << endl << *obj._predmetniProfesor << endl;

		return COUT;
	}

};



class Student : public Osoba { // IZVEDENA KLASA
public:
	char* _brojIndeksa;
	Predmet* _polozeniPredmeti;  // ako imamo trenjutno ili max ---> [ ]  
	int* _trenutno;
	int _max;
	// Kreirati i implementirati



	// 2. Konstruktor koji prima parametre (user-defined constructor)
	Student(const char* brojIndeksa = "IB00000", const char* imePrezime = "---", const char* JMBG ="---", int max = 3) : Osoba(imePrezime, JMBG) {

		_brojIndeksa = GetNizKaraktera(brojIndeksa);
		
		_trenutno = new int(0);
		_max = max;

		_polozeniPredmeti = new Predmet[_max];
		
	}
	// 3. Konstruktor kopije (copy constructor)
	Student(const Student& obj) : Osoba(obj) {
		_brojIndeksa = GetNizKaraktera(obj._brojIndeksa);

		_trenutno = new int(*obj._trenutno);

		_max = obj._max;

		_polozeniPredmeti = new Predmet[_max];

		for (int i = 0; i < *_trenutno; i++) {
			_polozeniPredmeti[i] = obj._polozeniPredmeti[i];
		}
	}
	// 4. Operator dodjele (operator =)
	Student& operator=(const Student& obj) {
		if (this != &obj) {
			delete[] _brojIndeksa;
			delete _trenutno;
			delete[] _polozeniPredmeti;

			Osoba :: operator=(obj);

			_brojIndeksa = GetNizKaraktera(obj._brojIndeksa);
			_trenutno = new int(*obj._trenutno);
			_max = obj._max;

			_polozeniPredmeti = new Predmet[_max];
			for (int i = 0; i < *_trenutno; i++) {
				_polozeniPredmeti[i] = obj._polozeniPredmeti[i];
			}
		}
		return *this;
	}
	// 5. Dealokacija (destructor)
	~Student() {
		delete[] _brojIndeksa; _brojIndeksa = nullptr;
		delete _trenutno; _trenutno = nullptr;
		delete[] _polozeniPredmeti; _polozeniPredmeti = nullptr;
	}

	// 8. Override ciste virtuelne funkcije npr. void Info()

	void Info() override {

		cout << *this;

	}


	// 6. Operator ispisa (operator <<) 
	friend ostream& operator<<(ostream& COUT, const Student& obj) {

			COUT << "Nema polozenih predmeta: " << endl;
		}
		else {
			COUT << "Polozeni predmeti: " << endl;
			for (int i = 0; i < *obj._trenutno; i++) {
				COUT << obj._polozeniPredmeti[i];
			}
		}
		return COUT;
	}
