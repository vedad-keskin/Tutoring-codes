#include <iostream>
using namespace std;




// POKAZIVACI

// 1. ALOKACIJA ( new )

// --> ( ) --> oznacavaju jednu vrijednost 

// --> [ ] --> oznacavaju vise vrijednosti 

// 2. DEREFERENCIRANJE ( * )

// --> ( ) --> oznacavaju jednu vrijednost --> *_ECTS = 0 , _ECTS = 16527638510

// --> [ ] --> oznacavaju vise vrijednosti  --> _naziv = Nema predmeta , *_naziv = N 

// 3. DESTRUKTOR / DEALOKACIJA

// --> ( ) --> oznacavaju jednu vrijednost --> delete _ECTS;

// --> [ ] --> oznacavaju vise vrijednosti  --> delete[] _naziv;


class Predmet {
public:			   // dft. constr.	//
	char* _naziv;  // Nema predmeta	//
	//char _naziv[50];
	int _ocjena;   //      5		//
	int* _ECTS;	   //      0		//


	// dft. constr.

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char tekst1[] = "Nema predmeta"; // 14 
		//string tekst2 = "Nema predmeta";
		const char* tekst3 = "Nema predmeta"; // string literal --> read-only memoriju

		int vel = strlen(tekst3) + 1; // 13 --> null terminator

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, tekst3); // 1. gdje , 2. koliko, 3. sta

	}


	// user-def. constr.
	//Predmet predmet1("Programiranje II", 9, 7);

	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);

	}

	//Predmet predmetKopija2(predmet1); // copy constr. 

	// sa & --> proslijedili original 
	// bez & --> proslijedili kopiju 
	// const --> nesto je kontrantno i ne moze se mijenjati 

	Predmet(const Predmet& predmet) {

		//predmet._ocjena = 10;

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);

	}

	//predmetOperatorDodjele1 = predmet1; // operator dodjele , oprator na znaku = 


	Predmet& operator = (const Predmet& predmet) {


		if (this != &predmet) {

			delete _ECTS;

			delete[] _naziv;


			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);

		}


		return *this;

	}










	// destructor / deacloakcija

	// TILDA  -> ~ alt gr + 1

	~Predmet() {

		//cout << "DESTUKTOR SE POZVAO" << endl;

		delete _ECTS;

		_ECTS = nullptr;

		delete[] _naziv;

		_naziv = nullptr;

	}




	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;

	}




};





class Student {
	char* _imePrezime;  // --- [ ] 

	Predmet* _polozeni; // 1000% [ ] 
	int _trenutno;      // koliko trenutno imamo pohranjenim Predmeta u nizu
	int _max;           // koliko maksimalno mozemo imati Predmeta u nizu
public:


	Student() {

		const char* dftImePrezime = "---";

		int vel = strlen(dftImePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, dftImePrezime);

		_trenutno = 0;

		_max = 3;

		_polozeni = new Predmet[_max];

	}

	//Student student1("Vedad Keskin");
	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, imeprezime);

		_trenutno = 0;

		_max = max;

		_polozeni = new Predmet[_max];

	}

	Student(const Student& obj) {

		int vel = strlen(obj._imePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, obj._imePrezime);

		_trenutno = obj._trenutno;

		_max = obj._max;

		_polozeni = new Predmet[_max];

		for (int i = 0; i < _trenutno; i++) // 3 .. 0 1 2 
		{
			_polozeni[i] = obj._polozeni[i]; // pol[2] = obj.pol[2]
		}

	}

	Student& operator = (const Student& obj) {
		if (this != &obj) {

			delete[] _imePrezime;

			delete[] _polozeni;

			int vel = strlen(obj._imePrezime) + 1;
			_imePrezime = new char[vel];
			strcpy_s(_imePrezime, vel, obj._imePrezime);

			_trenutno = obj._trenutno;

			_max = obj._max;

			_polozeni = new Predmet[_max];

			for (int i = 0; i < _trenutno; i++)
			{
				_polozeni[i] = obj._polozeni[i];
			}
		}
		return *this;
	}


	~Student() {

		delete[] _imePrezime;
		
		_imePrezime = nullptr;

		delete[] _polozeni;

		_polozeni = nullptr; 

	}


	void Ispis() {


		cout << "Ime i prezime studenta: " << _imePrezime << endl;
		cout << "=== POLOZENI PREDMETI ===" << endl << endl;

		for (int i = 0; i < _trenutno; i++) // 0 1
		{
			_polozeni[i].Ispis();
		}

	}












	//student1.AddPolozeniPredmet(predmet1); --> MATII


	void AddPolozeniPredmet(Predmet predmet) {


		// Voditi računa da se ne može prekoračiti kapacitet niza

		//   3   ==  3
		if (_max == _trenutno) {

			cout << "Kapacitet niza je popunjen" << endl;
			return;

		}


		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		


		for (int i = 0; i < _trenutno; i++) // 0 1 , 
		{
			// PRII ili MATII == MATII


			// char* == char*
			// strcmp ( char* , char* ) == 0 


			if (strcmp( _polozeni[i]._naziv , predmet._naziv) == 0) // pol[0] = PRII == MATII , pol[1] = MATII == MATII
			{

				cout << "Duplikati predmeta nisu dozvoljeni" << endl;
				return;

			}

		}





		// Dodaje novi polozeni predmet u niz

		_polozeni[_trenutno] = predmet; // pol[0] = PRII , pol[1] = MATII

		_trenutno++; // 0 --> 1 --> 2


	}

};
