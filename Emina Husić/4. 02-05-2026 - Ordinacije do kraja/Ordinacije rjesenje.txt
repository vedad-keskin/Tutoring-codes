char* AlocirajNiz(const char* sadrzaj, bool dealociraj = false) {
	if (sadrzaj == nullptr)return nullptr;
	int vel = strlen(sadrzaj) + 1;
	char* temp = new char[vel];
	strcpy_s(temp, vel, sadrzaj);
	if (dealociraj)
		delete[]sadrzaj;
	return temp;
}

validnom sifrom treba smatrati i onu koja umjesto znaka crtica '-' ima znak razmak npr: 25 DM 003 ili 25 DM-003

int GetTrenutnaGodinaZadnjeDvijeCifre() {

	time_t trenutnoVrijeme = time(nullptr); // ili 0 

	tm lokalnoVrijeme;

	localtime_s(&lokalnoVrijeme, &trenutnoVrijeme);

	int godina = 1900 + lokalnoVrijeme.tm_year; // 1900 + 126 = 2026

	return godina % 100;

}

char* GenerisiSifru(string imePrezime, int id) {

	//int trenutnaGodina = GetTrenutnaGodinaZadnjeDvijeCifre();
	char PrvoSlovoImena = toupper(imePrezime[0]);
	char PrvoSlovoPrezimena = toupper(imePrezime[imePrezime.find(' ') + 1]);


	string sifra;


	sifra += to_string(GetTrenutnaGodinaZadnjeDvijeCifre()) + '-'; // INT + STRING --> to_string( broj )
	sifra += PrvoSlovoImena;
	sifra += PrvoSlovoPrezimena;
	sifra += '-';

	if (id < 10) {

		sifra += "00";

	}
	else if (id < 100) {

		sifra += "0";
	}
	else {

		sifra += "";

	}

	sifra += to_string(id);


	return AlocirajNiz(sifra.c_str()); // STRING -> CHAR* .c_str()

}

// regex_match --> validira da li je nesto napisano nekim redosljednom postujuci neka pravila 
// regex_search --> validira da li neki skup pravila postoji u nekom tekstu 

// 1. [ ] --> RANGE [a-z] , [A-Z] , [0-9] , [a-zA-Z0-9]  samo jedno slovo iz datog range-a 

// 2. { } --> VELICINA  {3} -> tacno 3 puta , {3,7} od 3 do 7 , {3,} 3 ili vise , {,5} do 5 karaktera , { } se vezuju za ono sto je napisano ispred njih 

// 3. ( ) --> TACAN TEKST , (gmail) tacno gmail , (gmail|yahoo) 

// vedad.keskin@edu.fit.ba --> [a-z]+[.][a-z]+@(edu)[.](fit)[.](ba|com)  ,,, . znaci bilo koji karakter


bool ValidirajSifru(const char* sifra) {

	regex reg("[0-9]{2}[- ][A-Z]{2}[- ][0-9]{3}");

	return regex_match(sifra, reg);

}







template<class T1, class T2, int max>
class KolekcijaG1 {
	T1* _elementi1[max];
	T2* _elementi2[max];
	int _trenutno;
public:

	KolekcijaG1() {

		_trenutno = 0;

		// ovdje ne moramo raditi nista

		//_elementi1 = nullptr;

	}

	KolekcijaG1(const KolekcijaG1& obj) {

		_trenutno = obj._trenutno;

		for (int i = 0; i < _trenutno; i++)
		{

			_elementi1[i] = new T1(*obj._elementi1[i]);
			_elementi2[i] = new T2(*obj._elementi2[i]);

		}

	}

	KolekcijaG1& operator = (const KolekcijaG1& obj) {
		if (this != &obj) {


			for (int i = 0; i < _trenutno; i++)
			{

				delete _elementi1[i];
				delete _elementi2[i];

			}

			_trenutno = obj._trenutno;

			for (int i = 0; i < _trenutno; i++)
			{

				_elementi1[i] = new T1(*obj._elementi1[i]);
				_elementi2[i] = new T2(*obj._elementi2[i]);

			}



		}
		return *this;
	}

	void AddElement(T1 el1, T2 el2) {

		//if (max == _trenutno) {
		//	return;
		//}



		_elementi1[_trenutno] = new T1(el1);
		_elementi2[_trenutno] = new T2(el2);

		_trenutno++;

	}

	/* metoda InsertAt treba da doda vrijednosti drugog i treceg parametra na lokaciju koja je definisana prvim		parametrom. Povratna vrijednost metode je objekat (pozivaoc metode, u konkretnom slucaju objekat kolekcija1) u	okviru koga su, na definisanu lokaciju, dodati proslijedjeni parametri. 	KolekcijaG1 InsertAt(int lokacija, T1 el1, T2 el2) {


		KolekcijaG1 kolekcija2;

		for (int i = 0; i < _trenutno; i++)
		{

			if (lokacija == i) {

				kolekcija2.AddElement(el1, el2);

			}
			kolekcija2.AddElement(*_elementi1[i], *_elementi2[i]);

		}

		*this = kolekcija2;

		return kolekcija2;

	}

	KolekcijaG1* RemoveRange(int lokacijaOd, int brojElemenata) {

		if (lokacijaOd < 0 || lokacijaOd + brojElemenata > _trenutno) {

			throw exception("Opseg ne postoji");

		}



		KolekcijaG1* kolekcija3 = new KolekcijaG1();

		KolekcijaG1 kolekcija1;


		for (int i = 0; i < _trenutno; i++)
		{

			if (i >= lokacijaOd && i < lokacijaOd + brojElemenata) {

				kolekcija3->AddElement(*_elementi1[i], *_elementi2[i]);

			}
			else {

				kolekcija1.AddElement(*_elementi1[i], *_elementi2[i]);

			}


		}

		return kolekcija3;

	}



	~KolekcijaG1() {

			delete _elementi1[i];
			_elementi1[i] = nullptr;

			delete _elementi2[i];

class Termin {
	int* _sati; int* _minute; int* _sekunde;
public:
	Termin(int s = 0, int m = 0, int k = 0) {
		_sati = new int(s);
		_minute = new int(m);
		_sekunde = new int(k);
	}

	Termin(const Termin& obj) {

		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);

	}

	Termin& operator = (const Termin& obj) {
		if (this != &obj) {

			delete _sati; delete _minute; delete _sekunde;

			_sati = new int(*obj._sati);
			_minute = new int(*obj._minute);
			_sekunde = new int(*obj._sekunde);
		}
		return *this;
	}

	bool operator == (const Termin& obj) {

		return *_sati == *obj._sati && *_minute == *obj._minute && *_sekunde == *obj._sekunde;

	}

	int toSeconds() {

		return *_sekunde + *_minute * 60 + *_sati * 3600;

	}

	~Termin() {
		delete _sati; delete _minute; delete _sekunde;
		_sati = nullptr; _minute = nullptr; _sekunde = nullptr;
	}

	int getSati() const { return *_sati; }
	int getMinute() const { return *_minute; }
	int getSekunde() const { return *_sekunde; }

	friend ostream& operator<<(ostream& COUT, const Termin& t) {
		COUT << *t._sati << ":" << *t._minute << ":" << *t._sekunde;
		return COUT;
	}
};

class Ucesnik {

	// 2. KOPIRANJE --> ( BAZNA KOJA POKAZUJE NA IZVEDENA -> BAZNA KOJA POKAZUJE NA IZVEDENA ) -> KLONIRANJE

	// 3. POHRANA --> KLASICNO DODAVANJE

		_sifra = GenerisiSifru(imePrezime, _id++);

		// ako vector nije pok onda ovdje ne radimo nista

	}

	Ucesnik(const Ucesnik& obj) {

		_imePrezime = AlocirajNiz(obj._imePrezime);
		_sifra = AlocirajNiz(obj._sifra);

		//trenutno = obj.ttrenutno;
		//_prijavljeni.size() = obj._prijavljeni.size();

		for (int i = 0; i < obj._prijavljeni.size(); i++)
		{

			_prijavljeni.push_back(obj._prijavljeni[i]->Clone());

		}


	}

	Ucesnik& operator = (const Ucesnik& obj) {
		if (this != &obj) {

			delete[] _imePrezime;
			delete[] _sifra;

			for (auto prijava : _prijavljeni)
			{
				delete prijava;
			}
			_prijavljeni.clear();


			_imePrezime = AlocirajNiz(obj._imePrezime);
			_sifra = AlocirajNiz(obj._sifra);

			//trenutno = obj.ttrenutno;
			//_prijavljeni.size() = obj._prijavljeni.size();

			for (int i = 0; i < obj._prijavljeni.size(); i++)
			{

				_prijavljeni.push_back(obj._prijavljeni[i]->Clone());

			}
		}
		return *this;
	}


	bool PrijaviSe(Predavanje predavanje) {




		for (int i = 0; i < _prijavljeni.size(); i++)
		{

			Predavanje* castedPredavanje = dynamic_cast<Predavanje*>(_prijavljeni[i]); // *DOGADJAJ -> *PREDAVANJE

			// PREDAVANJE == *DOGADJAJ -> *PREDAVANJE

			//  PREDAVANJE == PREDAVANJE
			if (predavanje == *castedPredavanje) { // IZVEDENA == IZVEDENA 

				return false;

			}


		}


		_prijavljeni.push_back(new Predavanje(predavanje));
		//_prijavljeni.push_back(predavanje.Clone());

		return true;

	}

	bool operator == (const Ucesnik& obj) {
		return strcmp(_sifra, obj._sifra) == 0;
	}


	~Ucesnik() {
		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _sifra;
		_sifra = nullptr;

		for (auto prijava : _prijavljeni)
		{
			delete prijava;
			prijava = nullptr;
		}

		_prijavljeni.clear();
	}

	const char* getSifra() const { return _sifra; }
	const char* getImePrezime() const { return _imePrezime; }
	vector<Dogadjaj*>& getDogadjaji() { return _prijavljeni; }

	friend ostream& operator << (ostream& COUT, const Ucesnik& obj) {

		COUT << obj._imePrezime << " [" << obj._sifra << "]" << endl;

		for (int i = 0; i < obj._prijavljeni.size(); i++)
		{
			//COUT << "\t" << *obj._prijavljeni[i];
			COUT << "\t\t\t" << obj._prijavljeni[i]->Info();
		}

		return COUT;
	}

};

		if (this != &obj) {

			delete[] _naziv;

			_clanovi = obj._clanovi;
		}
		return *this;
	}

		return strcmp(_naziv, obj._naziv) == 0;

	}

	void DodajUcesnika(Ucesnik ucesnik) {

		for (int i = 0; i < _clanovi.size(); i++)
		{

			if (ucesnik == _clanovi[i]) {


				throw exception("Ucesnik je vec u timu");

			}

		}



		_clanovi.push_back(ucesnik);

	}

	~Tim() {
		delete[] _naziv;
		_naziv = nullptr;
	}

	char* getNaziv() const { return _naziv; }	
	vector<Ucesnik>& getUcesnici() { return _clanovi; }

	friend ostream& operator << (ostream& COUT, const Tim& obj) {

		COUT << "Naziv tima: "<< obj._naziv << endl;

		for (int i = 0; i < obj._clanovi.size(); i++)
		{
			COUT << "\t\t" << obj._clanovi[i] << endl;
		}


		return COUT;
	}

};

	Konferencija(const char* naziv)
	{
		_naziv = AlocirajNiz(naziv);

	}

	Konferencija(const Konferencija& obj) {

		_naziv = AlocirajNiz(obj._naziv);

		_timovi = obj._timovi;

	}

	Konferencija& operator = (const Konferencija& obj) {
		if (this != &obj) {
			delete[] _naziv;

			_naziv = AlocirajNiz(obj._naziv);

			_timovi = obj._timovi;

		}
		return *this;
	}

	void DodajTimove(Tim tim1, Tim tim2) {




		_timovi.AddElement(new Tim(tim1), new Tim(tim2)); // 1. razlika

	}

	bool PrijaviDogadjaj(const char* nazivTima, const char* sifraUcesnika, Predavanje predavanje) {

		// proci kroz timove

		for (int i = 0; i < _timovi.getTrenutno(); i++)
		{
			
			// ako smo pronasli tim1 

			if (strcmp(nazivTima , _timovi.getElement1(i)->getNaziv() ) == 0) {

				// tim1
				// proci kroz ucesnike tima1

				for (int j = 0; j < _timovi.getElement1(i)->getUcesnici().size(); j++)
				{

					// naci ucesnika cija je sifra proslijedjena kao parametar tima1

					if (strcmp(sifraUcesnika, _timovi.getElement1(i)->getUcesnici()[j].getSifra()) == 0) {

						// dodaje predavanje ucesniku ciju sifru smo pronasli

						return _timovi.getElement1(i)->getUcesnici()[j].PrijaviSe(predavanje); // false ili true u zavisnosti od tacnosti dodavanja


					}


				}
				return false; // ako tim postoji ali ne postoji ucesnik tog tima (opcionlano)

			}

			
			// ako smo pronasli tim2

			if (strcmp(nazivTima, _timovi.getElement2(i)->getNaziv()) == 0) {

					// tim2

					// proci kroz ucesnike tima2

					for (int j = 0; j < _timovi.getElement2(i)->getUcesnici().size(); j++)
					{

						// naci ucesnika cija je sifra proslijedjena kao parametar tima2

						if (strcmp(sifraUcesnika, _timovi.getElement2(i)->getUcesnici()[j].getSifra()) == 0) {

							// dodaje predavanje ucesniku ciju sifru smo pronasli

							return _timovi.getElement2(i)->getUcesnici()[j].PrijaviSe(predavanje);


						}


					}
					return false; // ako tim postoji ali ne postoji ucesnik tog tima (opcionlano)

			}




		}
		return false;

	}

	//metoda vraca sve ucesnike koji su se na odredjenoj konferenciji prijavili na minimalno proslijedjeni broj predavanja

	vector<Ucesnik*> operator () (int brojPredavanja) {

		vector<Ucesnik*> vrijedniUcesnici;

		// proci kroz sve timove konferencije

		for (int i = 0; i < _timovi.getTrenutno(); i++)
		{

			// proci kroz sve ucesnike tima1

			for (int j = 0; j < _timovi.getElement1(i)->getUcesnici().size(); j++)
			{

				// preko size() mozemo doci do velicine vectora (broj predavanja nekog korisnika) i odma na taj nacin mozemo raditi poredjenje

				if (_timovi.getElement1(i)->getUcesnici()[j].getDogadjaji().size() >= brojPredavanja) {


					vrijedniUcesnici.push_back(new Ucesnik(_timovi.getElement1(i)->getUcesnici()[j]));

				}


			}


			// proci kroz sve ucesnike tima2

			for (int j = 0; j < _timovi.getElement2(i)->getUcesnici().size(); j++)
			{

				// preko size() mozemo doci do velicine vectora (broj predavanja nekog korisnika) i odma na taj nacin mozemo raditi poredjenje

				if (_timovi.getElement2(i)->getUcesnici()[j].getDogadjaji().size() >= brojPredavanja) {


					vrijedniUcesnici.push_back(new Ucesnik(_timovi.getElement2(i)->getUcesnici()[j]));

				}


			}

		}

		return vrijedniUcesnici;


	}




	~Konferencija() {
		delete[] _naziv; _naziv = nullptr;
	}

	char* getNaziv() const { return _naziv; }
	KolekcijaG1<Tim*, Tim*, 20>& getTimovi() { return _timovi; }

	friend ostream& operator << (ostream& COUT, const Konferencija& obj) {

		for (int i = 0; i < obj._timovi.getTrenutno(); i++)

		}

		return COUT;
	}

bool UcitajUcesnike(const char* nazivFajla, vector<Tim>& timoviIzFajla) {

	// ofstream
	// ifstream

	ifstream file(nazivFajla);

	string eachRow;

	while (getline(file, eachRow)) {

		int pozCrtice = eachRow.find('|');

		string imePrezime = eachRow.substr(0, pozCrtice); // Goran Skondric|Tim Alpha  --> Goran Skondric

		string nazivTima = eachRow.substr(pozCrtice + 1); // Goran Skondric|Tim Alpha --> Tim Alpha --> idemo do pozicije | + 1 do kraja reda

		Tim noviTim(nazivTima.c_str());

		bool timExists = false;

		for (int i = 0; i < timoviIzFajla.size(); i++)
		{
			if (noviTim == timoviIzFajla[i]) {

				timExists = true;

			}

		}
	
		if (!timExists) {

			timoviIzFajla.push_back(noviTim);

		}

		Ucesnik noviUcesnik(imePrezime.c_str());

		bool ucesnikExists = false;


		// prolazimo kroz sve timovi
		for (int i = 0; i < timoviIzFajla.size(); i++)
		{

			if (timoviIzFajla[i] == noviTim) {


				for (int j = 0; j < timoviIzFajla[i].getUcesnici().size(); j++)
				{

					if (timoviIzFajla[i].getUcesnici()[j] == noviUcesnik) {

						ucesnikExists = true;

					}

				}

				if (!ucesnikExists) {

					timoviIzFajla[i].DodajUcesnika(noviUcesnik);

				}

			}

			

		}


	}

	return true;

}
