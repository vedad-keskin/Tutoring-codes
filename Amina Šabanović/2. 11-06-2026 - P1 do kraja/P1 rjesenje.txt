// POKAZIVACI

// 1. ALOKACIJA ( new ) 

// AKO PRAVIMO JEDNU VRIJEDNOST --> ( ) 

// AKO PRAVIMO VISE VRIJEDNOSTI --> [ ]

// 2. DEREFERENCIRANJE ( * ) prilikom ispisa i kopiranja

// AKO PRAVIMO JEDNU VRIJEDNOST --> ( )  _ECTS = 000002090BF08360 , *_ECTS = 0

// AKO PRAVIMO VISE VRIJEDNOSTI --> [ ]  *_naziv = N , _naziv = Nema predmeta

// 3. DESTRUKTOR / DEALOKACIJA 

// AKO PRAVIMO JEDNU VRIJEDNOST --> ( )  delete _ECTS;

// AKO PRAVIMO VISE VRIJEDNOSTI --> [ ]  delete[] _naziv;


class Predmet {
public:			  // dft. constr.  // user-def constr. //
	char* _naziv; // Nema predmeta // Programiranje II //
	//char _naziv[50];								   //
	int _ocjena;  //      5		   //        9 		   //
	int* _ECTS;	  //      0		   //        7		   //


	// dft. constr.

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char dftNaziv[] = "Nema predmeta"; // staticki niz char --> 14 char
		//string dftNaziv = "Nema predmeta";
		const char* dftNaziv = "Nema predmeta"; // STRING LITERAL --> read-only mem. 

		int vel = strlen(dftNaziv) + 1; // 13 + 1 = 14 ..  null terminator ->  \0

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, dftNaziv);

	}

	// user-def constr.

	//Predmet predmet1("Programiranje II", 9, 7); // user-def constr. (naziv, ocjena, ects)


	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv); // 1. GDJE , 2. KOLIKO , 3. STA

	}


	//Predmet predmetKopija2 ( predmet2 ); // copy constr. --> SHALLOW COPY --> DEEP COPY 

	// bez & --> napravili kopiju tog predmeta
	// sa & --> saljemo original 
	// const --> nesto je konstantno -> ne moze se mijenjati

	Predmet(const Predmet& predmet) {

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);

	}

	//predmetOperatorDodjele = predmet2; // operator = 

	Predmet& operator = (const Predmet& predmet) {

		// adresa lijevog predmeta != adresa desnog predmeta
		if (this != &predmet) {

			delete _ECTS;

			delete[] _naziv;

			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);
		}


		return *this;

	}


	// TILDA -> alt gr + 1

	~Predmet() {

		//cout << "DESTRUKTOR SE POZVAO" << endl;

		delete _ECTS;

		_ECTS = nullptr;

		delete[] _naziv;

		_naziv = nullptr;

	}


	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl << endl;

	}




};


class Student {
	char* _imePrezime; // niz char

	Predmet* _polozeni; // [ ] 
	int _trenutno;      // koliko trenutno imamo pol predmeta
	int _max;           // koliko maksimalno mozemo imati polozenih predmeta
public:

	Student() {

		const char* dftNaziv = "---";

		int vel = strlen(dftNaziv) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, dftNaziv);

		_trenutno = 0;

		_max = 3;

		_polozeni = new Predmet[_max];


	}


	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, imeprezime);

		_trenutno = 0;
		_max = max;

		_polozeni = new Predmet[_max];

	}

	Student(const Student& obj) {

		int vel = strlen(obj._imePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, obj._imePrezime);

		_trenutno = obj._trenutno;
		_max = obj._max;

		_polozeni = new Predmet[_max];

		for (int i = 0; i < _trenutno; i++) // 0 1 2 
		{
			_polozeni[i] = obj._polozeni[i];
		}


	}
	

	Student& operator = (const Student& obj) {
		if (this != &obj) {


			delete[] _imePrezime;
			delete[] _polozeni;


			int vel = strlen(obj._imePrezime) + 1;
			_imePrezime = new char[vel];
			strcpy_s(_imePrezime, vel, obj._imePrezime);

			_trenutno = obj._trenutno;
			_max = obj._max;

			_polozeni = new Predmet[_max];

			for (int i = 0; i < _trenutno; i++) // 0 1 2 
			{
				_polozeni[i] = obj._polozeni[i];
			}
		}
		return *this;
	}


	~Student() {

		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _polozeni;
		_polozeni = nullptr;

	}


	void Ispis() {

		cout << "Ime i prezime studenta: " << _imePrezime << endl;

		cout << "===== POLOZENI PREDMETI STUDENTA =====" << endl;

		for (int i = 0; i < _trenutno; i++) // 0 1 
		{
			_polozeni[i].Ispis(); 
		}


	}









	//student.AddPolozeniPredmet(predmet1); // MATII

	void AddPolozeniPredmet(Predmet predmet) {


		// Voditi računa da se ne može prekoračiti kapacitet niza

		if (_max == _trenutno) {

			cout << "Prekoracen niz" << endl;
			return;

		}


		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		        

		for (int i = 0; i < _trenutno; i++)
		{

			// char == char

			// strcmp ( naziv1 , naziv2 ) == 0

			if (strcmp( predmet._naziv , _polozeni[i]._naziv ) == 0) {

				cout << "Predmet je vec dodan" << endl;
				return;

			}

		}




		// Dodaje novi polozeni predmet u niz

		_polozeni[_trenutno] = predmet; // pol[0] = PRII , pol[1] = MATII

		_trenutno++; // 0 --> 1 --> 2


	}

};



void main() {

	Predmet predmet1("Programiranje II", 9, 7); // user-def constr. (naziv, ocjena, ects)
	Predmet predmet2("Matematika II", 6, 4); // user-def constr. (naziv, ocjena, ects)
	Predmet predmet3("Operativni sistemi", 7, 4); // user-def constr. (naziv, ocjena, ects)
	Predmet predmet4("Web razvoj i dizajn", 7, 5); // user-def constr. (naziv, ocjena, ects)


	Student student("Vedad Keskin", 3);

	student.AddPolozeniPredmet(predmet1); // PRII
	student.AddPolozeniPredmet(predmet2); // PRII
	student.AddPolozeniPredmet(predmet2); // PRII
	student.AddPolozeniPredmet(predmet3); // PRII
	student.AddPolozeniPredmet(predmet4); // PRII


	Student studentKopija = student; // copy constr. 



	student.Ispis();
	studentKopija.Ispis();




	system("pause>0");
}
