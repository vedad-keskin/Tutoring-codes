class Predmet {
public:			  // DFT. CONSTR.  // USER-DEF CONSTR.
	char* _naziv; // Nema predmeta //
	//char _naziv[50];
	int _ocjena;  //      5		   //
	int* _ECTS;	  //      0 	   //

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char dftNaziv1[] = "Nema predmeta"; // 14 
		//string dftNaziv2 = "Nema predmeta";
		const char* dftNaziv3 = "Nema predmeta"; // STRING LITERAL --> read-only memorija 

		int vel = strlen(dftNaziv3) + 1; // 13 null terminator \0 

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, dftNaziv3);

	}


	//Predmet predmet1("Programiranje II", 9, 7); // user-def constr. (naziv, ocjena, ects)
	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);


	}

	//Predmet predmetKopija2(predmet1);

	// bez & --> dobijes copy 
	// sa &  --> original 
	// const --> varijabla je konstantna tj nije promijenjljiva

	Predmet(const Predmet& predmet) {

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);

	}
	//predmetOperatorDodjele = predmet1; // operator = 
	Predmet& operator = (const Predmet& predmet) {

		if (this != &predmet) {

			delete _ECTS;

			delete[] _naziv;

			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);
		}
		return *this;
	}







	// TILDA ~ alt gr + 1
	~Predmet() {

		//cout << "DESTRUKTOR SE POZVAO" << endl;

		delete _ECTS;
		_ECTS = nullptr;

		delete[] _naziv;
		_naziv = nullptr;

	}


	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;

	}




};

class Student {
	char* _imePrezime;

	Predmet* _polozeni; // ( ) ili [ ] <-- 1000% 
	int _trenutno;		// koliko trenutno imamo polozenih predmeta
	//int _max;			// koliko maksimalno mozemoi imati predmeta
public:


	Student() {

		const char* dftImePrezime = "Nema imena i prezimena";

		int vel = strlen(dftImePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, dftImePrezime);

		_trenutno = 0;

		_polozeni = new Predmet[_trenutno];


	}

	Student(const char* imePrezime) {

		int vel = strlen(imePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, imePrezime);

		_trenutno = 0;

		_polozeni = new Predmet[_trenutno];

	}

	Student(const Student& student) {

		int vel = strlen(student._imePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, student._imePrezime);

		_trenutno = student._trenutno;

		_polozeni = new Predmet[_trenutno];


		for (int i = 0; i < _trenutno; i++) // 0 1 2 3
		{
			_polozeni[i] = student._polozeni[i];
		}


	}

	Student& operator = (const Student& student) {
		if (this != &student) {


			delete[] _imePrezime;
			delete[] _polozeni;


			int vel = strlen(student._imePrezime) + 1;
			_imePrezime = new char[vel];
			strcpy_s(_imePrezime, vel, student._imePrezime);

			_trenutno = student._trenutno;

			_polozeni = new Predmet[_trenutno];


			for (int i = 0; i < _trenutno; i++) // 0 1 2 3
			{
				_polozeni[i] = student._polozeni[i];
			}
		}
		return *this;
	}

	~Student() {

		//cout << "STUDENT DESTRUKTOR" << endl;

		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _polozeni;
		_polozeni = nullptr;

	}

	void Ispis() {

		cout << "Ime i prezime studenta: " << _imePrezime << endl;
		cout << "POLOZENI PREDMETI STUDENTA: " << endl << endl;

		for (int i = 0; i < _trenutno; i++) // 0 1 
		{
			_polozeni[i].Ispis();
			cout << "====================" << endl << endl;
		}



	}

	//student.AddPolozeniPredmet(predmet1); <-- MATII DRUGI PUT


	void AddPolozeniPredmet(Predmet predmet) {

		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)	


		// strcmp ( naziv1 , naziv2 ) == 0 


		for (int i = 0; i < _trenutno; i++) // 0 1 
		{

			if (strcmp(_polozeni[i]._naziv, predmet._naziv) == 0) {

				cout << "Predmet " << _polozeni[i]._naziv << " postoji" << endl;
				return;

			}
		}

		// 1. napraviti temp koji ima staro stanje

		Predmet* temp = _polozeni;

		// 2. prosiriti pravi niz da ima jendo dodatno mjesto

		_polozeni = new Predmet[_trenutno + 1]; // kada alociramo niz taj niz ce uvijek biti prazan

		// 3. vratiti stare elemenete iz tempa nazad u nas niz polozeni

		for (int i = 0; i < _trenutno; i++)
		{
			_polozeni[i] = temp[i];
		}

		// 4. Klasicno dodavanje
		_polozeni[_trenutno] = predmet;

		_trenutno++; // 0 --> 1 --> 2

		// 5. Osloboditi temp (opcionalno)

		delete[] temp;
		temp = nullptr;


	}

};
