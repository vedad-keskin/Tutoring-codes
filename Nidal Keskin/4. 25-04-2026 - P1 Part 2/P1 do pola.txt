class Predmet {
public:			  // DFT. CONSTR.  // USER-DEF CONSTR.
	char* _naziv; // Nema predmeta //
	//char _naziv[50];
	int _ocjena;  //      5		   //
	int* _ECTS;	  //      0 	   //

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char dftNaziv1[] = "Nema predmeta"; // 14 
		//string dftNaziv2 = "Nema predmeta";
		const char* dftNaziv3 = "Nema predmeta"; // STRING LITERAL --> read-only memorija 

		int vel = strlen(dftNaziv3) + 1; // 13 null terminator \0 

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, dftNaziv3);

	}


	//Predmet predmet1("Programiranje II", 9, 7); // user-def constr. (naziv, ocjena, ects)
	Predmet(const char* naziv, int ocjena, int ects ) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);


	}

	//Predmet predmetKopija2(predmet1);

	// bez & --> dobijes copy 
	// sa &  --> original 
	// const --> varijabla je konstantna tj nije promijenjljiva

	Predmet(const Predmet& predmet) {

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);

	}
	//predmetOperatorDodjele = predmet1; // operator = 
	Predmet& operator = (const Predmet& predmet) {

		if (this != &predmet) {

			delete _ECTS;

			delete[] _naziv;

			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);
		}
		return *this;
	}

	





	// TILDA ~ alt gr + 1
	~Predmet() {

		cout << "DESTRUKTOR SE POZVAO" << endl;

		delete _ECTS;
		_ECTS = nullptr;

		delete[] _naziv;
		_naziv = nullptr;

	}


	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;

	}




};
