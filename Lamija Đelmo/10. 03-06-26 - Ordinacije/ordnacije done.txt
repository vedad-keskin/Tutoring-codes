char* AlocirajNiz(const char* sadrzaj, bool dealociraj = false) {
	if (sadrzaj == nullptr)return nullptr;
	int vel = strlen(sadrzaj) + 1;
	char* temp = new char[vel];
	strcpy_s(temp, vel, sadrzaj);
	if (dealociraj)
		delete[]sadrzaj;
	return temp;
}
int godina() {
	time_t sistemsko = time(nullptr);
	tm lokalno;
	localtime_s(&lokalno, &sistemsko);
	int god = 1900 + lokalno.tm_year;
	return god % 100;
}
char* GenerisiSifru(string imeprezime, int id) {
	string sifra;
	sifra += to_string(godina());
	sifra += "-";
	sifra += toupper(imeprezime[0]);
	sifra += toupper(imeprezime[imeprezime.find(" ") + 1]);
	sifra += "-";
	if (id < 10) sifra += "00";
	else if (id < 100)sifra += "0";
	else sifra += "";
	sifra += to_string(id);
	return AlocirajNiz(sifra.c_str());
}
bool ValidirajSifru(const char* sifra) {
	regex lozinka("[0-9]{2}[- ][A-Z]{2}[- ][0-9]{3}");
	return regex_match(sifra, lozinka);
}
template<class T1, class T2, int max>
class KolekcijaG1 {
	T1* _elementi1[max];
	T2* _elementi2[max];
	int _trenutno;
public:
	KolekcijaG1() {
		_trenutno = 0;
	}
	KolekcijaG1(const KolekcijaG1& obj) {
		_trenutno = obj._trenutno;
		for (int i = 0; i < _trenutno; i++)
		{
			_elementi1[i] = new T1(*obj._elementi1[i]);
			_elementi2[i] = new T2(*obj._elementi2[i]);
		}
	}
	KolekcijaG1& operator = (const KolekcijaG1& obj) {
		if (this != &obj) {
			for (int i = 0; i < _trenutno; i++)
			{
				delete _elementi1[i];
				delete _elementi2[i];
			}

			_trenutno = obj._trenutno;
			for (int i = 0; i < _trenutno; i++)
			{
				_elementi1[i] = new T1(*obj._elementi1[i]);
				_elementi2[i] = new T2(*obj._elementi2[i]);
			}
		}
		return *this;
	}
	void AddElement(T1 el1, T2 el2) {
		if (_trenutno == max) return;

		_elementi1[_trenutno] = new T1(el1);
		_elementi2[_trenutno] = new T2(el2);
		_trenutno++;
	}
	KolekcijaG1& InsertAt(int lokacija, T1 el1, T2 el2) {
		KolekcijaG1 kolekcija2;
		for (int i = 0; i < _trenutno; i++)
		{
			if (lokacija == i) {
				kolekcija2.AddElement(el1, el2);
			}
			kolekcija2.AddElement(*_elementi1[i], *_elementi2[i]);
		}
		*this = kolekcija2;
		return *this;
	}
	KolekcijaG1* RemoveRange(int odakle, int brojel) {
		if (odakle + brojel > _trenutno) {
			throw exception("Broj elemenata ne postoji");
		}
		KolekcijaG1* kolekcija3 = new KolekcijaG1();
		KolekcijaG1 kolekccija1;
		for (int i = 0; i < _trenutno; i++)
		{
			if (i >= odakle && i < odakle + brojel) {
				kolekcija3->AddElement(*_elementi1[i], *_elementi2[i]);
			}
			else kolekccija1.AddElement(*_elementi1[i], *_elementi2[i]);
		}
		*this = kolekccija1;
		return kolekcija3;
	}
	~KolekcijaG1() {
		for (int i = 0; i < _trenutno; i++)
		{
			delete _elementi1[i];
			_elementi1[i] = nullptr;
			delete _elementi2[i];
			_elementi2[i] = nullptr;
		}
	}
	T1& getElement1(int lokacija)const { return *_elementi1[lokacija]; }
	T2& getElement2(int lokacija)const { return *_elementi2[lokacija]; }
	int getTrenutno() const { return _trenutno; }

	friend ostream& operator<< (ostream& COUT, KolekcijaG1& obj) {
		for (size_t i = 0; i < obj._trenutno; i++)
			COUT << obj.getElement1(i) << " " << obj.getElement2(i) << endl;
		return COUT;
	}
};
class Termin {
	int* _sati; int* _minute; int* _sekunde;
public:
	Termin(int s = 0, int m = 0, int k = 0) {
		_sati = new int(s);
		_minute = new int(m);
		_sekunde = new int(k);
	}
	Termin(const Termin& obj) {
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);
	}
	Termin& operator = (const Termin& obj) {
		if (this != &obj) {
			delete _sati; delete _minute; delete _sekunde;

			_sati = new int(*obj._sati);
			_minute = new int(*obj._minute);
			_sekunde = new int(*obj._sekunde);
		}
		return *this;
	}
	~Termin() {
		delete _sati; delete _minute; delete _sekunde;
		_sati = nullptr; _minute = nullptr; _sekunde = nullptr;
	}
	int getSati() const { return *_sati; }
	int getMinute() const { return *_minute; }
	int getSekunde() const { return *_sekunde; }
	friend ostream& operator<<(ostream& COUT, const Termin& t) {
		COUT << *t._sati << ":" << *t._minute << ":" << *t._sekunde;
		return COUT;
	}
	bool operator == (const Termin& obj)const {
		return *_sati == *obj._sati && *_minute == *obj._minute && *_sekunde == *obj._sekunde;
	}
};

class Predavanje :public Dogadjaj {
	char* _tema;
	Specializacija _specijalizacija;
public:
	Predavanje(Termin termin, const char* tema, Specializacija spec) :Dogadjaj(termin) {
		_tema = AlocirajNiz(tema);
		_specijalizacija = spec;
	}
	Predavanje(const Predavanje& obj) :Dogadjaj(obj) {
		_tema = AlocirajNiz(obj._tema);
		_specijalizacija = obj._specijalizacija;
	}
	Predavanje& operator = (const Predavanje& obj) {
		if (this != &obj) {

			delete[] _tema;

			Dogadjaj::operator=(obj);

			_tema = AlocirajNiz(obj._tema);
			_specijalizacija = obj._specijalizacija;
		}
	}
	~Predavanje() {
		delete[] _tema;
		_tema = nullptr;
	}
	string Info()const override {
		stringstream ss;
		ss << *this;
		return ss.str();
	}
	const char* GetTema() const { return _tema; }
	Specializacija GetSpecijalizacija() const {
		return _specijalizacija;
	}
	friend ostream& operator << (ostream& COUT, const Predavanje& obj) {
		COUT << obj._tema << " " << obj._termin << " " << obj._specijalizacija << endl;
		return COUT;
	}
	bool operator == (const Predavanje& obj)const {
		return strcmp(_tema, obj._tema) == 0 && _termin == obj._termin && _specijalizacija == obj._specijalizacija;
	}
};
class Ucesnik {
	static int _id; //iskoristiti za praćenje rednog broja ucesnika i generisanje jedinstvene sifre
	char* _sifra; //sifra u formatu GG-IN-BBB, pojasnjena u main funkciji, generisati prilikom kreiranja objekta
	char* _imePrezime;
	vector<Dogadjaj*> _prijavljeni;
public:
	Ucesnik(const char* imePrezime) {
		_imePrezime = AlocirajNiz(imePrezime);
		_sifra = GenerisiSifru(_imePrezime, _id++);
	}
	Ucesnik(const Ucesnik& obj) {
		_imePrezime = AlocirajNiz(obj._imePrezime);
		_sifra = AlocirajNiz(obj._sifra);
		for (int i = 0; i < obj._prijavljeni.size(); i++)
		{
			Predavanje* casted = dynamic_cast<Predavanje*>(obj._prijavljeni[i]);
			_prijavljeni.push_back(new Predavanje(*casted));
		}
	}
	Ucesnik& operator = (const Ucesnik& obj) {
		if (this != &obj) {
			delete[] _imePrezime;
			delete[] _sifra;
			for (auto prijava : _prijavljeni)
				delete prijava;
			_prijavljeni.clear();

			_imePrezime = AlocirajNiz(obj._imePrezime);
			_sifra = AlocirajNiz(obj._sifra);
			for (int i = 0; i < obj._prijavljeni.size(); i++)
			{
				Predavanje* casted = dynamic_cast<Predavanje*>(obj._prijavljeni[i]);
				_prijavljeni.push_back(new Predavanje(*casted));
			}
		}
		return *this;
	}
	bool PrijaviSe(Predavanje predavanje) {
		for (int i = 0; i < _prijavljeni.size(); i++)
		{
			Predavanje* casted = dynamic_cast<Predavanje*>(_prijavljeni[i]);
			if (predavanje == *casted) {
				return false;
			}
		}
		_prijavljeni.push_back(new Predavanje(predavanje));
		return true;
	}
	~Ucesnik() {
		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _sifra;
		_sifra = nullptr;
		for (auto prijava : _prijavljeni)
		{
			delete prijava;
			prijava = nullptr;
		}
		_prijavljeni.clear();
	}
	const char* getSifra() const { return _sifra; }
	const char* getImePrezime() const { return _imePrezime; }
	vector<Dogadjaj*>& getDogadjaji() { return _prijavljeni; }

	friend ostream& operator<< (ostream& COUT, const Ucesnik& obj) {
		COUT << "Ucesnik: " << obj._imePrezime << " [" << obj._sifra << "]" << endl;
		for (int i = 0; i < obj._prijavljeni.size(); i++)
		{
			COUT << "\t\t\t" << obj._prijavljeni[i]->Info();
		}
		return COUT;
	}
	bool operator == (const Ucesnik& obj)const {
		return strcmp(_imePrezime, obj._imePrezime) == 0 && strcmp(_sifra, obj._sifra) == 0;
	}
};

			_clanovi = obj._clanovi;
		}
	}
	void DodajUcesnika(Ucesnik ucesnik) {
		for (int i = 0; i < _clanovi.size(); i++)
		{
			if (_clanovi[i] == ucesnik)
				throw exception("Ucesnik je vec dodan u tim");
		}
		_clanovi.push_back(ucesnik);

	}
	~Tim() {
		delete[] _naziv;
		_naziv = nullptr;
	}
	friend ostream& operator << (ostream& COUT, const Tim& obj) {
		cout << "Naziv tima: " << obj._naziv << endl;
		for (int i = 0; i < obj._clanovi.size(); i++)
		{
			COUT <<"\t\t"<< obj._clanovi[i] << endl;
		}
		return COUT;
	}

	const char* getNaziv() const { return _naziv; }
	vector<Ucesnik>& getUcesnici() { return _clanovi; }

	bool operator == (const Tim& obj)const {
		if (strcmp(_naziv, obj._naziv) != 0) 
			return false;


		for (int i = 0; i < obj._clanovi.size(); i++)
		{
			if (!(_clanovi[i] == obj._clanovi[i])) 
				return false;
		}
		return true;
	}
};
