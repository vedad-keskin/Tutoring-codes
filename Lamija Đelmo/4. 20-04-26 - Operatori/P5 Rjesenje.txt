//    AlocirajNiz
char* GetNizKaraktera(const char* tekst) {
	if (tekst == nullptr)
		return nullptr;
	int vel = strlen(tekst) + 1;
	char* novi = new char[vel];
	strcpy_s(novi, vel, tekst);
	return novi;
}

class Artikal {
	char* _sifra;
	char* _naziv;
	double _cijena;
	int _kolicinaNaSkladistu;
public:


	Artikal(const char* sifra = "SIF000", const char* naziv = "---", double cijena = 0.0, int kolicina = 0 ) {

		_sifra = GetNizKaraktera(sifra);
		_naziv = GetNizKaraktera(naziv);
		_cijena = cijena;
		_kolicinaNaSkladistu = kolicina;

	}

	Artikal(const Artikal& obj) {

		_sifra = GetNizKaraktera(obj._sifra);
		_naziv = GetNizKaraktera(obj._naziv);
		_cijena = obj._cijena;
		_kolicinaNaSkladistu = obj._kolicinaNaSkladistu;

	}

	Artikal& operator = (const Artikal& obj) {
		if (this != &obj) {

			delete[] _sifra;
			delete[] _naziv;

			_sifra = GetNizKaraktera(obj._sifra);
			_naziv = GetNizKaraktera(obj._naziv);
			_cijena = obj._cijena;
			_kolicinaNaSkladistu = obj._kolicinaNaSkladistu;

		}
		return *this;
	}

	~Artikal() {

		delete[] _sifra;
		_sifra = nullptr;
		delete[] _naziv;
		_naziv = nullptr;

	}

	friend ostream& operator << (ostream& COUT, const Artikal& obj) {

		COUT << obj._naziv << " [" << obj._sifra << "] - " << obj._cijena << " KM" << endl;
		COUT << "Kolicina na skladistu: " << obj._kolicinaNaSkladistu << " kom" << endl;

		return COUT;
	}




	// Operator -= na način da se stanje na skladištu artikla promijeni za proslijeđenu vrijednost
	// Voditi računa da se ne može ići u negativnu vrijednost

	//artikal1 -= 3;

	Artikal& operator -= (int kolicina) {

		if (kolicina > _kolicinaNaSkladistu) {
			return *this;
		}


		_kolicinaNaSkladistu -= kolicina;
		//_kolicinaNaSkladistu = _kolicinaNaSkladistu - kolicina;

		return *this;
	}


	// Operator += na način da se stanje na skladištu artikla promijeni za proslijeđenu vrijednost

	//artikal1 += (30); // op +=

	Artikal& operator += (int kolicina) {


		_kolicinaNaSkladistu += kolicina;

		return *this;

	}


	// Operator ++ na način da se cijena artikla uveća za 1

	//artikal1++; // op ++ --> prefiksni operatori 

	Artikal& operator ++ () {


		_cijena++;
		//_cijena += 1;
		//_cijena = _cijena + 1;

		return *this;

	}

	// Operator -- na način da se cijena artikla umanji za 1
	// Voditi računa da se ne može ići u negativnu vrijednost


	Artikal& operator -- () {

		if (_cijena < 1) {

			return *this;

		}


		_cijena--;
		//_cijena += 1;
		//_cijena = _cijena + 1;

		return *this;

	}

	// Operator == koji poredi artikle na osnovu sifre
	// Operatori poredjenja --> == != > < >= <= --> uvijek bool povratne vrijednosti

	//if (artikal1 == artikal2) {

	bool operator == (const Artikal& obj) {

		//if (strcmp(_sifra, obj._sifra) == 0) {
		//	return true;
		//}
		//else {
		//	return false;
		//}

		return strcmp(_sifra, obj._sifra) == 0;

		//         PROVJERA                    ? AKO JE TRUE : FALSE;
		//return strcmp(_sifra, obj._sifra) == 0 ? true : false;

	}

	// getteri --> READ ONLY NACIN PRISTUPA PRIVATNIM PARAMETRIMA BEZ MOGUCNOSTI NJEGOVO MIJENJANJA

	int GetKolicinaNaSkladistu() { return _kolicinaNaSkladistu; }
	double GetCijena() { return _cijena; }
	char* GetSifra() { return _sifra; }
	char* GetNaziv() { return _naziv; }

};

class StavkaRacuna {
	Artikal* _artikal;
	int* _kupljenaKolicina;
public:

	StavkaRacuna(Artikal artikal = Artikal() , int kolicina = 0) {

		_artikal = new Artikal(artikal);
		_kupljenaKolicina = new int(kolicina);

	}

	StavkaRacuna(const StavkaRacuna& obj) {

		_artikal = new Artikal(*obj._artikal);
		_kupljenaKolicina = new int(*obj._kupljenaKolicina);

	}

	StavkaRacuna& operator = (const StavkaRacuna& obj) {
		if (this != &obj) {

			delete _artikal;
			delete _kupljenaKolicina;

			_artikal = new Artikal(*obj._artikal);
			_kupljenaKolicina = new int(*obj._kupljenaKolicina);
		}
		return *this;
	}

	~StavkaRacuna() {

		delete _artikal;
		_artikal = nullptr;
		delete _kupljenaKolicina;
		_kupljenaKolicina = nullptr;

	}


	friend ostream& operator << (ostream& COUT, const StavkaRacuna& obj) {

		COUT << *obj._artikal ;
		COUT << "Kupljena kolicina : " << *obj._kupljenaKolicina << " kom" << endl << endl ;

		return COUT;
	}


	// Operator += koji uvećava stanje kupljene količine

	//stavka1 += 5;

	StavkaRacuna& operator += (int kolicina) {


		//_kupljenaKolicina += kolicina;
		*_kupljenaKolicina += kolicina;
		//*_kupljenaKolicina = *_kupljenaKolicina + kolicina;
		//_kupljenaKolicina = new int(*_kupljenaKolicina + kolicina);

		return *this;

	}


	// Operator -= koji umanjuje stanje kupljene količine
	// Voditi računa da se ne može ići u negativnu vrijednost

	//stavka1 -= 5;

	StavkaRacuna& operator -= (int kolicina) {

		if (kolicina > *_kupljenaKolicina) {
			return *this;
		}

		//_kupljenaKolicina += kolicina;
		*_kupljenaKolicina -= kolicina;
		//*_kupljenaKolicina = *_kupljenaKolicina + kolicina;
		//_kupljenaKolicina = new int(*_kupljenaKolicina + kolicina);

		return *this;

	}

	// Operator == koji poredi stavke na osnovu sifre

	//if (stavka1 == stavka2) {

	bool operator == (const StavkaRacuna& obj) {

		//if (strcmp(_artikal._sifra, obj._artikal._sifra) == 0) {
		//if (strcmp( _artikal->GetSifra(), obj._artikal->GetSifra()) == 0) {
		//	return true;
		//}
		//else {

		//	return false;
		//}

		//return strcmp(_artikal->GetSifra(), obj._artikal->GetSifra()) == 0;

		//return strcmp(_artikal->GetSifra(), obj._artikal->GetSifra()) == 0 ? true : false;


		//if ( *_artikal == *obj._artikal ) { 
		//	return true;
		//}
		//else {

		//	return false;
		//}

		return *_artikal == *obj._artikal;

		//return *_artikal == *obj._artikal ? true : false;

	}



	double GetUkupnaCijena() { // vraća cijenu stavke racuna (kolicina * cijena artikla) 

		return _artikal->GetCijena() * *_kupljenaKolicina;

	}

	double GetKolicina() { // vraća količinu stavke računa 

		return *_kupljenaKolicina;

	}

	Artikal& GetArtikal() { // vraća originalni artikal
		return *_artikal;
	}

};


class Racun {
	static int _id; // unikatni identifikator, koristi se za povećanje broja računa uz osiguravanje unikatnosti 
	int _brojRacuna; // 1 --> 2 --> 3 --> 4 .. 

	StavkaRacuna* _stavke;
	int _trenutno;
	int _max;

	double _iznosBezPDV;
	double _iznosSaPDV;
	bool _naplaceno;
public:

	Racun(int max = 3) {

		//_brojRacuna = _id;
		//_id++; // 1 --> 2 --> 3 --> 4

		_brojRacuna = _id++;

		_trenutno = 0;
		_max = max;

		_stavke = new StavkaRacuna[_max];

		_iznosBezPDV = 0;
		_iznosSaPDV = 0;

		_naplaceno = false;


	}

	Racun(const Racun& obj) {

		//_id = obj._id; static se nikada ne kopira jer bi pokvario logiku povecavanja id-eva

		_brojRacuna = obj._brojRacuna;

		_trenutno = obj._trenutno;
		_max = obj._max;

		_stavke = new StavkaRacuna[_max];

		for (int i = 0; i < _trenutno; i++)
		{
			_stavke[i] = obj._stavke[i];
		}

		_iznosBezPDV = obj._iznosBezPDV;
		_iznosSaPDV = obj._iznosSaPDV;

		_naplaceno = obj._naplaceno;

	}

	Racun& operator = (const Racun& obj) {
		if (this != &obj) {

			delete[]_stavke;

			_brojRacuna = obj._brojRacuna;

			_trenutno = obj._trenutno;
			_max = obj._max;

			_stavke = new StavkaRacuna[_max];

			for (int i = 0; i < _trenutno; i++)
			{
				_stavke[i] = obj._stavke[i];
			}

			_iznosBezPDV = obj._iznosBezPDV;
			_iznosSaPDV = obj._iznosSaPDV;

			_naplaceno = obj._naplaceno;
		}
		return *this;
	}

	~Racun() {

		delete[]_stavke;
		_stavke = nullptr;

	}




	// Operator () - Dodaje nove artikle u _stavke (ulazni parametri su Artikal i kolicina)
	// Voditi računa da nema prekoračenja niza
	// Prilikom svakog dodavanja artikla, ažurirati vrijednosti iznosBezPDV i iznosSaPDV
	// Ažurirati količinu na skladištu artikla nakon svake kupovine
	// Voditi računa da prilikom dodavanja duplikata ažuriramo kupljenu količinu, količinu na skladištu i iznose bez i sa PDV ili samo zabraniti unos duplikata

	//racun1(artikal1, 5); // HLJEB --> 5 

	Racun& operator () (Artikal artikal , int kolicina) {

		if (_max == _trenutno) {

			return *this;
		}

		StavkaRacuna stavka(artikal, kolicina); // napravili smo stavku preko constr. koristeci parametre koji su nam dati


		//for (int i = 0; i < _trenutno; i++)
		//{
		//	if (_stavke[i] == stavka) { // op == 

		//		return *this;

		//	}
		//}

		for (int i = 0; i < _trenutno; i++)
		{
			if (_stavke[i].GetArtikal() == artikal) { // op == 

				// TODO 
				// Voditi računa da prilikom dodavanja duplikata ažuriramo kupljenu količinu, količinu na skladištu i iznose bez i sa PDV ili samo zabraniti unos duplikata


				return *this;

			}
		}




		_stavke[_trenutno] = stavka;

		// ARTIKAL -= kolicina 
		_stavke[_trenutno].GetArtikal() -= kolicina;

		_trenutno++;



		//_iznosBezPDV += artikal.GetCijena() * kolicina;
		_iznosBezPDV += stavka.GetUkupnaCijena();

		//                  15.20 + 17% od 15.20
		//_iznosSaPDV += stavka.GetUkupnaCijena() + (stavka.GetUkupnaCijena() * 0.17);
		_iznosSaPDV += stavka.GetUkupnaCijena() * 1.17;

		return *this;
	}





	void PrintRacun() {
		// Ispisuje sve stavke računa zajedno sa iznosom za plaćanje sa i bez PDV-a i brojem računa
		// Voditi računa da se jedan račun ne može dva puta isprintati (paziti na bool _naplaceno)

		//if (_naplaceno == true) {
		if (_naplaceno) {

			cout << "Racun je vec naplacen." << endl;
			return;

		}

		cout << "========= Br. racuna " << _brojRacuna << " =========" << endl << endl;


		for (int i = 0; i < _trenutno; i++)
		{
			cout << _stavke[i] ;
			cout << "================================" << endl << endl;
		}

		cout << "Iznos bez PDV : " << _iznosBezPDV << " KM" << endl;
		cout << "Iznos sa PDV : " << _iznosSaPDV << " KM" << endl;

		_naplaceno = true;

	}

};
