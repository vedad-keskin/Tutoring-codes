// POKAZIVACI : 

// 1. ALOKACIJA -- ( new ) 

// AKO KREIRAMO NESTO STO JE JEDNA VRIJEDNOST -> ( ) 

// AKO KREIRAMO NESTO STO JE VISE VRIJEDNOSTI -> [ ]  

// 2. DEREFERENCIRANJE --> ( * )

// AKO KREIRAMO NESTO STO JE JEDNA VRIJEDNOST -> ( ) _ECTS -> 000002CD4AD58320 *_ECTS -> 0

// AKO KREIRAMO NESTO STO JE VISE VRIJEDNOSTI -> [ ] _naziv = Nema predmeta , *_naziv = N

// 3. DESTRUKTOR / DEALOKACIJA

// AKO KREIRAMO NESTO STO JE JEDNA VRIJEDNOST -> ( ) -->  delete _ECTS 

// AKO KREIRAMO NESTO STO JE VISE VRIJEDNOSTI -> [ ] -->  delete[] _naziv;


class Predmet {
public:			  // dft. constr.  //
	char* _naziv; // Nema predmeta //
	//char _naziv[50];
	int _ocjena;  //      5		   //
	int* _ECTS;	  //      0 	   //

	// dft constr.

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char tekst1[] = "Nema predmeta";
		//string tekst2 = "Nema predmeta";
		const char* tekst3 = "Nema predmeta"; // string litreal -> READ-ONLY MEM.

		int vel = strlen(tekst3) + 1; // 13 -> 14 null terminator

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, tekst3);

	}

	// user-def constr.
	//Predmet predmet1("Programiranje II", 7, 9);

	Predmet(const char* naziv, int ects, int ocjena) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);

	}

	// copy constr.
	//Predmet predmetKopija2(predmet1); //  copy constr.

	// sa & --> original 
	// bez & --> kopiju
	// const --> nesto kontantno -> ne moze se mijenjati

	Predmet(const Predmet& predmet) { // napravila kopija Predmeta

		//predmet._ocjena = 10;

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);

	}

	//predmetOperatorDodjele = predmet1; // operator = ( operator dodjele)


	Predmet& operator = (const Predmet& predmet) {

		// adresu lijevog predmeta != adresa desnog predmeta
		if (this != &predmet) {

			delete _ECTS;

			delete[] _naziv;


			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);

		}

		return *this;

	}







	// destruktor 
	// TILDA -> altgr + 1 ~~~~~

	~Predmet() {

		//cout << "DESTURKOR SE POZVAO" << endl;

		delete _ECTS;

		_ECTS = nullptr;

		delete[] _naziv;

		_naziv = nullptr;

	}


	// ispis 

	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;

	}




};


class Student {
	char* _imePrezime;

	Predmet* _polozeni; // ( ) ili [ ] 
	int _trenutno;      // koja nam kaze koliko trenutno imamo polozneih predmeta
	int _max;           // koja nam kaze koliko maksimalno mozemo imati polozenih predmeta
public:


	Student() {

		const char* dftImePrezime = "Nema imena i prezimena";
		int vel = strlen(dftImePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, dftImePrezime);

		_trenutno = 0;

		_max = 3;

		_polozeni = new Predmet[_max];

	}

	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, imeprezime);

		_trenutno = 0;
		_max = max;

		_polozeni = new Predmet[_max];


	}

	Student(const Student& obj) {

		int vel = strlen(obj._imePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, obj._imePrezime);

		_trenutno = obj._trenutno;
		_max = obj._max;

		_polozeni = new Predmet[_max]; // alokacija niza znaci da alocirani niz ima toliko prazhnih elementa

		for (int i = 0; i < _trenutno; i++) // 0 1 2 
		{
			_polozeni[i] = obj._polozeni[i]; // op = 
		}

	}

	Student& operator = (const Student& obj) {
		if (this != &obj) {


			delete[] _imePrezime;
			delete[] _polozeni;

			int vel = strlen(obj._imePrezime) + 1;
			_imePrezime = new char[vel];
			strcpy_s(_imePrezime, vel, obj._imePrezime);

			_trenutno = obj._trenutno;
			_max = obj._max;

			_polozeni = new Predmet[_max]; // alokacija niza znaci da alocirani niz ima toliko prazhnih elementa

			for (int i = 0; i < _trenutno; i++) // 0 1 2 
			{
				_polozeni[i] = obj._polozeni[i]; // op = 
			}
		}
		return *this;
	}


	~Student() {

		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _polozeni;
		_polozeni = nullptr;

	}

	void Ispis() {

		cout << "Ime i prezime studenta: " << _imePrezime << endl;
		cout << "POLOZENI PREDMETI STUDENTA: " << endl << endl;

		for (int i = 0; i < _trenutno; i++) // 0 1 2
		{
			_polozeni[i].Ispis();
		}

	}




	//student.AddPolozeniPredmet(predmet1);
	void AddPolozeniPredmet(Predmet predmet) { // MATII

		// Voditi računa da se ne može prekoračiti kapacitet niza

		//     3 == 2
		if (_max == _trenutno) {

			cout << "Kapacitet niza je popunjen" << endl;
			return;

		}

		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		        


		for (int i = 0; i < _trenutno; i++) // 0 1 
		{
			// pol[0] = PRII       == MATII ,, pol[1] = MATII == MATII 

			// char* == char* 
			// strcmp( naziv1 , naziv2 ) == 0

			//if (_polozeni[i]._naziv == predmet._naziv) {
			if (strcmp( _polozeni[i]._naziv , predmet._naziv) == 0) {

				cout << "Duplikati nisu dozvoljeni" << endl;
				return;

			}

		}







		// Dodaje novi polozeni predmet u niz

		// Predmet   = predmet 
		_polozeni[_trenutno] = predmet; // pol[0] = PRII , pol[1] = MATII , pol[2] = OS
		// pol[3] = WRD

		_trenutno++; // 0 --> 1 --> 2


	}
