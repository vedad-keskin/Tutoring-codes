char* AlocirajNiz(const char* sadrzaj, bool dealociraj = false) {
	if (sadrzaj == nullptr)return nullptr;
	int vel = strlen(sadrzaj) + 1;
	char* temp = new char[vel];
	strcpy_s(temp, vel, sadrzaj);
	if (dealociraj)
		delete[]sadrzaj;
	return temp;
}

/* sifra korisnika treba biti u formatu GG-IN-BBB pri cemu su:

GG - posljednje dvije cifre trenutne godine (npr.za 2025 -> 25), preuzeti vrijednost iz sistema
IN - inicijali ucesnika, velika slova(prvo slovo imena i prezimena)
BBB → trocifreni redni broj ucesnika kreiran na osnovu _id-a (npr. 001, 023, 105)

validnom sifrom treba smatrati i onu koja umjesto znaka crtica '-' ima znak razmak npr: 25 DM 003 ili 25 DM-003
*/



int GetCurrentYearDigits() {

	time_t CurrentTime = time(0); // i nullptr

	tm LocalTime;

	localtime_s(&LocalTime, &CurrentTime);

	int FullYear = 1900 + LocalTime.tm_year; // 2025

	return FullYear % 100; // 25

}

char GetFirstLetterLastName(const char* imeprezime) {

	for (int i = 0; i < strlen(imeprezime); i++)
	{
		if (isspace(imeprezime[i])) {
			return imeprezime[i + 1];
		}
	}


}

char* GenerisiSifru(const char* imeprezime, int id) {

	char FirstLetterN = toupper(imeprezime[0]); // D

	char FirstLetterLN = toupper(GetFirstLetterLastName(imeprezime)); // M 

	int YearDigits = GetCurrentYearDigits(); // 25

	char temp[10];

	if (id < 10) { // treba vratiti 25-DM-003

		sprintf_s(temp, "%d-%c%c-00%d", YearDigits, FirstLetterN, FirstLetterLN, id);

	}
	else if (id < 100) { // treba vratiti 25-JA-014

		sprintf_s(temp, "%d-%c%c-0%d", YearDigits, FirstLetterN, FirstLetterLN, id);

	}
	else { // treba vratiti 25-GS-156

		sprintf_s(temp, "%d-%c%c-%d", YearDigits, FirstLetterN, FirstLetterLN, id);

	}

	return AlocirajNiz(temp);

}

bool ValidirajSifru(const char* sifra) {

	regex pattern("[0-9]{2}[- ][A-Z]{2}[- ][0-9]{3}");

	return regex_match(sifra, pattern);

}



template<class T1, class T2, int max>
class KolekcijaG1 {
	T1* _elementi1[max];
	T2* _elementi2[max];
	int _trenutno;
public:

	KolekcijaG1() {

		_trenutno = 0;

		// kada imamo zadatke s maxom ovdje ne radimo nista

	}

	KolekcijaG1(const KolekcijaG1& obj) {

		_trenutno = obj._trenutno;

		for (int i = 0; i < _trenutno; i++)
		{
			_elementi1[i] = new T1(*obj._elementi1[i]);
			_elementi2[i] = new T2(*obj._elementi2[i]);
		}

	}

	KolekcijaG1& operator = (const KolekcijaG1& obj) {
		if (this != &obj) {


			for (int i = 0; i < _trenutno; i++)
			{
				delete _elementi1[i];
				delete _elementi2[i];
			}


			_trenutno = obj._trenutno;

			for (int i = 0; i < _trenutno; i++)
			{
				_elementi1[i] = new T1(*obj._elementi1[i]);
				_elementi2[i] = new T2(*obj._elementi2[i]);
			}

		}
		return *this;
	}

	void AddElement(T1 el1, T2 el2) {


		_elementi1[_trenutno] = new T1(el1);
		_elementi2[_trenutno] = new T2(el2);

		_trenutno++;

	}

	///* metoda InsertAt treba da doda vrijednosti drugog i treceg parametra na lokaciju koja je definisana prvim parametrom. Povratna vrijednost metode je objekat (pozivaoc metode, u konkretnom slucaju objekat kolekcija1) u okviru koga su, na definisanu lokaciju, dodati proslijedjeni parametri. Nakon izvrsenja metode InsertAt, oba objekta, kolekcija1 i kolekcija2, bi trebali posjedovati sljedeci sadrzaj:
	//0 Vrijednost -> 0
	//10 Vrijednost -> 10
	//1 Vrijednost -> 1
	//2 Vrijednost -> 2
	//* ....
	//*/

	//KolekcijaG1<int, string, 10> kolekcija2 = kolekcija1.InsertAt(1, 10, "Vrijednost -> 10");
	//cout << kolekcija2 << crt;

	KolekcijaG1 InsertAt(int lokacija, T1 el1, T2 el2) {

		KolekcijaG1 kolekcija2;

		for (int i = 0; i < _trenutno; i++)
		{
			if (i == lokacija) {

				kolekcija2.AddElement(el1, el2);

			}
			kolekcija2.AddElement(*_elementi1[i], *_elementi2[i]);

		}

		*this = kolekcija2;

		return kolekcija2;

	}

	///*Metoda RemoveRange kao prvi parametar prihvata pocetnu lokaciju a kao drugi parametar broj elemenata koje, od pocetne lokacije uklanja iz kolekcije koja je pozvala tu metodu. U slucaju da zahtijevani broj elemenata ne postoji u kolekciji metoda treba baciti izuzetak. Na kraju, metoda treba da vrati pokazivac na novi objekat tipa Kolekcija koji sadrzi samo uklonjene elemente*/

	//KolekcijaG1<int, string, 10>* kolekcija3 = kolekcija1.RemoveRange(1, 3);//uklanja 3 elementa pocevsi od lokacije 1

	//cout << "Uklonjeni:\n" << *kolekcija3;
	//cout << "Preostali:\n" << kolekcija1;

	KolekcijaG1* RemoveRange(int lokacijaOd, int brojElementa) { // 1 3 

		if (lokacijaOd < 0 || lokacijaOd + brojElementa > _trenutno) {

			throw exception("Dati opseg ne postoji");

		}

		KolekcijaG1* kolekcija3 = new KolekcijaG1();

		KolekcijaG1 kolekcija1;

		for (int i = 0; i < _trenutno; i++)
		{
			if (i >= lokacijaOd && i < lokacijaOd + brojElementa) {


				kolekcija3->AddElement(*_elementi1[i], *_elementi2[i]);

			}
			else {

				kolekcija1.AddElement(*_elementi1[i], *_elementi2[i]);

			}
		}

		*this = kolekcija1;

		return kolekcija3;

	}


	~KolekcijaG1() {

		for (int i = 0; i < _trenutno; i++)
		{
			delete _elementi1[i];
			_elementi1[i] = nullptr;
			delete _elementi2[i];
			_elementi2[i] = nullptr;
		}

	}

	T1& getElement1(int lokacija) { return *_elementi1[lokacija]; }
	T2& getElement2(int lokacija) { return *_elementi2[lokacija]; }
	int getTrenutno() const { return _trenutno; }

	friend ostream& operator<< (ostream& COUT, KolekcijaG1& obj) {
		for (size_t i = 0; i < obj._trenutno; i++)
			COUT << obj.getElement1(i) << " " << obj.getElement2(i) << endl;
		return COUT;
	}
};
class Termin {
	int* _sati; int* _minute; int* _sekunde;
public:
	Termin(int s = 0, int m = 0, int k = 0) {
		_sati = new int(s);
		_minute = new int(m);
		_sekunde = new int(k);
	}

	Termin(const Termin& obj) {

		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);

	}

	Termin& operator = (const Termin& obj) {
		if (this != &obj) {

			delete _sati; delete _minute; delete _sekunde;


			_sati = new int(*obj._sati);
			_minute = new int(*obj._minute);
			_sekunde = new int(*obj._sekunde);
		}
		return *this;
	}

	bool operator == (const Termin& obj) {

		return *_sati == *obj._sati && *_minute == *obj._minute && *_sekunde == *obj._sekunde;

	}

	int toSeconds() {

		return *_sekunde + *_minute * 60 + *_sati * 3600;

	}

	~Termin() {
		delete _sati; delete _minute; delete _sekunde;
		_sati = nullptr; _minute = nullptr; _sekunde = nullptr;
	}

	int getSati() const { return *_sati; }
	int getMinute() const { return *_minute; }
	int getSekunde() const { return *_sekunde; }

	friend ostream& operator<<(ostream& COUT, const Termin& t) {
		COUT << *t._sati << ":" << *t._minute << ":" << *t._sekunde;
		return COUT;
	}
};

		_tema = AlocirajNiz(obj._tema);
		_specijalizacija = obj._specijalizacija;

		if (this != &obj) {

			delete[] _tema;

			_specijalizacija = obj._specijalizacija;
			
		}

		stringstream ss;

		ss << *this;

		return ss.str();

		return new Predavanje(*this);

	}

	bool operator == (const Predavanje& obj) {

		return _termin == obj._termin && strcmp(_tema, obj._tema) == 0 && _specijalizacija == obj._specijalizacija;

	}

	}

class Ucesnik {
	static int _id; //iskoristiti za praćenje rednog broja ucesnika i generisanje jedinstvene sifre
	char* _sifra; //sifra u formatu GG-IN-BBB, pojasnjena u main funkciji, generisati prilikom kreiranja objekta
	char* _imePrezime;
	vector<Dogadjaj*> _prijavljeni; // bazna klasa koja pokazuje na izvedenu 

	// PRISTUP: (BAZNA -> IZVEDENA) - DYNAMIC_CAST
	// iz bazne klase prelazimo u izvedenu koristeći dynamic_cast (BAZNA -> IZVEDENA)

	// POLIMORFNO KOPIRANJE: (BAZNA koja pokazuje na IZVEDENU -> nova BAZNA koja pokazuje na IZVEDENU) - KLONIRANJE
	// Dogadjaj je apstraktna klasa, pa ga ne možemo direktno instancirati;
	// koristimo Clone() da polimorfno napravimo novi objekat istog izvedenog tipa

	// POHRANA: (IZVEDENA -> BAZNA) - KLONIRANJE ILI KLASCINA POHRANA
	// ako znamo tačan tip (npr. Predavanje), možemo direktno kreirati izvedenu: new Predavanje(predavanje);
	// ako tip ne znamo ili želimo da kod radi za bilo koji izvedeni tip (polimorfno), koristimo Clone()
	// U oba slučaja, u vector<Dogadjaj*> pohranjujemo pokazivač bazne klase koji pokazuje na izvedenu

public:
	Ucesnik(const char* imePrezime) {
		_imePrezime = AlocirajNiz(imePrezime);

		_id++;

		_sifra = GenerisiSifru(_imePrezime, _id);

	}

	Ucesnik(const Ucesnik& obj) {

		_imePrezime = AlocirajNiz(obj._imePrezime);
		_sifra = AlocirajNiz(obj._sifra);

		for (int i = 0; i < obj._prijavljeni.size(); i++)
		{
			_prijavljeni.push_back(obj._prijavljeni[i]->Clone());
		}

	}

	Ucesnik& operator = (const Ucesnik& obj) {
		if (this != &obj) {

			delete[] _imePrezime;
			delete[] _sifra;

			for (auto prijava : _prijavljeni) {
				delete prijava;
			}
			_prijavljeni.clear();


			_imePrezime = AlocirajNiz(obj._imePrezime);
			_sifra = AlocirajNiz(obj._sifra);

			for (int i = 0; i < obj._prijavljeni.size(); i++)
			{
				_prijavljeni.push_back(obj._prijavljeni[i]->Clone());
			}

		}
		return *this;
	}

	bool operator == (const Ucesnik& obj) {

		return strcmp(_sifra, obj._sifra) == 0;

	}


	bool PrijaviSe(Predavanje predavanje) {

		for (int i = 0; i < _prijavljeni.size(); i++)
		{
			// PRISTUP --> ( *BAZNA --> *IZVEDENU ) --> CAST
			Predavanje* castedPredavanje = dynamic_cast<Predavanje*>(_prijavljeni[i]);

			if (*castedPredavanje == predavanje) {

				return false;

			}
		}

		_prijavljeni.push_back(new Predavanje(predavanje));

		return true;

	}

	~Ucesnik() {
		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _sifra;
		_sifra = nullptr;

		for (auto prijava : _prijavljeni) {
			delete prijava;
			prijava = nullptr;
		}

		//for (int i = 0; i < _prijavljeni.size(); i++)
		//{
		//	delete _prijavljeni[i];
		//}

		
		_prijavljeni.clear();
	}

	const char* getSifra() const { return _sifra; }
	const char* getImePrezime() const { return _imePrezime; }
	vector<Dogadjaj*>& getDogadjaji() { return _prijavljeni; }

	friend ostream& operator<<(ostream& COUT, const Ucesnik& obj) {

		COUT << "Ucesnik: " << obj._imePrezime << " " << obj._sifra << endl;

		for (int i = 0; i < obj._prijavljeni.size(); i++)
		{

			COUT << "\t\t\t" << obj._prijavljeni[i]->Info() << endl;

		}

		return COUT;
	}

};

		for (int i = 0; i < _clanovi.size(); i++)

			if (ucesnik == _clanovi[i]) {

				throw exception("Ucesnik je vec clan tima");

			}
		}



		_clanovi.push_back(ucesnik);

	}

	~Tim() {
		delete[] _naziv;
		_naziv = nullptr;
	}

	char* getNaziv() const { return _naziv; }
	vector<Ucesnik>& getUcesnici() { return _clanovi; }

		{
			COUT << "\t\t" << obj._clanovi[i] << endl;
		}

		return COUT;
	}

};

	}

	}

	void DodajTimove(Tim tim1, Tim tim2) {


		_timovi.AddElement(new Tim(tim1), new Tim(tim2)); // 1. razlika 

	}

	bool PrijaviDogadjaj(const char* nazivTima, const char* sifraUcesnika, Predavanje predavanje) {

		// proci kroz sve timove

		for (int i = 0; i < _timovi.getTrenutno(); i++)
		{
			// provjera da li tim1 ima naziv kao naziv koji je proslijedjen kao parametar 

			if (strcmp(nazivTima, _timovi.getElement1(i)->getNaziv() ) == 0) {

				// proci kroz sve ucesnike tima1

				for (int j = 0; j < _timovi.getElement1(i)->getUcesnici().size(); j++)
				{

					// pronaci ucesnika sa datom sifrom

					if (strcmp(sifraUcesnika, _timovi.getElement1(i)->getUcesnici()[j].getSifra()) == 0) {

						_timovi.getElement1(i)->getUcesnici()[j].PrijaviSe(predavanje) ? true : false;

					}

				}
				return false; // ako tim postoji ali ucesnik u tom timu ne postoji ili je proslijedjena pogresna sifra

			}


			// provjera da li tim2 ima naziv kao naziv koji je proslijedjen kao parametar 

			if (strcmp(nazivTima, _timovi.getElement2(i)->getNaziv()) == 0) {

				// proci kroz sve ucesnike tima2

				for (int j = 0; j < _timovi.getElement2(i)->getUcesnici().size(); j++)
				{

					// pronaci ucesnika sa datom sifrom

					if (strcmp(sifraUcesnika, _timovi.getElement2(i)->getUcesnici()[j].getSifra()) == 0) {

						_timovi.getElement2(i)->getUcesnici()[j].PrijaviSe(predavanje) ? true : false;

						return true;

					}

				}
				return false; // ako tim postoji ali ucesnik u tom timu ne postoji ili je proslijedjena pogresna sifra

			}


		}
		return false; // tim ne postoji



	}

	////metoda vraca sve ucesnike koji su se na odredjenoj konferenciji prijavili na minimalno proslijedjeni broj predavanja
	//vector<Ucesnik*> vrijedniUcesnici = savremena_medicina(2);
	//for (auto ucesnik : vrijedniUcesnici)
	//	cout << ucesnik->getImePrezime() << "\n";

	vector<Ucesnik*> operator () (int brojPredavanja) {

		vector<Ucesnik*> vrijedniUcesnici;

		// proci kroz sve timove

		for (int i = 0; i < _timovi.getTrenutno(); i++)
		{

			// proci kroz ucesnike tim1

			for (int j = 0; j < _timovi.getElement1(i)->getUcesnici().size(); j++)
			{

				// provjeriti da li ucesnik ima vise ili jednako broju proslijedjenih predavanja

				if (_timovi.getElement1(i)->getUcesnici()[j].getDogadjaji().size() >= brojPredavanja ) {

					vrijedniUcesnici.push_back(new Ucesnik(_timovi.getElement1(i)->getUcesnici()[j]));

				}

			}

			// proci kroz ucesnike tim2

			for (int j = 0; j < _timovi.getElement2(i)->getUcesnici().size(); j++)
			{

				// provjeriti da li ucesnik ima vise ili jednako broju proslijedjenih predavanja

				if (_timovi.getElement2(i)->getUcesnici()[j].getDogadjaji().size() >= brojPredavanja) {

					vrijedniUcesnici.push_back(new Ucesnik(_timovi.getElement2(i)->getUcesnici()[j]));

				}

			}

		}

		return vrijedniUcesnici;


	}

	~Konferencija() {
		delete[] _naziv; _naziv = nullptr;
	}
