char* AlocirajTekst(const char* tekst) {
	if (!tekst) return nullptr;
	size_t vel = strlen(tekst) + 1;
	char* temp = new char[vel];
	strcpy_s(temp, vel, tekst);
	return temp;
}




int trenutnaGodina() {
	time_t utcV = time(0);
	tm lokalno;
	localtime_s(&lokalno, &utcV);
	return lokalno.tm_year + 1900;
}

string GenerisiSifru(string imeprezime, int id) {
	char prvoSlovoIme = toupper(imeprezime[0]);
	char prvoSlovoPrezime = toupper(imeprezime[imeprezime.find(' ') + 1]);

	string sifra;

	sifra += prvoSlovoIme;
	sifra += prvoSlovoPrezime;
	sifra += ":";
	sifra += to_string(trenutnaGodina());
	sifra += "-";

	if (id < 10)
		sifra += "00";
	else if (id < 100)
		sifra += "0";

	sifra += to_string(id);

	return sifra;
}

bool ValidirajSifru(const char* sifra) {
	regex reg("[A-Z]{2}[:][0-9]{4}[-][0-9]{3}");
	return regex_match(sifra, reg);

template<class T1, class T2, int max>
class KolekcijaParova {
	T1* _prvi[max] = { nullptr };
	T2* _drugi[max] = { nullptr };
	int _trenutno = 0;
public:

	KolekcijaParova() = default;

	KolekcijaParova(const KolekcijaParova& obj) {
		_trenutno = obj._trenutno;

		for (int i = 0; i < _trenutno; i++)
		{
			_prvi[i] = new T1(*obj._prvi[i]);
			_drugi[i] = new T2(*obj._drugi[i]);
		}

	}

	KolekcijaParova& operator=(const KolekcijaParova& obj) {
		if (this != &obj) {

			for (int i = 0; i < _trenutno; i++) {
				delete _prvi[i];
				delete _drugi[i];
			}
			_trenutno = obj._trenutno;
			for (int i = 0; i < _trenutno; i++)
			{
				_prvi[i] = new T1(*obj._prvi[i]);
				_drugi[i] = new T2(*obj._drugi[i]);
			}
		}
		return *this;
	}

	void Dodaj(T1 el1, T2 el2) {
		_prvi[_trenutno] = new T1(el1);
		_drugi[_trenutno] = new T2(el2);
		_trenutno++;
	}

	KolekcijaParova<int, string, 20> DodajNaPoziciju(int lokacija, T1 el1, T2 el2) {
		KolekcijaParova<int, string, 20> kolekcija1;

		for (int i = 0; i < _trenutno; i++)
		{
			if (i == lokacija) {
				kolekcija1.Dodaj(el1, el2);
			}
			kolekcija1.Dodaj(*_prvi[i], *_drugi[i]);
		}
		return kolekcija1;
	}

	KolekcijaParova<int, string, 20>* UkloniRaspon(int od, int raspon) {

		KolekcijaParova<int, string, 20>kolekcija1;

		KolekcijaParova<int, string, 20>* kolekcija3 = new KolekcijaParova();

		for (int i = 0; i < _trenutno; i++)
		{
			if (i >= od && i < od + raspon) {
				kolekcija3->Dodaj(*_prvi[i], *_drugi[i]);
			}
			else
				kolekcija1.Dodaj(*_prvi[i], *_drugi[i]);
		}

		*this = kolekcija1;

		return kolekcija3;

	}


	~KolekcijaParova() {
		for (int i = 0; i < _trenutno; i++) {
			delete _prvi[i]; _prvi[i] = nullptr;
			delete _drugi[i]; _drugi[i] = nullptr;
		}
	}
	int GetTrenutno() const { return _trenutno; }
	T1& GetPrvi(int indeks)const { return *_prvi[indeks]; }
	T2& GetDrugi(int indeks)const { return *_drugi[indeks]; }
	T1& operator[](int indeks) { return *_prvi[indeks]; }

	friend ostream& operator<<(ostream& COUT, const KolekcijaParova& obj) {
		for (int i = 0; i < obj._trenutno; i++)
			COUT << obj.GetPrvi(i) << " " << obj.GetDrugi(i) << "\n";
		return COUT;
	}
};

class DatumVrijeme {
	int* _godina, * _mjesec, * _dan, * _sati, * _minute, * _sekunde;
public:
	DatumVrijeme(int dan = 1, int mjesec = 1, int godina = 2000, int
		sati = 0, int minute = 0, int sekunde = 0) {
		_godina = new int(godina);
		_mjesec = new int(mjesec);
		_dan = new int(dan);
		_sati = new int(sati);
		_minute = new int(minute);
		_sekunde = new int(sekunde);
	}

	DatumVrijeme(const DatumVrijeme& obj) {
		_godina = new int(*obj._godina);
		_mjesec = new int(*obj._mjesec);
		_dan = new int(*obj._dan);
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);
	}

	DatumVrijeme& operator=(const DatumVrijeme& obj) {
		if (this != &obj) {
			delete _godina; delete _mjesec; delete _dan;
			delete _sati; delete _minute; delete _sekunde;

			_godina = new int(*obj._godina);
			_mjesec = new int(*obj._mjesec);
			_dan = new int(*obj._dan);
			_sati = new int(*obj._sati);
			_minute = new int(*obj._minute);
			_sekunde = new int(*obj._sekunde);
		}
		return *this;
	}

	bool operator==(const DatumVrijeme& obj) {
		return *_godina == *obj._godina && *_mjesec == *obj._mjesec && *_dan == *obj._dan
			&& *_sati == *obj._sati && *_minute == *obj._minute && *_sekunde == *obj._sekunde;
	}

	bool operator==(const DatumVrijeme& obj) const {
		return *_godina == *obj._godina && *_mjesec == *obj._mjesec && *_godina == *obj._godina
			&& *_sati == *obj._sati && *_minute == *obj._minute && *_sekunde == *obj._sekunde;
	}


	~DatumVrijeme() {
		delete _godina; delete _mjesec; delete _dan;
		delete _sati; delete _minute; delete _sekunde;
		_godina = nullptr; _mjesec = nullptr; _dan = nullptr;
		_sati = nullptr; _minute = nullptr; _sekunde = nullptr;
	}

	friend ostream& operator<<(ostream& COUT, const DatumVrijeme& obj) {
		COUT << *obj._dan << "." << *obj._mjesec << "." << *obj._godina << " " << *obj._sati << ":" << *obj._minute << ":" << *obj._sekunde;
		return COUT;
	}
};

class Proizvod {
	char* _naziv;
	Kategorija _kategorija;
	int _cijena;
public:
	Proizvod(const char* naziv = "", Kategorija kategorija = ELEKTRONIKA, int cijena = 0) : _kategorija(kategorija), _cijena(cijena) {
		_naziv = AlocirajTekst(naziv);
	}

	Proizvod(const Proizvod& obj) {
		_naziv = AlocirajTekst(obj._naziv);
		_kategorija = obj._kategorija;
		_cijena = obj._cijena;
	}

	Proizvod& operator=(const Proizvod& obj) {
		if (this != &obj)
		{
			delete[] _naziv;

			_naziv = AlocirajTekst(obj._naziv);
			_kategorija = obj._kategorija;
			_cijena = obj._cijena;
		}
		return *this;
	}

	bool operator==(const Proizvod& obj) {
		return strcmp(_naziv, obj._naziv) == 0 && _kategorija == obj._kategorija && _cijena == obj._cijena;
	}

	bool operator==(const Proizvod& obj) const {
		return strcmp(_naziv, obj._naziv) == 0 && _kategorija == obj._kategorija && _cijena == obj._cijena;
	}

	~Proizvod() {
		delete[] _naziv;
		_naziv = nullptr;
	}

	const char* GetNaziv() const { return _naziv; }
	Kategorija GetKategorija() const { return _kategorija; }
	int GetCijena() const { return _cijena; }

	friend ostream& operator<<(ostream& COUT, const Proizvod& p) {

		COUT << p._naziv << " " << KategorijaNazivi[(int)p._kategorija] << " " << p._cijena;

		return COUT;
	}
};

class Transakcija {
protected:
	DatumVrijeme _vrijemeRealizacije;
	int _iznos;
public:
	Transakcija(DatumVrijeme vrijemeRealizacije, int iznos = 0) : _vrijemeRealizacije(vrijemeRealizacije), _iznos(iznos) {
	}

	Transakcija(const Transakcija& obj) {
		_vrijemeRealizacije = obj._vrijemeRealizacije;
		_iznos = obj._iznos;
	}

	Transakcija& operator=(const Transakcija& obj) {
		if (this != &obj) {
			_vrijemeRealizacije = obj._vrijemeRealizacije;
			_iznos = obj._iznos;
		}
		return *this;
	}

	bool operator==(const Transakcija& obj) {
		return _vrijemeRealizacije == obj._vrijemeRealizacije && _iznos == obj._iznos;
	}

	virtual ~Transakcija() {}

	virtual string Info() const = 0;

	const DatumVrijeme& GetVrijemeRealizacije() const {
		return
			_vrijemeRealizacije;
	}

	int GetIznos() const { return _iznos; }

	friend ostream& operator<<(ostream& COUT, const Transakcija& obj) {
		COUT << obj._vrijemeRealizacije << " " << obj._iznos << endl;
		return COUT;
	}

};

class Kupac {

			_sifra = AlocirajTekst(obj._sifra);

	KolekcijaParova<Kupac, int, 50> PotrosnjaPoKategoriji(Kategorija kategorija) {

		KolekcijaParova<Kupac, int, 50> potrosnja;

		// proci kroz sve kupce

		for (int i = 0; i < _kupci.size(); i++)
		{

			int ukupnaPotrosnja = 0;
			bool imaKupovinu = false;

			for (int j = 0; j < _kupci[i].GetTransakcije().size(); j++)
			{

				Kupovina* castedK = dynamic_cast<Kupovina*>(_kupci[i].GetTransakcije()[j]);
				Povrat* castedP = dynamic_cast<Povrat*>(_kupci[i].GetTransakcije()[j]);


				if (castedK) {


					for (int k = 0; k < castedK->GetProizvodi().size(); k++)
					{

						if (castedK->GetProizvodi()[k].GetKategorija() == kategorija) {

							ukupnaPotrosnja += castedK->GetProizvodi()[k].GetCijena();
							imaKupovinu = true;

						}


					}



				}
				else {

					for (int k = 0; k < castedP->GetProizvodi().size(); k++)
					{

						if (castedP->GetProizvodi()[k].GetKategorija() == kategorija) {

							ukupnaPotrosnja -= castedP->GetProizvodi()[k].GetCijena();

						}


					}


				}




			}


			if (imaKupovinu) {

				potrosnja.Dodaj(_kupci[i], ukupnaPotrosnja);

			}



		}


		return potrosnja;


	}



	~Prodavnica() {
		delete[] _naziv;
		_naziv = nullptr;
	}

	bool operator == (const Prodavnica& obj) {
		return strcmp(_naziv, obj._naziv) == 0;
	}

	const char* GetNaziv() const { return _naziv; }
	vector<Kupac>& GetKupci() { return _kupci; }

	friend ostream& operator << (ostream& COUT, const Prodavnica& obj) {

		COUT << "Naziv prodavnice: " << obj._naziv << endl;

		for (int i = 0; i < obj._kupci.size(); i++)
		{
			COUT << obj._kupci[i] << endl;
		}

		return COUT;
	}

};

bool UcitajPodatke(string nazivFajla, vector<Prodavnica>& prodavnice) {

	ifstream file(nazivFajla);

	if (!file.is_open()) {
		return false;
	}


	string eachRow;

	while (getline(file, eachRow)) {

		int pozicijaCrtice = eachRow.find('|');
		string imePrezime = eachRow.substr(0, pozicijaCrtice);
		string nazivProdavnice = eachRow.substr(pozicijaCrtice + 1);
		Prodavnica novaProdavnica(nazivProdavnice.c_str());
		bool prodavnicaExists = false;

		for (int i = 0; i < prodavnice.size(); i++) {
			if (novaProdavnica == prodavnice[i]) {
				prodavnicaExists = true;
			}
		}
		if (!prodavnicaExists) {
			prodavnice.push_back(novaProdavnica);
		}

		Kupac noviKupac(imePrezime.c_str());
		bool kupacExists = false;
		for (int i = 0; i < prodavnice.size(); i++) {
			if (prodavnice[i] == novaProdavnica) {
				for (int j = 0; j < prodavnice[i].GetKupci().size(); j++) {
					if (prodavnice[i].GetKupci()[j] == noviKupac) {
						kupacExists = true;

					}
				}
				if (!kupacExists) {
					prodavnice[i].DodajKupca(noviKupac);

				}
			}
		}

	}

	return true;



}
