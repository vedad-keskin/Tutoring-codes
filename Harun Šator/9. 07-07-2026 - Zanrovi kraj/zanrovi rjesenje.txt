char* AlocirajTekst(const char* tekst) { // GetNizKaraktera , AlocirajNiz
	if (!tekst) return nullptr;
	size_t vel = strlen(tekst) + 1;
	char* temp = new char[vel];
	strcpy_s(temp, vel, tekst);
	return temp;
}

//funkcija generise clanski broj na osnovu imena i prezimena, rednog broja i trenutne godine.
//clanski broj je u formatu GGGG/IN-BBB, gdje GGGG predstavlja trenutnu godinu, IN inicijale,
//a BBB redni broj clana popunjen nulama na tri mjesta.
//funkciju koristiti u konstruktoru klase ClanBiblioteke za inicijalizaciju atributa _clanskiBroj.

//cout << "Clanski broj OK" << crt;

string GenerisiClanskiBroj(string imePrezime, int redniBroj) {

	string clanskiBroj;

	// sistemko vrijeme (GGGG) 

	time_t utcVrijeme = time(0); // UTC 

	tm lokalnoVrijeme;

	localtime_s(&lokalnoVrijeme, &utcVrijeme);

	int godina = lokalnoVrijeme.tm_year + 1900; // 126 + 1900 = 2026

	// INT -> STRING
	clanskiBroj += to_string(godina);
	clanskiBroj += "/";

	// inicijali (IN) 

	char prvoSlovoImena = imePrezime[0];
	char prvoSlovoPrezimena = imePrezime[imePrezime.find(' ') + 1];


	clanskiBroj += prvoSlovoImena;
	clanskiBroj += prvoSlovoPrezimena;
	clanskiBroj += "-";

	// redni broj (BBB)

	if (redniBroj < 10) {

		clanskiBroj += "00";

	}
	else if (redniBroj < 100) {

		clanskiBroj += "0";

	}

	clanskiBroj += to_string(redniBroj);



	return clanskiBroj;

}

bool ValidirajClanskiBroj(const string& clanskiBroj) {

	regex uzorak("[0-9]{4}[/][A-Z]{2}[-][0-9]{3}");

	return regex_match(clanskiBroj, uzorak);


}

template<class T1, class T2, int max>
class Kolekcija {
	T1* _prvi;
	T2* _drugi;
	int _trenutno;
public:
	Kolekcija() : _prvi(nullptr), _drugi(nullptr), _trenutno(0) {

		_trenutno = 0;

		//_prvi = new T1[_trenutno];

	}

	Kolekcija(const Kolekcija& obj) {

		_trenutno = obj._trenutno;

		_prvi = new T1[_trenutno];
		_drugi = new T2[_trenutno];

		for (int i = 0; i < _trenutno; i++)
		{
			_prvi[i] = obj._prvi[i];
			_drugi[i] = obj._drugi[i];
		}

	}

	Kolekcija& operator = (const Kolekcija& obj) {
		if (this != &obj) {

			delete[] _prvi;
			delete[] _drugi;

			_trenutno = obj._trenutno;

			_prvi = new T1[_trenutno];
			_drugi = new T2[_trenutno];

			for (int i = 0; i < _trenutno; i++)
			{
				_prvi[i] = obj._prvi[i];
				_drugi[i] = obj._drugi[i];
			}
		}
		return *this;
	}

	void Dodaj(T1 el1, T2 el2) {



		T1* temp1 = _prvi;
		T2* temp2 = _drugi;

		_prvi = new T1[_trenutno + 1];
		_drugi = new T2[_trenutno + 1];

		for (int i = 0; i < _trenutno; i++)
		{
			_prvi[i] = temp1[i];
			_drugi[i] = temp2[i];
		}

		_prvi[_trenutno] = el1;
		_drugi[_trenutno] = el2;

		_trenutno++;

		delete[]temp1;
		delete[]temp2;


	}


	//DodajNaPoziciju dodaje novi par na lokaciju definisanu prvim parametrom. metoda vraca trenutno stanje kolekcije.
	//u slucaju popunjene kolekcije ili neispravne lokacije potrebno je baciti izuzetak (za potrebe testiranja mozete dodati try catch blok).
	//Kolekcija<int, string, 20> prosireniInventar = inventar.DodajNaPoziciju(2, 99, "Posebno izdanje");
	//cout << prosireniInventar << crt;


	Kolekcija DodajNaPoziciju(int lokacija, T1 el1, T2 el2) {

		if (lokacija < 0 || lokacija >= _trenutno) {

			throw exception("Lokacija nije validna");

		}

		Kolekcija prosireniInventar;

		for (int i = 0; i < _trenutno; i++)
		{

			if (i == lokacija) { // 2 == 2
				prosireniInventar.Dodaj(el1, el2);
			}

			prosireniInventar.Dodaj(_prvi[i], _drugi[i]);

		}



		return prosireniInventar;

	}

	//UkloniRaspon uklanja broj elemenata definisan drugim parametrom, pocevsi od lokacije
	//definisane prvim parametrom (ukljucujuci tu lokaciju). metoda vraca pokazivac na novu kolekciju sa uklonjenim
	//elementima, a pozivalac je odgovoran za njenu dealokaciju.
	//Kolekcija<int, string, 20>* uklonjeneKnjige = prosireniInventar.UkloniRaspon(3, 2);


	Kolekcija* UkloniRaspon(int lokacijaOd, int brojElementa) {

		if (lokacijaOd < 0 || lokacijaOd + brojElementa > _trenutno) {

			throw exception("Opseg ne postoji");

		}


		Kolekcija* uklonjeneKnjige = new Kolekcija();

		Kolekcija preostaleKnjige;


		for (int i = 0; i < _trenutno; i++)
		{

			if (i >= lokacijaOd && i < lokacijaOd + brojElementa) {

				uklonjeneKnjige->Dodaj(_prvi[i], _drugi[i]);

			}
			else {

				preostaleKnjige.Dodaj(_prvi[i], _drugi[i]);

			}
		}


		*this = preostaleKnjige;


		return uklonjeneKnjige;
	}


	int GetTrenutno() const { return _trenutno; }
	T1& GetPrvi(int indeks) const { return _prvi[indeks]; }
	T2& GetDrugi(int indeks) const { return _drugi[indeks]; }
	T1& operator[](int indeks) const { return _prvi[indeks]; }

	friend ostream& operator<<(ostream& COUT, const Kolekcija& obj) {
		for (int i = 0; i < obj.GetTrenutno(); i++)
			COUT << obj.GetPrvi(i) << " " << obj.GetDrugi(i) << endl;
		return COUT;
	}

	~Kolekcija() {
		delete[] _prvi; _prvi = nullptr;
		delete[] _drugi; _drugi = nullptr;
	}
};

class Datum {
	int* _godina, * _mjesec, * _dan;
public:
	Datum(int dan = 1, int mjesec = 1, int godina = 2000) {
		_godina = new int(godina);
		_mjesec = new int(mjesec);
		_dan = new int(dan);
	}

	Datum(const Datum& obj) {

		_godina = new int(*obj._godina);
		_mjesec = new int(*obj._mjesec);
		_dan = new int(*obj._dan);

	}

	Datum& operator = (const Datum& obj) {
		if (this != &obj) {
			delete _godina;
			delete _mjesec;
			delete _dan;


			_godina = new int(*obj._godina);
			_mjesec = new int(*obj._mjesec);
			_dan = new int(*obj._dan);
		}
		return *this;
	}


	bool operator == (const Datum obj) {
		return *_dan == *obj._dan && *_mjesec == *obj._mjesec && *_godina == *obj._godina;
	}

	~Datum() {
		delete _dan; _dan = nullptr;
		delete _mjesec; _mjesec = nullptr;
		delete _godina; _godina = nullptr;
	}

	friend ostream& operator<< (ostream& COUT, const Datum& obj) {
		COUT << setw(2) << setfill('0') << *obj._dan << "." << setw(2) << setfill('0') << *obj._mjesec << "." << *obj._godina;
		return COUT;
	}


};

	}

};

class ClanBiblioteke {
	static int _id;
	char* _clanskiBroj;
	char* _imePrezime;
	vector<Posudba> _posudbe;

	// DODAJ -> AddElement/Dodaj -> _posudbe.push_back(posudba);

	// VELICINA -> _trenutno/GetTrenutno() -> _posudbe.size()

	// UKLONITI -> .............. ->  _posudbe.erase(_posudbe.begin() + i); -> iterator + pozicija


public:
	ClanBiblioteke(const char* imePrezime = "") {
		_imePrezime = AlocirajTekst(imePrezime);
		_clanskiBroj = AlocirajTekst(GenerisiClanskiBroj(imePrezime, _id).c_str());
		_id++;
	}

	ClanBiblioteke(const ClanBiblioteke& obj) {

		_clanskiBroj = AlocirajTekst(obj._clanskiBroj);
		_imePrezime = AlocirajTekst(obj._imePrezime);

		_posudbe = obj._posudbe;

	}

	ClanBiblioteke& operator = (const ClanBiblioteke& obj) {
		if (this != &obj) {
			delete[] _clanskiBroj;
			delete[] _imePrezime;

			_clanskiBroj = AlocirajTekst(obj._clanskiBroj);
			_imePrezime = AlocirajTekst(obj._imePrezime);

			_posudbe = obj._posudbe;
		}
		return *this;
	}


	//DodajPosudbu dodaje posudbu ako clan nema ranije zaduzenu knjigu istog naslova,
	//ako trenutno ima manje od tri posudbe i ako je broj dana posudbe u rasponu od 1 do 30.
	//metoda vraca true ako je posudba dodana, u suprotnom vraca false.
	//if (amina.DodajPosudbu(tvrdjava))


	bool DodajPosudbu(Posudba posudba) {


		//       3 
		if (_posudbe.size() >= 3 || posudba.GetBrojDana() < 1 || posudba.GetBrojDana() > 30) {
			return false;
		}

		//if (posudba.GetBrojDana() < 1 || posudba.GetBrojDana() > 30) {

		//	return false;

		//}


		for (int i = 0; i < _posudbe.size(); i++)
		{

			if (_posudbe[i] == posudba) {

				return false;

			}

		}



		_posudbe.push_back(posudba);

		return true;


	}

	//RazduziKnjigu uklanja posudbu na osnovu naslova knjige i vraca true ako je pronadjena
	//i uklonjena. ukoliko knjiga nije pronadjena metoda vraca false.
	//if (amina.RazduziKnjigu("Tvrdjava"))


	bool RazduziKnjigu(const char* naslovKnjige) {

		for (int i = 0; i < _posudbe.size(); i++)
		{

			if (strcmp(naslovKnjige , _posudbe[i].GetNaslov()) == 0) {

				_posudbe.erase(_posudbe.begin() + i);

				return true;

			}

		}


		return false;
	}


	bool operator == (const ClanBiblioteke& obj) {
		return strcmp(_clanskiBroj, obj._clanskiBroj) == 0;
	}

	~ClanBiblioteke() {
		delete[] _clanskiBroj; _clanskiBroj = nullptr;
		delete[] _imePrezime; _imePrezime = nullptr;
	}

	const char* GetClanskiBroj() const { return _clanskiBroj; }
	const char* GetImePrezime() const { return _imePrezime; }
	vector<Posudba>& GetPosudbe() { return _posudbe; }

	friend ostream& operator<<(ostream& COUT, const ClanBiblioteke& obj) {
		COUT << obj._imePrezime << " [" << obj._clanskiBroj << "]" << endl;
		for (auto& posudba : obj._posudbe) // foreach
			//ToString metoda vraca podatke o posudbi u formatu prikazanom u nastavku.
			//voditi racuna o prikazu jednocifrenih vrijednosti datuma (npr. 5 -> 05).
			//05.10.2026 Tvrdjava ROMAN 14 dana
			COUT << " - " << posudba.ToString() << endl;
		return COUT;
	}
};

class Biblioteka {
	char* _naziv;
	vector<ClanBiblioteke> _clanovi;
public:
	Biblioteka(const char* naziv) {
		_naziv = AlocirajTekst(naziv); 
	}

	Biblioteka(const Biblioteka& obj) {
		_naziv = AlocirajTekst(obj._naziv);
		_clanovi = obj._clanovi;
	}

	Biblioteka& operator = (const Biblioteka& obj) {
		if (this != &obj) {
			delete[] _naziv;

			_naziv = AlocirajTekst(obj._naziv);
			_clanovi = obj._clanovi;
		}
		return *this;
	}

	bool operator == (const Biblioteka& obj) {
		return strcmp(_naziv, obj._naziv) == 0;
	}

	void DodajClana(ClanBiblioteke clan) {

		for (int i = 0; i < _clanovi.size(); i++)
		{
			if (clan == _clanovi[i]) {
				throw exception("Clan je vec dio biblioteke");
			}
		}


		_clanovi.push_back(clan);

	}

	//EvidentirajPosudbu pronalazi clana na osnovu clanskog broja i dodaje mu posudbu.
	//i dalje vaze pravila definisana u metodi DodajPosudbu. metoda vraca true ili false.
	//if (gradska.EvidentirajPosudbu(goran.GetClanskiBroj(), pjesme))

	bool EvidentirajPosudbu(const char* clanskiBrojClana, Posudba posudba) {

		for (int i = 0; i < _clanovi.size(); i++)
		{

			if (strcmp(clanskiBrojClana , _clanovi[i].GetClanskiBroj()) == 0) {

				_clanovi[i].DodajPosudbu(posudba);

				return true;

			}

		}

		return false;

	}

	//AktivniClanovi vraca pokazivace na clanove koji imaju najmanje onoliko posudbi
	//koliko je definisano vrijednoscu proslijedjenog parametra.
	//vector<ClanBiblioteke*> aktivni = gradska.AktivniClanovi(1);

	vector<ClanBiblioteke*> AktivniClanovi(int brojPosudbi) {

		vector<ClanBiblioteke*> aktivni;

		for (int i = 0; i < _clanovi.size(); i++)
		{
			// 2 .. 1 
			if (_clanovi[i].GetPosudbe().size() >= brojPosudbi) {

				aktivni.push_back(new ClanBiblioteke(_clanovi[i]));

			}


		}

		return aktivni;

	}

	//PosudbePoZanru vraca kolekciju parova (clan, broj posudbi) za sve clanove koji imaju
	//najmanje jednu aktivnu posudbu knjige proslijedjenog zanra.
	//Kolekcija<ClanBiblioteke, int, 50> strucnePosudbe = gradska.PosudbePoZanru(STRUCNA_LITERATURA);


	Kolekcija<ClanBiblioteke, int, 50> PosudbePoZanru(Zanr zanr) {

		Kolekcija<ClanBiblioteke, int, 50> strucnePosudbe;


		// prolazi kroz clanove

		for (int i = 0; i < _clanovi.size(); i++)
		{

			int brojPosudbiUZanru = 0;

			// prolazi kroz posudbe tog clana

			for (int j = 0; j < _clanovi[i].GetPosudbe().size(); j++)
			{

				if (_clanovi[i].GetPosudbe()[j].GetZanr() == zanr) {

					brojPosudbiUZanru++;

				}

			}


			if (brojPosudbiUZanru > 0) {

				strucnePosudbe.Dodaj(_clanovi[i], brojPosudbiUZanru);

			}



		}


		return strucnePosudbe;




	}


	~Biblioteka() {
		delete[] _naziv;
		_naziv = nullptr;
	}


	const char* GetNaziv() const { return _naziv; }
	vector<ClanBiblioteke>& GetClanovi() { return _clanovi; }


	friend ostream& operator << (ostream& COUT, const Biblioteka& obj) {

		COUT << obj._naziv << endl;

		for (int i = 0; i < obj._clanovi.size(); i++)
		{
			COUT << obj._clanovi[i];
		}

		return COUT;
	}

};
