// 1. ALOKACIJA ( new ) 

// -- ako alociramo jednu vrijednost -> ( ) 

// -- ako alociramo vise vrijednost -> [ ]

// 2. DEREFERENCIRANJE ( * ) 

// -- ako alociramo jednu vrijednost -> ( ) --> _ECTS = 00000242675F8960 , *_ECTS = 0

// -- ako alociramo vise vrijednost -> [ ] --> _naziv = Nema predmeta , *_naziv = N

// 3. DESTRUKTOR / DEALOKACIJA

// -- ako alociramo jednu vrijednost -> ( ) --> delete _ECTS;

// -- ako alociramo vise vrijednost -> [ ] --> delete[] _naziv;

class Predmet {
public:			   // dft. constr.
	char* _naziv;  // Nema predmeta
	//char _naziv[50];
	int _ocjena;   //      5
	int* _ECTS;	   //      0

	// dft. constr.

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char dftNaziv[] = "Nema predmeta"; // 14 char 
		//string dftNaziv = "Nema predmeta"; 
		const char* dftNaziv = "Nema predmeta"; // STRING LITERAL --> read-only mem. 

		int vel = strlen(dftNaziv) + 1; // 13 char + \0 null terminator

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, dftNaziv);

	}

	// user-def constr.
	//Predmet predmet1("Programiranje II", 9, 7); // user-def constr.

	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);

	}

	// copy constr.

	//Predmet predmetKopija2 ( predmet2 );	 // copy constr. --> DEEP COPY , SHALLOW COPY

	// bez & --> pravimo kopiju predmeta
	// sa & --> saljemo original
	// const --> konstantno -> ne moze se mijenajti

	Predmet(const Predmet& predmet) {

		//predmet._ocjena = 10;

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);

	}

	//predmetOperatorDodjele = predmet1;

	Predmet& operator = (const Predmet& predmet) {

		// adresa lijevog predmetOperatorDodjele != adresa desnog predm
		if (this != &predmet) {


			delete _ECTS;

			delete[] _naziv;

			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);

		}


		return *this;

	}





	// destur. TILDA -> alt gr + 1

	~Predmet() {

		delete _ECTS;

		_ECTS = nullptr;

		delete[] _naziv;

		_naziv = nullptr;

	}


	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;

		cout << "ECTS predmeta: " << *_ECTS << endl;

		cout << "Ocjena predmeta: " << _ocjena << endl << endl;

	}



};



class Student {
	char* _imePrezime;

	Predmet* _polozeni; // [ ] 
	int _trenutno;      // koja nam kaze koliko trenutno imamo polozeinh predmeta
	int _max;           // koliko maksimalno mozemo imati polozenih predmeta
public:


	Student() {

		const char* dftImePrezime = "---";

		int vel = strlen(dftImePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, dftImePrezime);


		_trenutno = 0;

		_max = 3;

		_polozeni = new Predmet[_max];
	}

	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, imeprezime);

		_trenutno = 0;
		_max = max;

		_polozeni = new Predmet[_max];

	}

	Student(const Student& obj) {

		int vel = strlen(obj._imePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, obj._imePrezime);

		_trenutno = obj._trenutno;

		_max = obj._max;

		_polozeni = new Predmet[_max]; // kada god alociramo niz taj niz ce biit prazan 

		for (int i = 0; i < _trenutno; i++) // 0 1 2 
		{
			_polozeni[i] = obj._polozeni[i]; // op = Predmet
		}

	}

	Student& operator = (const Student& obj) {
		if (this != &obj) {

			delete[] _imePrezime;
			delete[] _polozeni;


			int vel = strlen(obj._imePrezime) + 1;
			_imePrezime = new char[vel];
			strcpy_s(_imePrezime, vel, obj._imePrezime);

			_trenutno = obj._trenutno;

			_max = obj._max;

			_polozeni = new Predmet[_max]; // kada god alociramo niz taj niz ce biit prazan 

			for (int i = 0; i < _trenutno; i++) // 0 1 2 
			{
				_polozeni[i] = obj._polozeni[i]; // op = Predmet
			}
		}
		return *this;
	}


	~Student() {

		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _polozeni;
		_polozeni = nullptr;

	}

	void Ispis() {


		cout << "Ime i prezime studenta: " << _imePrezime << endl;

		cout << "Polozeni predmeti studenta: " << endl << endl;

		for (int i = 0; i < _trenutno; i++) // 0 1  
		{
			_polozeni[i].Ispis();
		}

	}















	//student.AddPolozeniPredmet(predmet1);

	void AddPolozeniPredmet(Predmet predmet) { // MATII

		// Voditi računa da se ne može prekoračiti kapacitet niza

		if (_trenutno == _max) {
			
			cout << "Prekoracen maksimalan broj polozenih predmeta" << endl;

			return;

		}


		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		        

		for (int i = 0; i < _trenutno; i++) // 0 1 
		{

			//           char* == char*



			if (strcmp(predmet._naziv , _polozeni[i]._naziv) == 0) {

				cout << "Odabrani predmet je vec dodan" << endl;

				return;

			}

		}




		// Dodaje novi polozeni predmet u niz

		_polozeni[_trenutno] = predmet; // _pol[0] = PRII , pol[1] = MATII 

		_trenutno++; // 0 ---> 1 ---> 2

	}

};
