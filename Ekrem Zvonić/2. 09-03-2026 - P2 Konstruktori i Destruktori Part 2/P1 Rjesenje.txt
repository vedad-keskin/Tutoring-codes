#include <iostream>

// POKAZIVACI

// 1. ALOKACIJA ( new )

// -- jedna vrijednost ( ) 

// -- pravimo niz [ ]

// 2. DEREFERENCIRANJE

// -- jedna vrijednost ( ) -> mora se napisati * ispred *_ECTS -> 0

// -- pravimo niz [ ] -> ne pise se * -> *_naziv --> N 

// 3. DESTURKTOR / DEALOKACIJA 

// -- jedna vrijednost ( ) -> delete

// -- pravimo niz [ ] --> delete[]


class Predmet {
public:			   //  dft const. 	// user-def constr. 
	char* _naziv;   // Nema predmeta	// Programiranje II
	//char _naziv[50];
	int _ocjena;	   //       5		//      9
	int* _ECTS;	   //       0 		//      7

	// dft. konstruktor

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char tekst1[] = "Nema predmeta";
		//string tekst2 = "Nema predmeta";
		const char* tekst3 = "Nema predmeta"; // string literal --> read-only mem.

		int vel = strlen(tekst3) + 1; // null terminator

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, tekst3); // 1. gdje nesto hocemo da pohranimo , 2. koliko slova zelite prekopirati , 3. sta zelimo pohraniti 


	}

	// user-def constr.

	//Predmet predmet1("Programiranje II", 9, 7);
	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);

	}

	// copy constr. 
	//Predmet predmetKopija2(predmet1); // <-- copy constr. 

	// & -> posalji original 
	// bez & -> posalji kopiju 
	// const -> kontratno i ne moze se mijenajti



	//                     predmet1 -> PRII 
	Predmet(const Predmet& predmet) { // napravila bi se kopija tog predmeta

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);


	}

	// operator dodjele

	// predmetOperatorDodjele = predmet1; // <-- ne spada pod copy constr. 

	Predmet& operator = (const Predmet& predmet) {

		if (this != &predmet) {

			delete _ECTS;
			delete[] _naziv;




			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);

		}

		return *this;

	}



	// dealociranje / destruktor

	// TILDA -> alt gr + 1 ~~

	~Predmet() {


		delete _ECTS;
		_ECTS = nullptr;

		delete[] _naziv;
		_naziv = nullptr;

	}




	// ispis

	void Ispis() {


		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;


	}



};





class Student {
	char* _imePrezime; // niz karaktera [ ] char 

	Predmet* _polozeni; // pok na neku klasu --> ( ) ili [ ] 
	int _trenutno;      // nam kaze koliko trenutno imamo predmeta u nizu 
	int _max;           // nam kaze koliko maksimalno predmeta mozemo imati u nizu
public:



	// dft constr.

	Student() {

		const char* dftImePrezime = "Nema imena i prezimena";

		int vel = strlen(dftImePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, dftImePrezime);

		_trenutno = 0;

		_max = 3;

		_polozeni = new Predmet[_max];


	}

	// user-def constr. 

	//Student student1("Vedad Keskin", 3);

	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, imeprezime);


		_trenutno = 0;

		_max = max;

		_polozeni = new Predmet[_max];


	}

	Student(const Student& obj) {

		int vel = strlen(obj._imePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, obj._imePrezime);

		_trenutno = obj._trenutno;

		_max = obj._max;

		_polozeni = new Predmet[_max];

		// KADA RADIMO KOPIRANJE NEKOG NIZA (IZUZETAK JE NIZ KAR) UVIJEK JE POTREBNO DODATI JOS LOGIKE KOJA CE TAJ NIZ PREKOPIRATI

		//for (int i = 0; i < obj._trenutno; i++)
		for (int i = 0; i < _trenutno; i++)
		{

			_polozeni[i] = obj._polozeni[i]; // Predmet = Predmet ---> operator dodjele od klase Predmet

		}


	}


	Student& operator = (const Student& obj) {
		if (this != &obj) {



			delete[] _imePrezime;
			delete[] _polozeni;




			int vel = strlen(obj._imePrezime) + 1;

			_imePrezime = new char[vel];

			strcpy_s(_imePrezime, vel, obj._imePrezime);

			_trenutno = obj._trenutno;

			_max = obj._max;

			_polozeni = new Predmet[_max];

			// KADA RADIMO KOPIRANJE NEKOG NIZA (IZUZETAK JE NIZ KAR) UVIJEK JE POTREBNO DODATI JOS LOGIKE KOJA CE TAJ NIZ PREKOPIRATI

			//for (int i = 0; i < obj._trenutno; i++)
			for (int i = 0; i < _trenutno; i++)
			{

				_polozeni[i] = obj._polozeni[i]; // Predmet = Predmet ---> operator dodjele od klase Predmet

			}
		}
		return *this;
	}



	~Student() {

		delete[] _imePrezime;
		_imePrezime = nullptr;

		delete[] _polozeni;
		_polozeni = nullptr;

	}

	
	void Ispis() {

		cout << "Ime i prezime studenta: " << _imePrezime << endl;

		cout << "== POLOZENI PREDMETI STUDENTA ==" << endl << endl;

		for (int i = 0; i < _trenutno; i++) // 2 ---> 0 , 1
		{

			_polozeni[i].Ispis(); // pol[0] = PRII , pol[1] = MATII 

		}


	}













	//student1.AddPolozeniPredmet(predmet2); // MATII


	void AddPolozeniPredmet(Predmet predmet) { // predmet.naziv -- MATII 



		// Voditi računa da se ne može prekoračiti kapacitet niza

		//   3   == 3 
		if (_max == _trenutno) {

			cout << "Prekoracen kapacitet niza" << endl;
			return;

		}


		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		        


		for (int i = 0; i < _trenutno; i++) // 0 1 .. 
		{
			//    MATII        ==  pol[0] = PRII , pol[1] = MATII 

			//    char*  == char* 

			// kada poredimo niz karaktera -->   strcmp( naziv1 , naziv2 ) == 0 

			if ( strcmp( predmet._naziv , _polozeni[i]._naziv ) == 0) {

				cout << "Dodati predmet je vec polozen od strane studenta" << endl;
				return;

			}


		}







		// Dodaje novi polozeni predmet u niz
		_polozeni[_trenutno] = predmet; // PRII --> pol[0]  ,  pol[1] = MATII

		_trenutno++; // 0 ---> 1 ---> 2

	}

};
