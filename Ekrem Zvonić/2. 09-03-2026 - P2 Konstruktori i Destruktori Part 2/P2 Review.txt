class Izbori {
public:
	char* _glasackoMjesto;
	GlasackiListic* _glasovi;
	int _trenutnoGlasova;
	int _maxBrojGlasova;
	//Kreirati i implementirati
	//1. Defaultni konstruktor
	//2. Konstruktor koji prima parametre
	//3. Konstruktor kopije
	//4. Operator dodjele
	//5. Destruktor
	//6. Operator ispisa 
	//7. Potrebne funkcije

	Izbori() {
		const char* tekst1 = "Nema dostupnog glasackog mjesta";

		int vel = strlen(tekst1) + 1;

		_glasackoMjesto = new char[vel];

		strcpy_s(_glasackoMjesto, vel, tekst1);

		_trenutnoGlasova = 0;

		_maxBrojGlasova = 3;

		_glasovi = new GlasackiListic[_maxBrojGlasova];
	}
	Izbori(const char* glasackomjesto, int maxbrojglasova) {
		int vel = strlen(glasackomjesto) + 1;

		_glasackoMjesto = new char[vel];

		strcpy_s(_glasackoMjesto, vel, glasackomjesto);

		_trenutnoGlasova = 0;

		_maxBrojGlasova = maxbrojglasova;

		_glasovi = new GlasackiListic[_maxBrojGlasova];
	}
	Izbori(const Izbori& obj) {
		int vel = strlen(obj._glasackoMjesto) + 1;

		_glasackoMjesto = new char[vel];

		strcpy_s(_glasackoMjesto, vel, obj._glasackoMjesto);

		_trenutnoGlasova = obj._trenutnoGlasova;

		_maxBrojGlasova = obj._maxBrojGlasova;

		_glasovi = new GlasackiListic[_maxBrojGlasova];

		for (int i = 0; i < obj._trenutnoGlasova; i++)
		{
			_glasovi[i] = obj._glasovi[i];
		}
	}
	Izbori& operator = (const Izbori& obj) {
		if (this != &obj)
		{
			delete[] _glasackoMjesto;

			delete[] _glasovi;

			int vel = strlen(obj._glasackoMjesto) + 1;

			_glasackoMjesto = new char[vel];

			strcpy_s(_glasackoMjesto, vel, obj._glasackoMjesto);

			_trenutnoGlasova = obj._trenutnoGlasova;

			_maxBrojGlasova = obj._maxBrojGlasova;

			_glasovi = new GlasackiListic[_maxBrojGlasova];

			for (int i = 0; i < obj._trenutnoGlasova; i++)
			{
				_glasovi[i] = obj._glasovi[i];
			}
		}
		return *this;
	}
	~Izbori() {
		delete[] _glasackoMjesto;
		_glasackoMjesto = nullptr;

		delete[] _glasovi;
		_glasovi = nullptr;
	}
	void Ispis() {
		cout << "Glasacko mjesto: " << _glasackoMjesto << endl;
		cout << "== GLASOVI ==" << endl << endl;

		for (int i = 0; i < _trenutnoGlasova; i++)
		{
			_glasovi[i].Ispis();
		}
	}
	bool AddGlasackiListic(GlasackiListic listic)
	{
		// 1. Onemogucavanje dodavanja vise listica nego sto je dozvoljeno
		if (_maxBrojGlasova == _trenutnoGlasova) {
			cout << "Prekoracen kapacitet niza" << endl;
			return false;
		}

		// Provjera da ista osoba ne glasa više puta
		for (int i = 0; i < _trenutnoGlasova; i++) {
			if (strcmp(_glasovi[i]._glasac->_JMBG, listic._glasac->_JMBG) == 0) {
				cout << "Ovaj glasac je vec glasao" << endl;
				return false;
			}
		}

		// 3. Provjera starosti (osoba mora imati 18 ili više godina)
		int godinaRodjenja = *listic._glasac->_datumRodjenja._godina;
		int trenutnaGodina = 2026;
		if ((trenutnaGodina - godinaRodjenja) < 18)
			return false;


		// 4. Provjera prebivališta (osoba može glasati samo na svom mjestu)
		if (strcmp(_glasackoMjesto, listic._glasac->_prebivaliste) != 0) {
			cout << "Glasac nije sa ovog glasackog mjesta" << endl;
			return false;
		}

		// 5. Ako je prošao sve provjere, dodaj glas
		_glasovi[_trenutnoGlasova] = listic;
		_trenutnoGlasova++;

		return true;// uspješno dodan glas
	}


};
