#include <iostream>

char* GetNizKaraktera(const char* tekst) {

	int vel = strlen(tekst) + 1;

	char* novi = new char[vel];

	strcpy_s(novi, vel, tekst);

	return novi;
}


class Osoba {
protected:
	char* _imePrezime;
	char* _JMBG;
public:
	//Kreirati i implementirati
	//1. Defaultni konstruktor
	//2. Konstruktor koji prima parametre
	//3. Konstruktor kopije
	//4. Operator dodjele
	//5. Destruktor
	//6. Operator ispisa 

	Osoba(const char* imePrezime="---", const char* jmbg="---") {

		_imePrezime = GetNizKaraktera(imePrezime);

		_JMBG = GetNizKaraktera(jmbg);
	}

	Osoba(const Osoba& obj) {

		_imePrezime = GetNizKaraktera(obj._imePrezime);

		_JMBG = GetNizKaraktera(obj._JMBG);
	}

	Osoba& operator = (const Osoba& obj) {
		if (this!=&obj)
		{
			delete[] _imePrezime;

			delete[] _JMBG;


			_imePrezime = GetNizKaraktera(obj._imePrezime);

			_JMBG = GetNizKaraktera(obj._JMBG);
			
		}
		return *this;
	}

	virtual void Info() = 0;

	virtual ~Osoba() {
		delete[] _imePrezime;
		_imePrezime = nullptr;

		delete[] _JMBG;
		_JMBG = nullptr;
	}

	friend ostream& operator << (ostream& COUT, const Osoba& obj) {

		COUT << obj._imePrezime << "[" << obj._JMBG << "]" << endl;

		return COUT;
	}
};



class Profesor : public Osoba{
public:
	char* _titula;

	//3. Konstruktor kopije

	Profesor(const char* titula="prof.",const char* imeprezime="---",const char* jmbg="---") : Osoba(imeprezime,jmbg) {

class Predmet {
public:
	char* _naziv;
	int* _ECTS;
	Profesor* _predmetniProfesor;
	//Kreirati i implementirati
	//1. Defaultni konstruktor
	//2. Konstruktor koji prima parametre
	//3. Konstruktor kopije
	//4. Operator dodjele
	//5. Destruktor
	//6. Operator ispisa 

	Predmet(const char* naziv = "---", int ects = 0, Profesor predmetniProfesor = Profesor()) {

		_naziv = GetNizKaraktera(naziv);

		_ECTS = new int(ects);

		_predmetniProfesor = new Profesor(predmetniProfesor);
	}

	Predmet(const Predmet& obj) {
		_naziv = GetNizKaraktera(obj._naziv);

		_ECTS = new int(*obj._ECTS);

		_predmetniProfesor = new Profesor(*obj._predmetniProfesor);

	}
	Predmet& operator = (const Predmet& obj) {
		if (this!=&obj)
		{

			delete[] _naziv;
			
			delete _ECTS;
			
			delete _predmetniProfesor;

			_naziv = GetNizKaraktera(obj._naziv);

			_ECTS = new int(*obj._ECTS);

			_predmetniProfesor = new Profesor(*obj._predmetniProfesor);
			
		}
		return *this;
	}
	~Predmet() {
		delete[] _naziv;
		_naziv = nullptr;

		delete _ECTS;
		_ECTS = nullptr;

		delete _predmetniProfesor;
		_predmetniProfesor = nullptr;
	}

	friend ostream& operator << (ostream& COUT, const Predmet& obj) {

		COUT << "Naziv: " << obj._naziv << endl;
		COUT << "ECTS: " << *obj._ECTS << endl;

		COUT << "Profesor: " << *obj._predmetniProfesor << endl;
		

		return COUT;
	}
	
	
};



class Student : public Osoba{
public:
	char* _brojIndeksa;
	Predmet* _polozeniPredmeti;
	int* _trenutno;
	int _max;
	//Kreirati i implementirati
	//1. Defaultni konstruktor
	//2. Konstruktor koji prima parametre
	//3. Konstruktor kopije
	//4. Operator dodjele
	//5. Destruktor
	//6. Operator ispisa 
	//7. Potrebne funkcije

	Student(const char* brojIndexa="---",int max=3,const char* imeprezime="---",const char* jmbg="---") : Osoba(imeprezime,jmbg){

		_brojIndeksa = GetNizKaraktera(brojIndexa);

		_trenutno = new int(0);

		_max = max;

		_polozeniPredmeti = new Predmet[_max];
	}
	Student(const Student& obj) : Osoba(obj) {
		_brojIndeksa = GetNizKaraktera(obj._brojIndeksa);

		_trenutno = new int(*obj._trenutno);

		_max = obj._max;

		_polozeniPredmeti = new Predmet[_max];

		for (int i = 0; i < *obj._trenutno; i++)
		{
			_polozeniPredmeti[i] = obj._polozeniPredmeti[i];
		}
	}
	Student& operator = (const Student& obj) {
		if (this!=&obj)
		{
			delete[] _brojIndeksa;
			
			delete _trenutno;
			
			delete[] _polozeniPredmeti;


			Osoba::operator=(obj);


			_brojIndeksa = GetNizKaraktera(obj._brojIndeksa);

			_trenutno = new int(*obj._trenutno);

			_max = obj._max;

			_polozeniPredmeti = new Predmet[_max];

			for (int i = 0; i < *obj._trenutno; i++)
			{
				_polozeniPredmeti[i] = obj._polozeniPredmeti[i];
			}
		}
		return *this;
	}
	void Info() {
		cout << *this;
	}

	~Student() {
		delete[] _brojIndeksa;
		_brojIndeksa = nullptr;

		delete _trenutno;
		_trenutno = nullptr;

		delete[] _polozeniPredmeti;
		_polozeniPredmeti = nullptr;
	}

	friend ostream& operator << (ostream& COUT, const Student& obj) {
		COUT << "===== Student =====" << endl;
		COUT << "Ime i prezime: " << obj._imePrezime << endl;
		COUT << "JMBG: " << obj._JMBG << endl;
		COUT << "Broj indeksa: " << obj._brojIndeksa << endl;

		COUT << "==POLOZENI PREDMETI=="<<endl;

		for (int i = 0; i < *obj._trenutno; i++)
		{
			COUT<<obj._polozeniPredmeti[i];
		}

		return COUT;
	}

	void AddPolozeniPredmet(Predmet predmet) {
		if (_max==*_trenutno)
		{

			cout << "Prekoracen kapacitet niza" << endl;
			return;
		}


		for (int i = 0; i < *_trenutno; i++) // 0 1 .. 
		{
			if (strcmp(predmet._naziv, _polozeniPredmeti[i]._naziv) == 0) {

				cout << "Dodati predmet je vec polozen od strane studenta" << endl;
				return;

			}


		}

		_polozeniPredmeti[*_trenutno] = predmet;
		(*_trenutno)++;
	}
};
