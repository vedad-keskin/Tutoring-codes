string GenerisiSifru(const string& imePrezime, const int broj) {
	string sifra{};
	const char ime = imePrezime[0];
	const char prezime = imePrezime.find(' ') + 1;
	sifra += toupper(ime);
	sifra += toupper(imePrezime[prezime]);
	sifra += ":";

	time_t vrijeme = time(0);
	tm sada;
	localtime_s(&sada, &vrijeme);
	int godina = sada.tm_year + 1900;
	sifra += to_string(godina);
	sifra += "-";

	if (broj < 10)
		sifra += "00";
	else if (broj < 100)
		sifra += "0";
	sifra += to_string(broj);
	return sifra;
}

bool ValidirajSifru(const char* sifra) {
	regex reg("[A-B]{2}[:][0-9]{4}[-][0-9]{3}");
	return regex_match(sifra, reg);
}



template<class T1, class T2, int max>
class KolekcijaParova {
	T1* _prvi[max] = { nullptr };
	T2* _drugi[max] = { nullptr };
	int _trenutno = 0;
public:
	KolekcijaParova() = default;

	void Dodaj(const T1& br1, const T2& br2) {
		if (GetTrenutno() == max)
			return;

		_prvi[GetTrenutno()] = new T1(br1);
		_drugi[GetTrenutno()] = new T2(br2);
		_trenutno++;

	}

	KolekcijaParova(const KolekcijaParova& obj) {
		_trenutno = obj.GetTrenutno();

		for (int i = 0; i < GetTrenutno(); i++)
		{
			_prvi[i] = new T1(obj.GetPrvi(i));
			_drugi[i] = new T2(obj.GetDrugi(i));
		}

	}
	KolekcijaParova& operator=(const KolekcijaParova& obj) {
		if (this == &obj)
			return *this;

		for (int i = 0; i < _trenutno; i++) {
			delete _prvi[i]; 
			delete _drugi[i];
		}

		_trenutno = obj.GetTrenutno();
		for (int i = 0; i < GetTrenutno(); i++)
		{
			_prvi[i] = new T1(obj.GetPrvi(i));
			_drugi[i] = new T2(obj.GetDrugi(i));
		}
		return *this;
	}
	//KolekcijaParova& DodajNaPoziciju(const int lokacija, const T1& br1, const T2& br2) {
	//	for (int i = GetTrenutno(); i > lokacija; i--)
	//	{

	//		_drugi[i] = _drugi[i - 1];
	//	}
	//	_prvi[lokacija] = new T1(br1);
	//	_drugi[lokacija] = new T2(br2);
	//	(_trenutno)++;
	//	return *this;
	//}


	KolekcijaParova DodajNaPoziciju(int lokacija, T1 el1, T2 el2) {
		KolekcijaParova kolekcija1;

		for (int i = 0; i < _trenutno; i++)
		{
			if (i == lokacija) {
				kolekcija1.Dodaj(el1, el2);
			}
			kolekcija1.Dodaj(*_prvi[i], *_drugi[i]);
		}
		return kolekcija1;
	}


	KolekcijaParova* UkloniRaspon(const int lokacija, const int brEl) {
		if (lokacija<0 || (lokacija + brEl)>GetTrenutno())
			throw exception("Nije validan range.");
		KolekcijaParova* obrisani = new KolekcijaParova();
		KolekcijaParova sacuvani{};
		for (int i = 0; i < GetTrenutno(); i++)
		{
			if (i >= lokacija && i < (lokacija + brEl))
				obrisani->Dodaj(GetPrvi(i), GetDrugi(i));
			else
				sacuvani.Dodaj(GetPrvi(i), GetDrugi(i));
		}
		*this = sacuvani;
		return obrisani;
	}

	~KolekcijaParova() {
		for (int i = 0; i < _trenutno; i++) {
			delete _prvi[i]; _prvi[i] = nullptr;
			delete _drugi[i]; _drugi[i] = nullptr;
		}
	}

	int GetTrenutno() const { return _trenutno; }
	const T1& GetPrvi(int indeks)const { return *_prvi[indeks]; }
	const T2& GetDrugi(int indeks)const { return *_drugi[indeks]; }
	T1& operator[](int indeks)const { return *_prvi[indeks]; }

	friend ostream& operator<<(ostream& COUT, const KolekcijaParova& obj) {
		for (int i = 0; i < obj._trenutno; i++)
			COUT << obj.GetPrvi(i) << " " << obj.GetDrugi(i) << "\n";
		return COUT;
	}
};
class DatumVrijeme {
	int* _godina, * _mjesec, * _dan, * _sati, * _minute, * _sekunde;
public:
	DatumVrijeme(int dan = 1, int mjesec = 1, int godina = 2000, int
		sati = 0, int minute = 0, int sekunde = 0) {
		_godina = new int(godina);
		_mjesec = new int(mjesec);
		_dan = new int(dan);
		_sati = new int(sati);
		_minute = new int(minute);
		_sekunde = new int(sekunde);
	}
	DatumVrijeme(const DatumVrijeme& obj) {
		_godina = new int(*obj._godina);
		_mjesec = new int(*obj._mjesec);
		_dan = new int(*obj._dan);
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);
	}
	DatumVrijeme& operator=(const DatumVrijeme& obj) {
		if (this == &obj)
			return *this;

		delete _godina; delete _mjesec; delete _dan;
		delete _sati; delete _minute; delete _sekunde;

		_godina = new int(*obj._godina);
		_mjesec = new int(*obj._mjesec);
		_dan = new int(*obj._dan);
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
		_sekunde = new int(*obj._sekunde);
		return *this;
	}
	~DatumVrijeme() {
		delete _godina; delete _mjesec; delete _dan;
		_godina = nullptr; _mjesec = nullptr; _dan = nullptr;
		delete _sati; delete _minute; delete _sekunde;
		_sati = nullptr; _minute = nullptr; _sekunde = nullptr;
	};

	//string ToString()const {

	//	return ts.str();
	//}

	friend ostream& operator << (ostream& COUT, const DatumVrijeme& obj) {
		COUT << *obj._dan << "." << *obj._mjesec << "." << *obj._godina << " " << *obj._sati << ":" << *obj._minute << ":" << *obj._sekunde;
		return COUT;
	}

	bool operator==(const DatumVrijeme& obj)const {
		return *obj._godina == *_godina &&
			*obj._mjesec == *_mjesec &&
			*obj._dan == *_dan &&
			*obj._sati == *_sati &&

class Artikal {
	char* _naziv;
	Kategorija _kategorija;
	int _cijena;
public:
	Artikal(const char* naziv = "", Kategorija kategorija = ELEKTRONIKA, int cijena = 0) : _kategorija(kategorija), _cijena(cijena) {
		_naziv = AlocirajTekst(naziv);
	}

	Artikal(const Artikal& obj) {
		_naziv = AlocirajTekst(obj.GetNaziv());
		_kategorija = obj.GetKategorija();
		_cijena = obj.GetCijena();
	}

	Artikal& operator=(const Artikal& obj) {
		if (this == &obj)
			return *this;

		delete[] _naziv;

		_naziv = AlocirajTekst(obj.GetNaziv());
		_kategorija = obj.GetKategorija();
		_cijena = obj.GetCijena();
		return *this;
	}

	~Artikal() { 
		delete[] _naziv;
		_naziv = nullptr;
	}

	const char* GetNaziv() const { return _naziv; }
	Kategorija GetKategorija() const { return _kategorija; }
	int GetCijena() const { return _cijena; }

	friend ostream& operator<<(ostream& COUT, const Artikal& p) {
		COUT << p._naziv << " " << KategorijaNazivi[(int)p._kategorija] << " " << p._cijena;
		return COUT;
	}

	bool operator==(const Artikal& obj)const {
		return strcmp(_naziv, obj._naziv) == 0 &&
			_cijena == obj._cijena &&
			_kategorija == obj._kategorija;
	}
};

	}

		for (int i = 0; i < _kupci.size(); i++)

			if (strcmp(sifraKupca, _kupci[i].GetSifra()) == 0) {

				return _kupci[i].DodajTransakciju(kupovina);

	KolekcijaParova<Kupac, int, 50> PotrosnjaPoKategoriji(Kategorija kategorija) {

		KolekcijaParova<Kupac, int, 50> potrosnja;

		// proci kroz sve kupce

		for (int i = 0; i < _kupci.size(); i++)
		{

			int ukupnaPotrosnja = 0;
			bool imaKupovinu = false;

			for (int j = 0; j < _kupci[i].GetTransakcije().size(); j++)
			{

				Kupovina* castedK = dynamic_cast<Kupovina*>(_kupci[i].GetTransakcije()[j]);
				Povrat* castedP = dynamic_cast<Povrat*>(_kupci[i].GetTransakcije()[j]);


				if (castedK) {


					for (int k = 0; k < castedK->GetProizvodi().size(); k++)
					{

						if (castedK->GetProizvodi()[k].GetKategorija() == kategorija) {

							ukupnaPotrosnja += castedK->GetProizvodi()[k].GetCijena();
							imaKupovinu = true;

						}


					}



				}
				else {

					for (int k = 0; k < castedP->GetProizvodi().size(); k++)
					{

						if (castedP->GetProizvodi()[k].GetKategorija() == kategorija) {

							ukupnaPotrosnja -= castedP->GetProizvodi()[k].GetCijena();

						}


					}


				}




			}


			if (imaKupovinu) {

				potrosnja.Dodaj(_kupci[i], ukupnaPotrosnja);

			}



		}


		return potrosnja;


	}




	friend ostream& operator << (ostream& COUT, const WebShop& obj) {
		COUT << "Naziv webshopa: "<<obj._naziv << endl;

		for (int i = 0; i < obj._kupci.size(); i++)
		{
			COUT << obj._kupci[i] << endl;
		}

		return COUT;
	}


	const char* GetNaziv() const { return _naziv; }
	vector<Kupac>& GetKupci() { return _kupci; }
};

bool UcitajPodatke(string nazivFajla, vector<WebShop>& prodavnice) {

	ifstream file(nazivFajla);

	if (!file.is_open()) {
		return false;
	}


	string eachRow;

	while (getline(file, eachRow)) {

		int pozicijaCrtice = eachRow.find('|');

		string imePrezime = eachRow.substr(0, pozicijaCrtice);

		string nazivProdavnice = eachRow.substr(pozicijaCrtice + 1);

		WebShop novaProdavnica(nazivProdavnice.c_str());

		bool prodavnicaExists = false;

		for (int i = 0; i < prodavnice.size(); i++) {
			if (novaProdavnica == prodavnice[i]) {
				prodavnicaExists = true;
			}
		}
		if (!prodavnicaExists) {
			prodavnice.push_back(novaProdavnica);
		}

		Kupac noviKupac(imePrezime.c_str());

		bool kupacExists = false;

		for (int i = 0; i < prodavnice.size(); i++) {

			if (prodavnice[i] == novaProdavnica) {

				for (int j = 0; j < prodavnice[i].GetKupci().size(); j++) {

					if (prodavnice[i].GetKupci()[j] == noviKupac) {

						kupacExists = true;

					}
				}
				if (!kupacExists) {

					prodavnice[i].DodajKupca(noviKupac);

				}
			}
		}

	}

	return true;



}
