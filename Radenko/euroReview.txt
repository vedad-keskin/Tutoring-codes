char* GetNizKaraktera(const char* sadrzaj, bool dealociraj = false) {
	if (sadrzaj == nullptr)return nullptr;

string GenerisiID(const string& imePrez, int broj) {
	string sifra{};

	const char ime = imePrez[0];
	const char prez = imePrez.find(' ') + 1;

	sifra += toupper(ime);

	if (broj < 10)
		sifra += "000";
	else if (broj < 100)
		sifra += "00";
	else if (broj < 1000)
		sifra += "0";

	string obrnutBroj = "";
	if (broj == 0) obrnutBroj = "0";
	while (broj > 0) {
		obrnutBroj += to_string(broj % 10);
		broj /= 10;
	}

	sifra += toupper(imePrez[prez]);
	sifra += obrnutBroj;
	return sifra;
}

bool ValidirajID(const char* ID) {

	//               D      000M3                        00A41                          0S651                      J8971
	regex pattern("[A-Z](000[A-Z][1-9]|00[A-Z][1-9][0-9]|0[A-Z][0-9]{3}|[A-Z][0-9]{4})");

	return regex_match(ID, pattern);

}




template<class T1, class T2, int max>
class Kolekcija {
	T1 _elementi1[max]{};
	T2 _elementi2[max]{};
	int* _trenutno;
public:
	Kolekcija() {
		_trenutno = new int{ 0 };

	}

	void AddElement(const T1& br1, const T2& br2) {
		if (getTrenutno() == max)
			return;

		_elementi1[getTrenutno()] = br1;
		_elementi2[getTrenutno()] = br2;
		(*_trenutno)++;
	}

	Kolekcija(const Kolekcija& obj) {
		_trenutno = new int(obj.getTrenutno());
		for (int i = 0; i < getTrenutno(); i++)
		{
			_elementi1[i] = obj.getElement1(i);
			_elementi2[i] = obj.getElement2(i);
		}
	}

	Kolekcija& operator=(const Kolekcija& obj) {
		if (this == &obj)
			return *this;

		delete _trenutno; 

		_trenutno = new int(obj.getTrenutno());
		for (int i = 0; i < getTrenutno(); i++)
		{
			_elementi1[i] = obj.getElement1(i);
			_elementi2[i] = obj.getElement2(i);
		}
		return *this;
	}

	Kolekcija& InsertAt(const T1& br1, const T2& br2, const int lokacija) {
		for (int i = getTrenutno(); i > lokacija; i--)
		{
			_elementi1[i] = _elementi1[i - 1];
			_elementi2[i] = _elementi2[i - 1];
		}
		_elementi1[lokacija] = br1;
		_elementi2[lokacija] = br2;
		(*_trenutno)++;
		return *this;
	}

	Kolekcija RemoveRange(const int lokacija, const int brEl) {
		if (lokacija<0 || (lokacija + brEl)>getTrenutno())
			throw exception("Nije validan range");

		Kolekcija obrisani{};
		Kolekcija sacuvani{};
		for (int i = 0; i < getTrenutno(); i++)
		{
			if (i >= lokacija && i < (lokacija + brEl))
				obrisani.AddElement(getElement1(i), getElement2(i));
			else
				sacuvani.AddElement(getElement1(i), getElement2(i));
		}
		*this = sacuvani;
		return obrisani;
	}
	~Kolekcija() {
		delete _trenutno; _trenutno = nullptr;
	}
	const T1& getElement1(int lokacija)const { return _elementi1[lokacija]; }
	const T2& getElement2(int lokacija)const { return _elementi2[lokacija]; }
	int getTrenutno() const { return *_trenutno; }
	friend ostream& operator<< (ostream& COUT, Kolekcija& obj) {
		for (size_t i = 0; i < *obj._trenutno; i++)
			COUT << obj.getElement1(i) << " " << obj.getElement2(i) <<
			endl;
		return COUT;
	}
};
class Vrijeme {
	int* _sat, * _minuta, * _sekunda;
public:
	Vrijeme(int sat = 10, int minuta = 0, int sekunda = 0) {
		_sat = new int(sat);
		_minuta = new int(minuta);
		_sekunda = new int(sekunda);
	}
	Vrijeme(const Vrijeme& obj) {
		_sat = new int(*obj._sat);
		_minuta = new int(*obj._minuta);
		_sekunda = new int(*obj._sekunda);
	}
	Vrijeme& operator=(const Vrijeme& obj) {
		if (this == &obj)
			return *this;

		delete _sat; 
		delete _minuta; 
		delete _sekunda;

		_sat = new int(*obj._sat);
		_minuta = new int(*obj._minuta);
		_sekunda = new int(*obj._sekunda);
		return *this;
	}

	bool operator == (const Vrijeme& obj) {

		return *_sat == *obj._sat && *_minuta == *obj._minuta && *_sekunda == *obj._sekunda;

	}

	~Vrijeme() {
		delete _sat; _sat = nullptr;
		delete _minuta; _minuta = nullptr;
		delete _sekunda; _sekunda = nullptr;
	}
	friend ostream& operator<< (ostream& COUT, const Vrijeme& obj) {
		COUT << *obj._sat << ":" << *obj._minuta << ":" << *obj._sekunda;
		return COUT;
	}
};
class Pogodak {
	Vrijeme* _vrijemePogotka;
	char* _napomena;
public:
	Pogodak(Vrijeme vrijeme, const char* napomena) {
		_napomena = GetNizKaraktera(napomena);
		_vrijemePogotka = new Vrijeme(vrijeme);
	}

	Pogodak(const Pogodak& obj) {
		_napomena = GetNizKaraktera(obj._napomena);
		_vrijemePogotka = new Vrijeme(*obj._vrijemePogotka);
	}

	Pogodak& operator=(const Pogodak& obj) {
		if (this == &obj)
			return *this;

		delete[] _napomena; 
		delete _vrijemePogotka;


		_napomena = GetNizKaraktera(obj._napomena);
		_vrijemePogotka = new Vrijeme(*obj._vrijemePogotka);
		return *this;
	}

	bool operator == (const Pogodak& obj) {

		return strcmp(_napomena, obj._napomena) == 0 && *_vrijemePogotka == *obj._vrijemePogotka;

	}

	~Pogodak() {
		delete[] _napomena; _napomena = nullptr;
		delete _vrijemePogotka; _vrijemePogotka = nullptr;
	}
	Vrijeme GetVrijemePogotka() { return *_vrijemePogotka; }
	char* GetNapomena() { return _napomena; }

	friend ostream& operator<< (ostream& COUT, const Pogodak& obj) {
		COUT << *obj._vrijemePogotka << " -> " << obj._napomena;
		return COUT;
	}
};
class Igrac {
	static int _id;
	char* _ID; // za inicijalizaciju _ID-a iskoristiti funkciju GenerisiID tj.atributa _imePrezime(inicijala imena i prezimena) i statickog clana _id
	char* _imePrezime;
	vector<Pogodak*> _pogoci;
public:
	Igrac(const char* imePrezime) {
		_imePrezime = GetNizKaraktera(imePrezime);
		_ID = GetNizKaraktera(GenerisiID(imePrezime, _id).c_str());
		_id++;
	}

	Igrac(const Igrac& obj) {
		_imePrezime = GetNizKaraktera(obj._imePrezime);
		_ID = GetNizKaraktera(obj._ID);

		for (Pogodak* pogodak : obj._pogoci)
		{
			_pogoci.push_back(new Pogodak{ *pogodak });
		}
	}
	Igrac& operator=(const Igrac& obj) {
		if (this == &obj)
			return *this;

		delete[]_ID; 
		delete[]_imePrezime; 

		for (size_t i = 0; i < _pogoci.size(); i++) {
			delete _pogoci[i];
			
		}

		_pogoci.clear();

		_imePrezime = GetNizKaraktera(obj._imePrezime);
		_ID = GetNizKaraktera(obj._ID);

		for (Pogodak* pogodak : obj._pogoci)
		{
			_pogoci.push_back(new Pogodak{ *pogodak });
		}
		return *this;
	}

	bool DodajPogodak(Pogodak pogodak) {

		for (int i = 0; i < _pogoci.size(); i++)
		{
			if (pogodak == *_pogoci[i]) {
				return false;
			}
		}

		_pogoci.push_back(new Pogodak(pogodak));

		return true;

	}

	bool operator == (const Igrac& obj) {
		return strcmp(_ID, obj._ID) == 0;
	}

	~Igrac() {
		delete[]_ID; _ID = nullptr;
		delete[]_imePrezime; _imePrezime = nullptr;
		for (size_t i = 0; i < _pogoci.size(); i++) {
			delete _pogoci[i];
			_pogoci[i] = nullptr;
		}
	}
	const char* GetImePrezime()const { return _imePrezime; }
	const char* GetID()const { return _ID; }
	vector<Pogodak*>& GetPogoci() { return _pogoci; }

	friend ostream& operator<< (ostream& COUT, Igrac& obj) {
		COUT << obj._ID << " -> " << obj._imePrezime;
		for (size_t i = 0; i < obj._pogoci.size(); i++)
			cout << *obj._pogoci[i] << endl;
		return COUT;
	}
};
