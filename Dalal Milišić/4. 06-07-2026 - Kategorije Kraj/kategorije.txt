int GetTrenutnoGodina() {
	time_t trenutnoVrijeme = time(0);
	tm lokalnovrijeme;
	localtime_s(&lokalnovrijeme, &trenutnoVrijeme);
	return lokalnovrijeme.tm_year + 1900;
}
string GenerisiSifru(string imePrezime, int redniBroj) {
	char PrvoSlovoImena = imePrezime[0];
	char PrvoSlovoPrezimena = imePrezime[imePrezime.find(" ") + 1];
	string novi;
	novi += PrvoSlovoImena;
	novi += PrvoSlovoPrezimena;
	novi += ":";
	novi += to_string(GetTrenutnoGodina());
	novi += "-";
	if (redniBroj < 10) {
		novi += "00";
	}
	else if (redniBroj < 100) {
		novi += "0";
	}
	novi += to_string(redniBroj);

	return novi;
}

bool ValidirajSifru(const string& sifra) {
	regex pravilo{ "[A-Z]{2}[:][0-9]{4}[-][0-9]{3}" };
	return regex_match(sifra, pravilo);
}

template<class T1, class T2, int max>
class KolekcijaParova {
	T1* _prvi[max] = { nullptr };
	T2* _drugi[max] = { nullptr };
	int _trenutno = 0;
public:
	KolekcijaParova() = default;

	~KolekcijaParova() {
		for (int i = 0; i < _trenutno; i++) {
			delete _prvi[i]; _prvi[i] = nullptr;
			delete _drugi[i]; _drugi[i] = nullptr;
		}
	}

	KolekcijaParova(const KolekcijaParova& obj) {
		_trenutno = obj.GetTrenutno();

		for (int i = 0; i < obj.GetTrenutno(); i++) {
			_prvi[i] = new T1{ obj.GetPrvi(i) };
			_drugi[i] = new T2{ obj.GetDrugi(i) };
		}
	}

	KolekcijaParova& operator=(const KolekcijaParova& obj) {
		if (this != &obj) {
			for (int i = 0; i < _trenutno; i++) {
				delete _prvi[i];
				delete _drugi[i];
			}

			_trenutno = obj.GetTrenutno();

			for (int i = 0; i < obj.GetTrenutno(); i++) {
				_prvi[i] = new T1{ obj.GetPrvi(i) };
				_drugi[i] = new T2{ obj.GetDrugi(i) };
			}
		}return *this;
	}
	void Dodaj(const T1& el1, const T2& el2) {
		if (_trenutno == max) { return; }

		_prvi[_trenutno] = new T1{ el1 };
		_drugi[_trenutno] = new T2{ el2 };
		_trenutno++;
	}
	KolekcijaParova DodajNaPoziciju(int pozicija, T1 el1, T2 el2) {
		if (GetTrenutno() == max || pozicija < 0 || pozicija > GetTrenutno())
		{
			return *this;
		}
		KolekcijaParova prosirena;

		for (int i = 0; i < GetTrenutno(); i++)
		{
			if (pozicija == i)
			{
				prosirena.Dodaj(el1, el2);
			}
			prosirena.Dodaj(GetPrvi(i), GetDrugi(i));
		}

		*this = prosirena;

		return *this;
	}

	// UkloniRaspon - od lokacije definisane prvim parametrom uklanja broj elemenata definisanih drugi parametrom
// (pocevsi od lokacije 2 ukloni 3 elementa), a vraca pokazivac na kolekciju parova s uklonjenim elementima
	//KolekcijaParova<int, string, 20>* uklonjeniProizvodi =
	//	prosirenaLista.UkloniRaspon(2, 3);
	//cout << "Uklonjeni:\n" << *uklonjeniProizvodi << crt;

	KolekcijaParova* UkloniRaspon(int pocetak, int broj) {

		if (pocetak >= _trenutno || pocetak < 0) {
			throw exception("ne ispravan opseg");
		}

		int kraj = pocetak + broj;
		if (kraj >= _trenutno) {
			throw exception("ne isparavan opseg");
		}


		KolekcijaParova* uklonjeni = new KolekcijaParova();

		KolekcijaParova preostali;

		for (int i = 0; i < GetTrenutno(); i++) {

			if (i >= pocetak && i < kraj) {

				uklonjeni->Dodaj(GetPrvi(i), GetDrugi(i));
			}
			else {
				preostali.Dodaj(GetPrvi(i), GetDrugi(i));
			}

		}

		*this = preostali;

		return uklonjeni;
	}


	int GetTrenutno() const { return _trenutno; }
	T1& GetPrvi(int indeks) const { return *_prvi[indeks]; }
	T2& GetDrugi(int indeks) const { return *_drugi[indeks]; }
	T1& operator[](int indeks) const { return *_prvi[indeks]; }

	friend ostream& operator<<(ostream& COUT, const KolekcijaParova& obj) {
		for (int i = 0; i < obj._trenutno; i++)
			COUT << obj.GetPrvi(i) << " " << obj.GetDrugi(i) << "\n";
		return COUT;
	}
};


class DatumVrijeme {
	int* _godina, * _mjesec, * _dan, * _sati, * _minute, * _sekunde;
public:
	DatumVrijeme(int dan = 1, int mjesec = 1, int godina = 2000, int
		sati = 0, int minute = 0, int sekunde = 0) {
		_godina = new int(godina);
		_mjesec = new int(mjesec);
		_dan = new int(dan);
		_sati = new int(sati);
		_minute = new int(minute);
		_sekunde = new int(sekunde);
	}
	DatumVrijeme(const DatumVrijeme& other) {
		_godina = new int(other.getGodina());
		_mjesec = new int(other.getMjesec());
		_dan = new int(other.getDan());
		_sati = new int(other.getSati());
		_minute = new int(other.getMinute());
		_sekunde = new int(other.getSekunde());
	}
	DatumVrijeme& operator=(const DatumVrijeme& other) {
		if (this != &other)
		{
			delete _godina; delete _mjesec; delete _dan;
			delete _sati; delete _minute; delete _sekunde;

			_godina = new int(other.getGodina());
			_mjesec = new int(other.getMjesec());
			_dan = new int(other.getDan());
			_sati = new int(other.getSati());
			_minute = new int(other.getMinute());
			_sekunde = new int(other.getSekunde());
		}
		return *this;
	}
	int getGodina() const { return *_godina; }
	int getMjesec() const { return *_mjesec; }
	int getDan() const { return *_dan; }
	int getSati() const { return *_sati; }
	int getMinute() const { return *_minute; }
	int getSekunde() const { return *_sekunde; }

	~DatumVrijeme() {
		delete _godina; delete _mjesec; delete _dan;
		delete _sati; delete _minute; delete _sekunde;
		_godina = nullptr; _mjesec = nullptr; _dan = nullptr;
		_sati = nullptr; _minute = nullptr; _sekunde = nullptr;
	}

	bool operator==(const DatumVrijeme& obj)const {
		return getGodina() == obj.getGodina() && getMjesec() == obj.getMjesec() && getDan() == obj.getDan()
			&& getSati() == obj.getSati() && getMinute() == obj.getMinute() && getSekunde() == obj.getSekunde();
	}


	friend ostream& operator << (ostream& COUT, const DatumVrijeme& obj) {
		COUT << setw(2) << setfill('0') << *obj._dan << "." << setw(2) << setfill('0') << *obj._mjesec << "." << *obj._godina << " " << setw(2) << setfill('0') << *obj._sati << ":" << setw(2) << setfill('0') << *obj._minute << ":" << setw(2) << setfill('0') << *obj._sekunde;
		return COUT;
	}

};
class Proizvod {
	char* _naziv;
	Kategorija _kategorija;
	int _cijena;
public:
	Proizvod(const char* naziv = "", Kategorija kategorija = ELEKTRONIKA, int cijena = 0) : _kategorija(kategorija), _cijena(cijena) {
		_naziv = AlocirajTekst(naziv);
	}

	~Proizvod() {
		delete[] _naziv;
		_naziv = nullptr;
	}

	const char* GetNaziv() const { return _naziv; }
	Kategorija GetKategorija() const { return _kategorija; }
	int GetCijena() const { return _cijena; }

	friend ostream& operator<<(ostream& COUT, const Proizvod& p) {
		COUT << p._naziv << " " << KategorijaNazivi[(int)p._kategorija] << " " << p._cijena;
		return COUT;
	}

	bool operator == (const Proizvod& obj) const {

		return strcmp(_naziv, obj._naziv) == 0 && _kategorija == obj._kategorija && _cijena == obj._cijena;

	}

	Proizvod(const Proizvod& obj) {
		_naziv = AlocirajTekst(obj.GetNaziv());
		_kategorija = obj.GetKategorija();
		_cijena = obj.GetCijena();
	}

	Proizvod& operator=(const Proizvod& obj) {
		if (this != &obj)
		{
			delete[] _naziv;

			_naziv = AlocirajTekst(obj.GetNaziv());
			_kategorija = obj.GetKategorija();
			_cijena = obj.GetCijena();
		}
		return *this;
	}
};

	}

	Transakcija(const Transakcija& obj) {

		_vrijemeRealizacije = obj._vrijemeRealizacije;

		if (this != &obj) {

			_vrijemeRealizacije = obj._vrijemeRealizacije;

class Prodavnica {
	char* _naziv;
	vector<Kupac> _kupci;
public:
	Prodavnica(const char* naziv) {
		_naziv = AlocirajTekst(naziv);
	}

	Prodavnica(const Prodavnica& obj) {
		_naziv = AlocirajTekst(obj._naziv);
		_kupci = obj._kupci;
	}

	Prodavnica& operator = (const Prodavnica& obj) {
		if (this != &obj) {
			delete[] _naziv;

			_naziv = AlocirajTekst(obj._naziv);
			_kupci = obj._kupci;
		}
		return *this;
	}

	void DodajKupca(Kupac kupac) {

		for (int i = 0; i < _kupci.size(); i++)
		{
			if (kupac == _kupci[i]) {

				throw exception("Kupac je vec dio prodavnice");

			}
		}


		_kupci.push_back(kupac);

	}

	//registraciju transakcije, pored direktnog nacina - preko kupca,
	//je moguce realizovati i u okviru odredjene prodavnice koja ce pronaci kupca
	//po sifri i dodati mu transakciju proslijedjenu kao parametar.u zavisnosti od rezultata
	//izvrsenja metoda vraca true ili false (i dalje vaze pravila vezana za dodavanje transakcije)
	//if (!tehnika.RegistrujTransakcijuKupcu(amina.GetSifra(), kupovinaSlusalice))

	bool RegistrujTransakcijuKupcu(const char* sifraKupca, Kupovina kupovina) {

		for (int i = 0; i < _kupci.size(); i++)
		{

			if (strcmp(sifraKupca , _kupci[i].GetSifra()) == 0) {

				return _kupci[i].DodajTransakciju(kupovina);

			}

		}

		return false;


	}

	//vraca listu svih kupaca iz prodavnice koji su imali najmanje jednu transakciju u proslijedjenoj kategoriji,
	//te koliko su ukupno potrosili na proizvode iz te kategorije (ukupna potrosnja za kategoriju se
	// izracunava: kupovina - povrat)
	//KolekcijaParova<Kupac, int, 50> potrosnja = tehnika.PotrosnjaPoKategoriji(ELEKTRONIKA);


	KolekcijaParova<Kupac, int, 50> PotrosnjaPoKategoriji(Kategorija kategorija) {

		KolekcijaParova<Kupac, int, 50> potrosnja;

		// proci kroz sve kupce

		for (int i = 0; i < _kupci.size(); i++)
		{

			bool imaTransakciju = false;
			int ukupnaPotrosnja = 0; // kupovina - povrat

			// proci kroz sve transakcije tog kupca

			for (int j = 0; j < _kupci[i].GetTransakcije().size(); j++)
			{

				// proci kroz sve proizvode tih transakcija tog kupca

				// TRANSAKCIJA -> KUPOVINA / POVRAT

				Kupovina* transakcijaPretvorenaUKupovina = dynamic_cast<Kupovina*>(_kupci[i].GetTransakcije()[j]);

				Povrat* transakcijaPretvorenaUPovrat = dynamic_cast<Povrat*>(_kupci[i].GetTransakcije()[j]);


				if (transakcijaPretvorenaUKupovina) {

					// proci kroz sve proizvode te kupovina

					for (int k = 0; k < transakcijaPretvorenaUKupovina->GetProizvodi().size(); k++)
					{
						// amina 
						// 1. tran. -> kupovina -> telefon("Telefon FITPhone", ELEKTRONIKA, 1500),

						// 2. tran -> kupovina -> knjiga 55 

						// 3. tran -> povrat -> knjiga 55


						// teleft -> ELEKTRONIKA == kategorija ELEKTRONIKA
						if (transakcijaPretvorenaUKupovina->GetProizvodi()[k].GetKategorija() == kategorija) {

							ukupnaPotrosnja += transakcijaPretvorenaUKupovina->GetProizvodi()[k].GetCijena(); // 1500 
							imaTransakciju = true; // true -> 

						}

					}


				}
				else if (transakcijaPretvorenaUPovrat) {

					// proci kroz sve proizvode tog povrata

					for (int k = 0; k < transakcijaPretvorenaUPovrat->GetProizvodi().size(); k++)
					{

						if (transakcijaPretvorenaUPovrat->GetProizvodi()[k].GetKategorija() == kategorija) {

							ukupnaPotrosnja -= transakcijaPretvorenaUPovrat->GetProizvodi()[k].GetCijena();



						}

					}


				}


			}


			if (imaTransakciju) {

				potrosnja.Dodaj(_kupci[i], ukupnaPotrosnja);

			}


		}



		return potrosnja;

	}




	~Prodavnica() { 
		delete[] _naziv;
		_naziv = nullptr;
	}
	const char* GetNaziv() const { return _naziv; }
	vector<Kupac>& GetKupci() { return _kupci; }


	friend ostream& operator << (ostream& COUT, const Prodavnica& obj) {
		COUT << obj._naziv << endl;

		for (int i = 0; i < obj._kupci.size(); i++)
		{
			COUT << obj._kupci[i];
		}

		return COUT;
	}

};
