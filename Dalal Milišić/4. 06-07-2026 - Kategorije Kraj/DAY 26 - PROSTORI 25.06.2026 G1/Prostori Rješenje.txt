char* AlocirajTekst(const char* tekst) {
	if (!tekst) return nullptr;
	size_t vel = strlen(tekst) + 1;
	char* temp = new char[vel];
	strcpy_s(temp, vel, tekst);
	return temp;
}

//funkcija generise sifru korisnika na osnovu imena i prezimena, rednog broja i trenutne godine.
//sifra je u formatu CW-IN-BBB/GGGG, gdje IN predstavlja inicijale, BBB redni broj korisnika
//popunjen nulama na slobodnim mjestima, a GGGG trenutnu godinu dobijenu iz sistema.
//funkciju koristiti u konstruktoru klase Korisnik za inicijalizaciju atributa _sifra.

//if (GenerisiSifru("Amar Macic", 15) == "CW-AM-015/2026")
//cout << "Sifra OK" << crt;
//if (GenerisiSifru("Maid Ramic", 156) == "CW-MR-156/2026")
//cout << "Sifra OK" << crt;

string GenerisiSifru(string imePrezime, int redniBroj) {

	string sifra;

	// CW

	sifra += "CW-";

	// IN

	char prvoSlovoImena = toupper(imePrezime[0]);
	char prvoSlovoPrezimena = toupper(imePrezime[imePrezime.find(' ') + 1]);

	sifra += prvoSlovoImena;
	sifra += prvoSlovoPrezimena;
	sifra += "-";

	// BBB

	if (redniBroj < 10)
		sifra += "00";
	else if (redniBroj < 100)
		sifra += "0";

	sifra += to_string(redniBroj);
	sifra += "/";

	// GGGG


	time_t trenutnoVrijeme = time(0);
	tm lokalnoVrijeme;
	localtime_s(&lokalnoVrijeme, &trenutnoVrijeme);
	int godina = lokalnoVrijeme.tm_year + 1900;

	sifra += to_string(godina);


	return sifra;

}


// CW-AM-015/2026
bool ValidirajSifru(const string& sifra) {

	regex pattern("(CW)[-][A-Z]{2}[-][0-9]{3}[/][0-9]{4}");

	return regex_match(sifra, pattern);

}

template<class T1, class T2, int max>
class Kolekcija {
	T1 _prvi[max];
	T2 _drugi[max];
	int* _trenutno;
public:
	Kolekcija() {
		_trenutno = new int(0);
	}

	Kolekcija(const Kolekcija& obj) {

		_trenutno = new int(*obj._trenutno);

		for (int i = 0; i < *_trenutno; i++)
		{
			_prvi[i] = obj._prvi[i];
			_drugi[i] = obj._drugi[i];
		}

	}

	Kolekcija& operator = (const Kolekcija& obj) {
		if (this != &obj) {
			delete _trenutno;

			_trenutno = new int(*obj._trenutno);

			for (int i = 0; i < *_trenutno; i++)
			{
				_prvi[i] = obj._prvi[i];
				_drugi[i] = obj._drugi[i];
			}
		}
		return *this;
	}



	void Dodaj(T1 el1, T2 el2) {


		_prvi[*_trenutno] = el1;
		_drugi[*_trenutno] = el2;

		(*_trenutno)++;

	}

	//DodajNaPoziciju dodaje novi par na lokaciju definisanu prvim parametrom, pomjera postojece
	//elemente udesno i vraca trenutno stanje kolekcije. u slucaju popunjene kolekcije ili
	//neispravne lokacije potrebno je baciti izuzetak.

	//Kolekcija<int, string, 20> prosireniTermini = termini.DodajNaPoziciju(2, 99, "Poseban termin");
	//cout << prosireniTermini << crt;

	Kolekcija DodajNaPoziciju(int lokacija, T1 el1, T2 el2) {

		if (lokacija < 0 || lokacija > *_trenutno) {
			throw exception("Lokacija ne postoji");
		}

		Kolekcija prosireniInventar;

		for (int i = 0; i < *_trenutno; i++)
		{
			if (i == lokacija) {
				prosireniInventar.Dodaj(el1, el2);
			}
			prosireniInventar.Dodaj(_prvi[i], _drugi[i]);
		}


		return prosireniInventar;
	}


	//UkloniRaspon od lokacije definisane prvim parametrom uklanja broj elemenata definisan
	//drugim parametrom, a vraca pokazivac na novu kolekciju koja sadrzi uklonjene elemente.
	//pozivalac je odgovoran za dealokaciju vracene kolekcije.
	//Kolekcija<int, string, 20>* uklonjeniTermini = prosireniTermini.UkloniRaspon(3, 2);
	//cout << "Uklonjeni elementi:" << crt << *uklonjeniTermini;
	//cout << "Preostali elementi:" << crt << prosireniTermini;

	Kolekcija* UkloniRaspon(int lokacijaOd, int brojElemenata) {

		if (lokacijaOd < 0 || lokacijaOd + brojElemenata > *_trenutno) {
			throw exception("Opseg nije ispravan");
		}

		Kolekcija* uklonjeniTermini = new Kolekcija();

		Kolekcija preostaliTermini;

		for (int i = 0; i < *_trenutno; i++)
		{
			if (i >= lokacijaOd && i < lokacijaOd + brojElemenata) {
				uklonjeniTermini->Dodaj(_prvi[i], _drugi[i]);
			}
			else
				preostaliTermini.Dodaj(_prvi[i], _drugi[i]);
		}

		*this = preostaliTermini;

		return uklonjeniTermini;

	}



	~Kolekcija() {
		delete _trenutno;
		_trenutno = nullptr;
	}

	// 1. OPCIJA

	int GetTrenutno() const { return *_trenutno; }
	T1& GetPrvi(int indeks) { return _prvi[indeks]; }
	T2& GetDrugi(int indeks) { return _drugi[indeks]; }
	T1& operator[](int indeks) const { return _prvi[indeks]; }

	friend ostream& operator<<(ostream& COUT, Kolekcija& obj) {
		for (int i = 0; i < obj.GetTrenutno(); i++)
			COUT << obj.GetPrvi(i) << " " << obj.GetDrugi(i) << endl;
		return COUT;
	}

	// 2. OPCIJA

	//int GetTrenutno() const { return *_trenutno; }
	//T1 GetPrvi(int indeks) const { return _prvi[indeks]; }
	//T2 GetDrugi(int indeks) const { return _drugi[indeks]; }

	//		COUT << obj.GetPrvi(i) << " " << obj.GetDrugi(i) << endl;
	//	return COUT;
	//}

};

class DatumVrijeme {
	int* _godina, * _mjesec, * _dan, * _sati, * _minute;
public:
	DatumVrijeme(int dan = 1, int mjesec = 1, int godina = 2000, int sati = 0, int minute = 0) {
		_godina = new int(godina);
		_mjesec = new int(mjesec);
		_dan = new int(dan);
		_sati = new int(sati);
		_minute = new int(minute);
	}

	DatumVrijeme(const DatumVrijeme& obj) {
		_godina = new int(*obj._godina);
		_mjesec = new int(*obj._mjesec);
		_dan = new int(*obj._dan);
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
	}

	DatumVrijeme& operator=(const DatumVrijeme& obj) {
		if (this != &obj) {
			delete _godina; delete _mjesec; delete _dan;
			delete _sati; delete _minute;

			_godina = new int(*obj._godina);
			_mjesec = new int(*obj._mjesec);
			_dan = new int(*obj._dan);
			_sati = new int(*obj._sati);
			_minute = new int(*obj._minute);

		}
		return *this;
	}

	bool operator==(const DatumVrijeme& obj) const {
		return *_godina == *obj._godina && *_mjesec == *obj._mjesec && *_dan == *obj._dan
			&& *_sati == *obj._sati && *_minute == *obj._minute;
	}

	bool IstiDan(const DatumVrijeme& obj) const {
		return *_dan == *obj._dan && *_mjesec == *obj._mjesec && *_godina == *obj._godina;
	}

	int MinuteOdPocetkaDana() const {
		return *_sati * 60 + *_minute;
	}



	~DatumVrijeme() {
		delete _godina; delete _mjesec; delete _dan;
		delete _sati; delete _minute;
		_godina = nullptr; _mjesec = nullptr; _dan = nullptr;
		_sati = nullptr; _minute = nullptr;
	}


	friend ostream& operator<<(ostream& COUT, const DatumVrijeme& obj) {
		COUT << setw(2) << setfill('0') << *obj._dan << "." << setw(2) << setfill('0') << *obj._mjesec << "." << *obj._godina << " " << setw(2) << setfill('0') << *obj._sati << ":" << setw(2) << setfill('0') << *obj._minute;
		return COUT;
	}
};

class Rezervacija {
	char* _oznaka;
	TipProstora _tipProstora;
	DatumVrijeme _pocetak;
	int _trajanjeMinuta;
public:
	Rezervacija(const char* oznaka, TipProstora tipProstora, DatumVrijeme pocetak, int trajanjeMinuta) : _tipProstora(tipProstora), _pocetak(pocetak), _trajanjeMinuta(trajanjeMinuta) {
		_oznaka = AlocirajTekst(oznaka);
	}

	Rezervacija(const Rezervacija& obj) {

		_oznaka = AlocirajTekst(obj._oznaka);
		_tipProstora = obj._tipProstora;
		_pocetak = obj._pocetak;
		_trajanjeMinuta = obj._trajanjeMinuta;

	}

	Rezervacija& operator = (const Rezervacija& obj) {
		if (this != &obj) {
			delete[] _oznaka;

			_oznaka = AlocirajTekst(obj._oznaka);
			_tipProstora = obj._tipProstora;
			_pocetak = obj._pocetak;
			_trajanjeMinuta = obj._trajanjeMinuta;

		}
		return *this;
	}

	bool operator == (const Rezervacija& obj) {
		return strcmp(_oznaka, obj._oznaka) == 0 && _tipProstora == obj._tipProstora && _pocetak == obj._pocetak && _trajanjeMinuta == obj._trajanjeMinuta;
	}

	~Rezervacija() {
		delete[] _oznaka;
		_oznaka = nullptr;
	}

	const char* GetOznaka() const { return _oznaka; }
	TipProstora GetTipProstora() const { return _tipProstora; }
	DatumVrijeme& GetPocetak() { return _pocetak; }
	int GetTrajanjeMinuta() const { return _trajanjeMinuta; }

	bool ImaKonfliktSa(Rezervacija drugaRezervacija) {

		if (!_pocetak.IstiDan(drugaRezervacija._pocetak)) {
			return false;
		}

		int pocetak1 = _pocetak.MinuteOdPocetkaDana();
		int kraj1 = pocetak1 + _trajanjeMinuta;

		int pocetak2 = drugaRezervacija._pocetak.MinuteOdPocetkaDana();
		int kraj2 = pocetak2 + drugaRezervacija._trajanjeMinuta;

		// primjer 1: salaA (9:00, 60 min) i studio1 (9:30, 90 min)
		// pocetak1 = 540, kraj1 = 600
		// pocetak2 = 570, kraj2 = 660
		// termini se preklapaju -> vraca true

		// primjer 2: salaA (9:00, 60 min) i radnoMjesto (10:00, 60 min)
		// pocetak1 = 540, kraj1 = 600
		// pocetak2 = 600, kraj2 = 660
		// druga pocinje kad prva zavrsava -> nema konflikta -> vraca false

		// primjer 1

		//salaA:       |==========|          540 -------- 600  (9:00 - 10:00)
		//studio1:          |==============| 570 -------- 660  (9:30 - 11:00)
		//                  ^^^^^^^^^^
		//                  imaju overlap

		// primjer 2

		//salaA:       |===========|            540 -------- 600  (9:00 - 10:00)
		//radnoMjesto:             |==========| 600 -------- 660  (10:00 - 11:00)
		//                         ^
		//                  dodiruju se, ali nema overlapa


		if (pocetak1 < kraj2 && pocetak2 < kraj1) {
			return true;
		}

		return false;
	}

	string ToString() const {

		stringstream ss;

		ss << *this;

		return ss.str();

	}

	//10.09.2026 09:00 SALA-A SALA 60 min

	friend ostream& operator << (ostream& COUT, const Rezervacija& obj) {

		COUT << obj._pocetak << " " << obj._oznaka << " " << obj._tipProstora << " " << obj._trajanjeMinuta << " min";

		return COUT;
	}


};

class Korisnik {
	static int _id;
	char* _sifra;
	char* _imePrezime;
	vector<Rezervacija> _rezervacije;
public:
	Korisnik(const char* imePrezime = "") {
		_imePrezime = AlocirajTekst(imePrezime);
		_sifra = AlocirajTekst(GenerisiSifru(imePrezime, _id).c_str());
		_id++;
	}

	Korisnik(const Korisnik& obj) {

		_imePrezime = AlocirajTekst(obj._imePrezime);
		_sifra = AlocirajTekst(obj._sifra);

		_rezervacije = obj._rezervacije;

	}

	Korisnik& operator = (const Korisnik& obj) {
		if (this != &obj) {
			delete[] _sifra;
			delete[] _imePrezime;

			_imePrezime = AlocirajTekst(obj._imePrezime);
			_sifra = AlocirajTekst(obj._sifra);

			_rezervacije = obj._rezervacije;
		}
		return *this;
	}


	//DodajRezervaciju dodaje rezervaciju korisniku ako se ona ne preklapa sa nekom od ranije
	//dodanih rezervacija i ako ukupno trajanje svih rezervacija korisnika u jednom danu ne
	//prelazi 480 minuta. metoda vraca true ako je rezervacija dodana, u suprotnom vraca false.

	bool DodajRezervaciju(Rezervacija rezervacija) {


		for (int i = 0; i < _rezervacije.size(); i++) {
			if (rezervacija.ImaKonfliktSa(_rezervacije[i])) {
				return false;
			}
		}

		int ukupnoMinuta = 0;

		for (int i = 0; i < _rezervacije.size(); i++) {
			if (rezervacija.GetPocetak().IstiDan(_rezervacije[i].GetPocetak())) {
				ukupnoMinuta += _rezervacije[i].GetTrajanjeMinuta();
			}
		}

		ukupnoMinuta += rezervacija.GetTrajanjeMinuta();

		if (ukupnoMinuta > 480) {
			return false;
		}

		_rezervacije.push_back(rezervacija);

		return true;
	}

	bool operator == (const Korisnik& obj) {
		return strcmp(_sifra, obj._sifra) == 0;
	}

	~Korisnik() {
		delete[] _sifra; _sifra = nullptr;
		delete[] _imePrezime; _imePrezime = nullptr;
	}

	const char* GetSifra() const { return _sifra; }
	const char* GetImePrezime() const { return _imePrezime; }
	vector<Rezervacija>& GetRezervacije() { return _rezervacije; }

	friend ostream& operator<<(ostream& COUT, const Korisnik& obj) {
		COUT << obj._imePrezime << " [" << obj._sifra << "]" << endl;
		for (auto& rezervacija : obj._rezervacije)
			//ToString metoda klase Rezervacija vraca podatke o rezervaciji u formatu:
			//10.09.2026 09:00 SALA-A SALA 60 min
			COUT << " - " << rezervacija.ToString() << endl;
		return COUT;
	}
};

class CentarZaRad {
	char* _naziv;
	vector<Korisnik> _korisnici;
public:
	CentarZaRad(const char* naziv) {
		_naziv = AlocirajTekst(naziv);
	}

	CentarZaRad(const CentarZaRad& obj) {
		_naziv = AlocirajTekst(obj._naziv);
		_korisnici = obj._korisnici;
	}

	CentarZaRad& operator = (const CentarZaRad& obj) {
		if (this != &obj) {
			delete[] _naziv;

			_naziv = AlocirajTekst(obj._naziv);
			_korisnici = obj._korisnici;
		}
		return *this;
	}

	bool operator == (const CentarZaRad& obj) {

		return strcmp(_naziv, obj._naziv) == 0;

	}

	void DodajKorisnika(Korisnik korisnik) {

		for (int i = 0; i < _korisnici.size(); i++)
		{
			if (korisnik == _korisnici[i]) {

				throw exception("Korisnik je vec dio centra za rad");
			}
		}

		_korisnici.push_back(korisnik);

	}

	//RegistrujRezervaciju pronalazi korisnika na osnovu sifre i dodaje mu rezervaciju.
	//i dalje vaze pravila definisana u metodi DodajRezervaciju. metoda vraca true ili false.
	//if (radniKutak.RegistrujRezervaciju(goran.GetSifra(), goranovaSala))
	//	cout << "Rezervacija registrovana" << crt;

	bool RegistrujRezervaciju(const char* sifraKorisnika, Rezervacija rezervacija) {


		for (int i = 0; i < _korisnici.size(); i++)
		{

			if (strcmp(sifraKorisnika, _korisnici[i].GetSifra()) == 0) {

				return _korisnici[i].DodajRezervaciju(rezervacija);

			}

		}

		return false;

	}

	//AktivniKorisnici vraca pokazivace na korisnike koji imaju najmanje onoliko rezervacija
	//koliko je definisano vrijednoscu proslijedjenog parametra.
	//vector<Korisnik*> aktivni = radniKutak.AktivniKorisnici(1);

	vector<Korisnik*> AktivniKorisnici(int brojRezervacija) {

		vector<Korisnik*> aktivni;


		for (int i = 0; i < _korisnici.size(); i++)
		{
			if (_korisnici[i].GetRezervacije().size() >= brojRezervacija) {

				aktivni.push_back(new Korisnik(_korisnici[i]));

			}
		}

		return aktivni;


	}

	//KoristenjePoTipu vraca kolekciju parova (korisnik, broj minuta) za sve korisnike koji
	//imaju najmanje jednu rezervaciju prostora proslijedjenog tipa.
	//Kolekcija<Korisnik, int, 50> koristenjeSala = radniKutak.KoristenjePoTipu(SALA);

	Kolekcija<Korisnik, int, 50> KoristenjePoTipu(TipProstora tip) {

		Kolekcija<Korisnik, int, 50> koristenjeSala;

		for (int i = 0; i < _korisnici.size(); i++)
		{

			int brojMinutaPoTipuProstorije = 0;

			for (int j = 0; j < _korisnici[i].GetRezervacije().size(); j++)
			{

				if (_korisnici[i].GetRezervacije()[j].GetTipProstora() == tip) {

					brojMinutaPoTipuProstorije += _korisnici[i].GetRezervacije()[j].GetTrajanjeMinuta();

				}

			}


			if (brojMinutaPoTipuProstorije >= 1) {
				koristenjeSala.Dodaj(_korisnici[i], brojMinutaPoTipuProstorije);
			}


		}

		return koristenjeSala;


	}



	~CentarZaRad() {
		delete[] _naziv;
		_naziv = nullptr;
	}

	const char* GetNaziv() const { return _naziv; }
	vector<Korisnik>& GetKorisnici() { return _korisnici; }

	friend ostream& operator << (ostream& COUT, const CentarZaRad& obj) {
		COUT << obj._naziv << endl;

		for (int i = 0; i < obj._korisnici.size(); i++)
		{
			COUT << obj._korisnici[i] << endl;
		}

		return COUT;
	}

};

/*
Funkcija UcitajPodatke ucitava podatke o centrima za rad i njihovim korisnicima iz
datoteke cije ime se prosljedjuje kao prvi parametar. Svaka linija je zapisana u formatu:

	naziv centra|ime i prezime korisnika

Za svaki ispravan red potrebno je:

bool UcitajPodatke(string nazivFajla, vector<CentarZaRad>& centriZaRad) {

	ifstream file(nazivFajla);

	bool imaNoviPodatak = false;

	if (!file.is_open()) {
		return false;
	}


	string eachRow;

	while (getline(file, eachRow)) {

		int pozicijaCrtice = eachRow.find('|');
		string imePrezime = eachRow.substr(pozicijaCrtice + 1);
		string nazivCentraZaRad = eachRow.substr(0, pozicijaCrtice);

		CentarZaRad noviCentarZaRad(nazivCentraZaRad.c_str());

		bool centarZaRadExists = false;

		for (int i = 0; i < centriZaRad.size(); i++) {
			if (noviCentarZaRad == centriZaRad[i]) {
				centarZaRadExists = true;
			}
		}
		if (!centarZaRadExists) {
			centriZaRad.push_back(noviCentarZaRad);
			imaNoviPodatak = true;
		}

		Korisnik noviKorisnik(imePrezime.c_str());
		bool korisnikExists = false;
		for (int i = 0; i < centriZaRad.size(); i++) {
			if (centriZaRad[i] == noviCentarZaRad) {
				for (int j = 0; j < centriZaRad[i].GetKorisnici().size(); j++) {
					if (centriZaRad[i].GetKorisnici()[j] == noviKorisnik) {
						korisnikExists = true;

					}
				}
				if (!korisnikExists) {
					centriZaRad[i].DodajKorisnika(noviKorisnik);
					imaNoviPodatak = true;

				}
			}
		}

	}

	if (!imaNoviPodatak) {
		return false;
	}
	else {

		return true;
	}

}
