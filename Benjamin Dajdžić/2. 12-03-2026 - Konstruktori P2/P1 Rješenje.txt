#include <iostream>

// POKAZIVACI

// 1. ALOKACIJA (new)

// -- int(); --> jedan broj 

// -- new char[vel]; --> neki niz/vise vrijednosti 

// 2. DEREFENCIRANJE --

// -- int(); --> jedan broj ---> *_ECTS --> 0 , _ECTS = 0000019A305784C0

// -- new char[vel]; --> neki niz/vise vrijednosti --> ne pisemo * nego pisemo bez nje 

// 3. DEALOKCAIJA / DESTRUKTOR 

// -- int(); --> jedan broj -> delete _ECTS;

// -- new char[vel]; --> neki niz/vise vrijednosti --> delete[] _naziv;



class Predmet {
public:				// 
	char* _naziv;	//  "Programiranje II",      
	//char _naziv[50]; 
	int _ocjena;	// 9, 
	int* _ECTS;		//   7


	// dft. constr.

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char tekst1[] = "Nema predmeta";
		//string tekst2 = "Nema predmeta";
		const char* tekst3 = "Nema predmeta"; // string literal --> read-only memorija

		int vel = strlen(tekst3) + 1; // 14 za null terminator \0

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, tekst3); // 1. gdje zelimo nesto pohraniti , 2. koliko slova kopiramo, 3. sta kopiramo


	}


	// user-def constr.

	//Predmet predmet1("Programiranje II", 9, 7);

	//            
	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);

		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv);


	}

	// copy constr.

	//Predmet predmetKopija2(predmet1); // copy constr.

	// sa & --> salje original --> imamo njegovu adresu 
	// bez & --> saljes kopiju 
	// const --> kontantna / ne moze se mijenjati 

	Predmet(const Predmet& predmet) {

		//predmet._ocjena = 10;

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);


	}


	// operator dodjele
	//predmetOperatorDodjele = predmet2; // <-- operator dodjele

	Predmet& operator = (const Predmet& predmet) {

		if (this != &predmet) {

			delete _ECTS;
			delete[] _naziv;


			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);

		}



		return *this;

	}




	// destruktor 

	// TILDA alt + 1

	~Predmet() {

		//cout << "DESTUKTOR SE POZVAO" << endl;

		delete _ECTS;

		_ECTS = nullptr;

		delete[] _naziv;

		_naziv = nullptr;

	}


	// ispis

	void Ispis() {


		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl << endl;

	}


};





class Student {
	char* _imePrezime; // niz char [ ] 

	Predmet* _polozeni; // da li je ovo ( ) ili [ ] 
	int _trenutno;		// var koja nam kaze koliko trenutno student ima polozenih predmeta
	int _max;			// var nam kaze koliko maksimalno moze student imati polozenih predmeta
public:


	// dft. constr. 

	Student() {

		const char* dftImePrezime = "Nema imena i prezimena";

		int vel = strlen(dftImePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, dftImePrezime);

		_trenutno = 0; // na pocetku niz je prazan i student ima 0 pol predmeta

		_max = 3;

		_polozeni = new Predmet[_max];

	}

	// Student student("Vedad Keskin", predmet1);
	
	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, imeprezime);

		_trenutno = 0;

		_max = max;

		_polozeni = new Predmet[_max];

	}

	Student(const Student& obj) {


		int vel = strlen(obj._imePrezime) + 1;

		_imePrezime = new char[vel];

		strcpy_s(_imePrezime, vel, obj._imePrezime);


		_max = obj._max;

		_trenutno = obj._trenutno;

		_polozeni = new Predmet[_max]; // kada god alociramo niz dobijemo prazan niz


		for (int i = 0; i < _trenutno; i++) // 2 --> 0 1  
		{
			_polozeni[i] = obj._polozeni[i];
		}


	}

	Student& operator = (const Student& obj) {
		if (this != &obj) {


			delete[] _imePrezime;
			delete[] _polozeni;




			int vel = strlen(obj._imePrezime) + 1;

			_imePrezime = new char[vel];

			strcpy_s(_imePrezime, vel, obj._imePrezime);


			_max = obj._max;

			_trenutno = obj._trenutno;

			_polozeni = new Predmet[_max]; // kada god alociramo niz dobijemo prazan niz


			for (int i = 0; i < _trenutno; i++) // 2 --> 0 1  
			{
				_polozeni[i] = obj._polozeni[i];
			}

		}
		return *this;
	}



	~Student() {

		delete[] _imePrezime;
		_imePrezime = nullptr;

		delete[] _polozeni;
		_polozeni = nullptr;

	}

	void Ispis() {


		cout << "Student: " << _imePrezime << endl;
		cout << "=== POLOZENI PREDMETI STUDENTA ===" << endl << endl;


		for (int i = 0; i < _trenutno; i++) // 2 ---> 0  1 
		{

			_polozeni[i].Ispis(); // 0 --> pol[0] = PRII , pol[1] = MATII 

		}

	}
	














	//student1.AddPolozeniPredmet(predmet1); --> PRII 

	void AddPolozeniPredmet(Predmet predmet) {


		// Voditi računa da se ne može prekoračiti kapacitet niza

		if (_trenutno == _max) {

			cout << "Prekoraceno je ogranicenje predmeta kod odabranog studenta" << endl;

			return;

		}


		// Voditi računa da se ne moze dodavati isti predmet (istog naziva)		        


		for (int i = 0; i < _trenutno; i++) // 2 --> 0 , 1
		{

			//         MATII    == MATII
			// ne moze se na ovaj nacin provjeravti char* == char* 

			// za poredjenje nizova karaktera koristmo func. strcmp ( naziv1  , naziv2  ) == 0


			if (strcmp(  predmet._naziv , _polozeni[i]._naziv  ) == 0) {

				cout << "Dupliranje predmeta nije dozvoljeno" << endl;

				return;

			}


		}




		// Dodaje novi polozeni predmet u niz

		_polozeni[_trenutno] = predmet; // pol[0] = PRII , pol[1] = MATII

		_trenutno++; // 0 --> 1 ---> 2 


	}

};
